44  public class FileUtil {

46      private static final Logger LOG = LoggerFactory.getLogger(FileUtil.class);
47  
48      private static final String READ = "Read {}";
49      private static final String READING_FILE = "Reading file {}";
50      private static final String ERROR_READING_FILE = "Error reading file {}. {}";
51      private static final String ERROR_CLOSING_FILE = "Error closing file {}. {}";
52  
53      private FileUtil() {
54      }
55  
56      /**
57       * Read an entire file at one time. Intended primarily for Linux /proc
58       * filesystem to avoid recalculating file contents on iterative reads.
59       *
60       * @param filename
61       *            The file to read
62       *
63       * @return A list of Strings representing each line of the file, or an empty
64       *         list if file could not be read or is empty
65       */
66      public static List<String> readFile(String filename) {
67          return readFile(filename, true);
68      }
69  
70      /**
71       * Read an entire file at one time. Intended primarily for Linux /proc
72       * filesystem to avoid recalculating file contents on iterative reads.
73       *
74       * @param filename
75       *            The file to read
76       * @param reportError
77       *            Whether to log errors reading the file
78       *
79       * @return A list of Strings representing each line of the file, or an empty
80       *         list if file could not be read or is empty
81       */
82      public static List<String> readFile(String filename, boolean reportError) {
83          List<String> lines = new ArrayList<String>();
84          File file = new File(filename);
85          if (file.canRead()) {
86              LOG.debug(READING_FILE, filename);
87              BufferedReader br = null;
88              InputStreamReader isr = null;
89              FileInputStream fis = null;
90              try {
91                  fis = new FileInputStream(file);
92                  isr = new InputStreamReader(fis, ParseUtil.UTF_8);
93                  br = new BufferedReader(isr);
94                  String line;
95                  while ((line = br.readLine()) != null) {
96                      lines.add(line);
97                  }
98              } catch (IOException e) {
99                  if (reportError) {
100                     LOG.error(ERROR_READING_FILE, filename, e.getMessage());
101                 }
102             } finally {
103                 try {
104                     if (fis != null) {
105                         fis.close();
106                     }
107                 } catch (IOException ex) {
108                     LOG.error(ERROR_CLOSING_FILE, filename, ex.getMessage());
109                 }
110                 try {
111                     if (isr != null) {
112                         isr.close();
113                     }
114                 } catch (IOException ex) {
115                     LOG.error(ERROR_CLOSING_FILE, filename, ex.getMessage());
116                 }
117                 try {
118                     if (br != null) {
119                         br.close();
120                     }
121                 } catch (IOException ex) {
122                     LOG.error(ERROR_CLOSING_FILE, filename, ex.getMessage());
123                 }
124             }
125         } else if (reportError) {
126             LOG.warn("File not found or not readable: {}", filename);
127         }
128         return lines;
129     }

131     /**
132      * Read a file and return the long value contained therein. Intended primarily
133      * for Linux /sys filesystem
134      *
135      * @param filename
136      *            The file to read
137      * @return The value contained in the file, if any; otherwise zero
138      */
139     public static long getLongFromFile(String filename) {
140         if (LOG.isDebugEnabled()) {
141             LOG.debug(READING_FILE, filename);
142         }
143         List<String> read = FileUtil.readFile(filename, false);
144         if (!read.isEmpty()) {
145             if (LOG.isTraceEnabled()) {
146                 LOG.trace(READ, read.get(0));
147             }
148             return ParseUtil.parseLongOrDefault(read.get(0), 0L);
149         }
150         return 0L;
151     }

175     /**
176      * Read a file and return the int value contained therein. Intended primarily
177      * for Linux /sys filesystem
178      *
179      * @param filename
180      *            The file to read
181      * @return The value contained in the file, if any; otherwise zero
182      */
183     public static int getIntFromFile(String filename) {
184         if (LOG.isDebugEnabled()) {
185             LOG.debug(READING_FILE, filename);
186         }
187         try {
188             List<String> read = FileUtil.readFile(filename, false);
189             if (!read.isEmpty()) {
190                 if (LOG.isTraceEnabled()) {
191                     LOG.trace(READ, read.get(0));
192                 }
193                 return Integer.parseInt(read.get(0));
194             }
195         } catch (NumberFormatException ex) {
196             LOG.warn("Unable to read value from {}. {}", filename, ex);
197         }
198         return 0;
199     }

201     /**
202      * Read a file and return the String value contained therein. Intended primarily
203      * for Linux /sys filesystem
204      *
205      * @param filename
206      *            The file to read
207      * @return The value contained in the file, if any; otherwise empty string
208      */
209     public static String getStringFromFile(String filename) {
210         if (LOG.isDebugEnabled()) {
211             LOG.debug(READING_FILE, filename);
212         }
213         List<String> read = FileUtil.readFile(filename, false);
214         if (!read.isEmpty()) {
215             if (LOG.isTraceEnabled()) {
216                 LOG.trace(READ, read.get(0));
217             }
218             return read.get(0);
219         }
220         return "";
221     }

223     /**
224      * Read a file and return a map of string keys to string values contained
225      * therein. Intended primarily for Linux /proc/[pid]/io
226      *
227      * @param filename
228      *            The file to read
229      * @param separator
230      *            Characters in each line of the file that separate the key and the
231      *            value
232      * @return The map contained in the file, if any; otherwise empty map
233      */
234     public static Map<String, String> getKeyValueMapFromFile(String filename, String separator) {
235         Map<String, String> map = new HashMap<String, String>();
236         if (LOG.isDebugEnabled()) {
237             LOG.debug(READING_FILE, filename);
238         }
239         List<String> lines = FileUtil.readFile(filename, false);
240         for (String line : lines) {
241             String[] parts = line.split(separator);
242             if (parts.length == 2) {
243                 map.put(parts[0], parts[1].trim());
244             }
245         }
246         return map;
247     }
