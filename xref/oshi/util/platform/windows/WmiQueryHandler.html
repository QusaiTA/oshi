42  public class WmiQueryHandler {
43  
44      private static final Logger LOG = LoggerFactory.getLogger(WmiQueryHandler.class);
45  
46      // Timeout for WMI queries
47      private int wmiTimeout = Wbemcli.WBEM_INFINITE;
48  
49      // Cache failed wmi classes
50      private final Set<String> failedWmiClassNames = new HashSet<String>();
51  
52      // Preferred threading model
53      private int comThreading = Ole32.COINIT_MULTITHREADED;
54  
55      // Track initialization of Security
56      private boolean securityInitialized = false;
57  
58      private static final Class<?>[] EMPTY_CLASS_ARRAY = new Class<?>[0];
59      private static final Object[] EMPTY_OBJECT_ARRAY = new Object[0];
60  
61      // Factory to create this or a subclass
62      private static Class<? extends WmiQueryHandler> customClass = null;
63  
64      /**
65       * Factory method to create an instance of this class. To override this class,
66       * use {@link #setInstanceClass(Class)} to define a sublcass which extends
67       * {@link WmiQueryHandler}.
68       *
69       * @return An instance of this class or a class defined by
70       *         {@link #setInstanceClass(Class)}
71       */
72      public static WmiQueryHandler createInstance() {
73          if (customClass == null) {
74              return new WmiQueryHandler();
75          }
76          try {
77              return customClass.getConstructor(EMPTY_CLASS_ARRAY).newInstance(EMPTY_OBJECT_ARRAY);
78          } catch (NoSuchMethodException e) {
79              LOG.error("Failed to find or access a no-arg constructor for {}", customClass);
80          } catch (SecurityException e) {
81              LOG.error("Failed to find or access a no-arg constructor for {}", customClass);
82          } catch (InstantiationException e) {
83              LOG.error("Failed to create a new instance of {}", customClass);
84          } catch (IllegalAccessException e) {
85              LOG.error("Failed to create a new instance of {}", customClass);
86          } catch (IllegalArgumentException e) {
87              LOG.error("Failed to create a new instance of {}", customClass);
88          } catch (InvocationTargetException e) {
89              LOG.error("Failed to create a new instance of {}", customClass);
90          }
91          return null;
92      }
93  
94      /**
95       * Define a subclass to be instantiated by {@link #createInstance()}. The class
96       * must extend {@link WmiQueryHandler}.
97       *
98       * @param instanceClass
99       *            The class to instantiate with {@link #createInstance()}.
100      */
101     public static void setInstanceClass(Class<? extends WmiQueryHandler> instanceClass) {
102         customClass = instanceClass;
103     }

105     /**
106      * Query WMI for values, with no timeout.
107      *
108      * @param <T>
109      *            The properties enum
110      * @param query
111      *            A WmiQuery object encapsulating the namespace, class, and
112      *            properties
113      * @return a WmiResult object containing the query results, wrapping an EnumMap
114      */
115     public <T extends Enum<T>> WbemcliUtil.WmiResult<T> queryWMI(WbemcliUtil.WmiQuery<T> query) {
116 
117         WbemcliUtil.WmiResult<T> result = WbemcliUtil.INSTANCE.new WmiResult<T>(query.getPropertyEnum());
118         if (failedWmiClassNames.contains(query.getWmiClassName())) {
119             return result;
120         }
121         boolean comInit = false;
122         try {
123             comInit = initCOM();
124             result = query.execute(wmiTimeout);
125         } catch (COMException e) {
126             // Ignore any exceptions with OpenHardwareMonitor
127             if (!WmiUtil.OHM_NAMESPACE.equals(query.getNameSpace())) {
128                 final int hresult = e.getHresult() == null ? -1 : e.getHresult().intValue();
129                 switch (hresult) {
130                 case Wbemcli.WBEM_E_INVALID_NAMESPACE:
131                     LOG.warn("COM exception: Invalid Namespace {}", query.getNameSpace());
132                     break;
133                 case Wbemcli.WBEM_E_INVALID_CLASS:
134                     LOG.warn("COM exception: Invalid Class {}", query.getWmiClassName());
135                     break;
136                 case Wbemcli.WBEM_E_INVALID_QUERY:
137                     LOG.warn("COM exception: Invalid Query: {}", WmiUtil.queryToString(query));
138                     break;
139                 default:
140                     handleComException(query, e);
141                     break;
142                 }
143                 failedWmiClassNames.add(query.getWmiClassName());
144             }
145         } catch (TimeoutException e) {
146             LOG.error("WMI query timed out after {} ms: {}", wmiTimeout, WmiUtil.queryToString(query));
147         }
148         if (comInit) {
149             unInitCOM();
150         }
151         return result;
152     }
153 
154     protected void handleComException(WbemcliUtil.WmiQuery<?> query, COMException ex) {
155         LOG.warn(
156                 "COM exception querying {}, which might not be on your system. Will not attempt to query it again. Error was: {}:",
157                 query.getWmiClassName(), ex.getMessage());
158     }

160     /**
161      * Initializes COM library and sets security to impersonate the local user
162      *
163      * @return True if COM was initialized and needs to be uninitialized, false
164      *         otherwise
165      */
166     public boolean initCOM() {
167         boolean comInit = false;
168         // Step 1: --------------------------------------------------
169         // Initialize COM. ------------------------------------------
170         comInit = initCOM(getComThreading());
171         if (!comInit) {
172             comInit = initCOM(switchComThreading());
173         }
174         // Step 2: --------------------------------------------------
175         // Set general COM security levels --------------------------
176         if (comInit && !isSecurityInitialized()) {
177             WinNT.HRESULT hres = Ole32.INSTANCE.CoInitializeSecurity(null, -1, null, null,
178                     Ole32.RPC_C_AUTHN_LEVEL_DEFAULT, Ole32.RPC_C_IMP_LEVEL_IMPERSONATE, null, Ole32.EOAC_NONE, null);
179             // If security already initialized we get RPC_E_TOO_LATE
180             // This can be safely ignored
181             if (COMUtils.FAILED(hres) && hres.intValue() != WinError.RPC_E_TOO_LATE) {
182                 Ole32.INSTANCE.CoUninitialize();
183                 throw new COMException("Failed to initialize security.", hres);
184             }
185             securityInitialized = true;
186         }
187         return comInit;
188     }
189 
190     protected boolean initCOM(int coInitThreading) {
191         WinNT.HRESULT hres = Ole32.INSTANCE.CoInitializeEx(null, coInitThreading);
192         switch (hres.intValue()) {
193         // Successful local initialization (S_OK) or was already initialized
194         // (S_FALSE) but still needs uninit
195         case COMUtils.S_OK:
196         case COMUtils.S_FALSE:
197             return true;
198         // COM was already initialized with a different threading model
199         case WinError.RPC_E_CHANGED_MODE:
200             return false;
201         // Any other results is impossible
202         default:
203             throw new COMException("Failed to initialize COM library.", hres);
204         }
205     }
206 
207     /**
208      * UnInitializes COM library. This should be called once for every successful
209      * call to initCOM.
210      */
211     public void unInitCOM() {
212         Ole32.INSTANCE.CoUninitialize();
213     }

215     /**
216      * Returns the current threading model for COM initialization, as OSHI is
217      * required to match if an external program has COM initialized already.
218      *
219      * @return The current threading model
220      */
221     public int getComThreading() {
222         return comThreading;
223     }
224 
225     /**
226      * Switches the current threading model for COM initialization, as OSHI is
227      * required to match if an external program has COM initialized already.
228      *
229      * @return The new threading model after switching
230      */
231     public int switchComThreading() {
232         if (comThreading == Ole32.COINIT_APARTMENTTHREADED) {
233             comThreading = Ole32.COINIT_MULTITHREADED;
234         } else {
235             comThreading = Ole32.COINIT_APARTMENTTHREADED;
236         }
237         return comThreading;
238     }
239 
240     /**
241      * Security only needs to be initialized once. This boolean identifies whether
242      * that has happened.
243      *
244      * @return Returns the securityInitialized.
245      */
246     public boolean isSecurityInitialized() {
247         return securityInitialized;
248     }

257         return wmiTimeout;
