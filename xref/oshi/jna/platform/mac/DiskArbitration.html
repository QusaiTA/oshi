24  package oshi.jna.platform.mac;
25  
26  import com.sun.jna.Library;
27  import com.sun.jna.Native;
28  import com.sun.jna.PointerType;
29  
30  import oshi.jna.platform.mac.CoreFoundation.CFAllocatorRef;
31  import oshi.jna.platform.mac.CoreFoundation.CFDictionaryRef;
32  
33  /**
34   * DiskArbitration framework for disk stats. This class should be considered
35   * non-API as it may be removed if/when its code is incorporated into the JNA
36   * project.
37   *
38   * @author widdis[at]gmail[dot]com
39   */
40  public interface DiskArbitration extends Library {
41      DiskArbitration INSTANCE = Native.load("DiskArbitration", DiskArbitration.class);
42  
43      class DASessionRef extends PointerType {
44      }
45  
46      class DADiskRef extends PointerType {
47      }
48  
49      /**
50       * Creates a new session. The caller of this function receives a reference to
51       * the returned object. The caller also implicitly retains the object and is
52       * responsible for releasing it with CFRelease().
53       *
54       * @param allocator
55       *            The allocator object to be used to allocate memory.
56       * @return A reference to a new DASession.
57       */
58      DASessionRef DASessionCreate(CFAllocatorRef allocator);
59  
60      /**
61       * Creates a new disk object. The caller of this function receives a reference
62       * to the returned object. The caller also implicitly retains the object and is
63       * responsible for releasing it with CFRelease().
64       *
65       * @param allocator
66       *            The allocator object to be used to allocate memory.
67       * @param session
68       *            The DASession in which to contact Disk Arbitration.
69       * @param diskName
70       *            he BSD device name.
71       * @return A reference to a new DADisk.
72       */
73      DADiskRef DADiskCreateFromBSDName(CFAllocatorRef allocator, DASessionRef session, String diskName);
74  
75      /**
76       * Creates a new disk object. The caller of this function receives a reference
77       * to the returned object. The caller also implicitly retains the object and is
78       * responsible for releasing it with CFRelease().
79       *
80       * @param allocator
81       *            The allocator object to be used to allocate memory.
82       * @param session
83       *            The DASession in which to contact Disk Arbitration.
84       * @param media
85       *            The I/O Kit media object.
86       * @return A reference to a new DADisk.
87       */
88      DADiskRef DADiskCreateFromIOMedia(CFAllocatorRef allocator, DASessionRef session, int media);
89  
90      /**
91       * Obtains the Disk Arbitration description of the specified disk. This function
92       * will contact Disk Arbitration to acquire the latest description of the
93       * specified disk, unless this function is called on a disk object passed within
94       * the context of a registered callback, in which case the description is
95       * current as of that callback event.
96       *
97       * The caller of this function receives a reference to the returned object. The
98       * caller also implicitly retains the object and is responsible for releasing it
99       * with CFRelease().
100      *
101      * @param disk
102      *            The DADisk for which to obtain the Disk Arbitration description.
103      * @return The disk's Disk Arbitration description.
104      */
105     CFDictionaryRef DADiskCopyDescription(DADiskRef disk);
106 
107     /**
108      * Obtains the BSD device name for the specified disk.
109      *
110      * @param disk
111      *            The DADisk for which to obtain the BSD device name.
112      *
113      * @return The disk's BSD device name.
114      */
115     String DADiskGetBSDName(DADiskRef disk);
116 }
