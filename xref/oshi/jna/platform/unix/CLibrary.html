27  import com.sun.jna.Pointer;
28  import com.sun.jna.Structure;

33  /**
34   * C library with code common to all *nix-based operating systems. This class
35   * should be considered non-API as it may be removed if/when its code is
36   * incorporated into the JNA project.
37   *
38   * @author widdis[at]gmail[dot]com
39   */
40  public interface CLibrary extends Library {
41  
42      /*
43       * For getaddrinfo()
44       */
45      int AI_CANONNAME = 2;
46  
47      /**
48       * Return type for BSD sysctl kern.boottime
49       */
50      @FieldOrder({ "tv_sec", "tv_usec" })
51      class Timeval extends Structure {
52          public long tv_sec; // seconds
53          public long tv_usec; // microseconds
54      }
55  
56      @FieldOrder({ "sa_family", "sa_data" })
57      class Sockaddr extends Structure {
58          public short sa_family;
59          public byte[] sa_data = new byte[14];
60  
61          public static class ByReference extends Sockaddr implements Structure.ByReference {
62          }
63      }
64  
65      @FieldOrder({ "ai_flags", "ai_family", "ai_socktype", "ai_protocol", "ai_addrlen", "ai_addr", "ai_canonname",
66              "ai_next" })
67      class Addrinfo extends Structure {
68          public int ai_flags;
69          public int ai_family;
70          public int ai_socktype;
71          public int ai_protocol;
72          public int ai_addrlen;
73          public Sockaddr.ByReference ai_addr;
74          public String ai_canonname;
75          public ByReference ai_next;
76  
77          public static class ByReference extends Addrinfo implements Structure.ByReference {
78          }
79  
80          public Addrinfo() {
81          }
82  
83          public Addrinfo(Pointer p) {
84              super(p);
85              read();
86          }
87      }
88  
89      /**
90       * The sysctl() function retrieves system information and allows processes with
91       * appropriate privileges to set system information. The information available
92       * from sysctl() consists of integers, strings, and tables.
93       *
94       * The state is described using a "Management Information Base" (MIB) style
95       * name, listed in name, which is a namelen length array of integers.
96       *
97       * The information is copied into the buffer specified by oldp. The size of the
98       * buffer is given by the location specified by oldlenp before the call, and
99       * that location gives the amount of data copied after a successful call and
100      * after a call that returns with the error code ENOMEM. If the amount of data
101      * available is greater than the size of the buffer supplied, the call supplies
102      * as much data as fits in the buffer provided and returns with the error code
103      * ENOMEM. If the old value is not desired, oldp and oldlenp should be set to
104      * NULL.
105      *
106      * The size of the available data can be determined by calling sysctl() with the
107      * NULL argument for oldp. The size of the available data will be returned in
108      * the location pointed to by oldlenp. For some operations, the amount of space
109      * may change often. For these operations, the system attempts to round up so
110      * that the returned size is large enough for a call to return the data shortly
111      * thereafter.
112      *
113      * To set a new value, newp is set to point to a buffer of length newlen from
114      * which the requested value is to be taken. If a new value is not to be set,
115      * newp should be set to NULL and newlen set to 0.
116      *
117      * @param name
118      *            MIB array of integers
119      * @param namelen
120      *            length of the MIB array
121      * @param oldp
122      *            Information retrieved
123      * @param oldlenp
124      *            Size of information retrieved
125      * @param newp
126      *            Information to be written
127      * @param newlen
128      *            Size of information to be written
129      * @return 0 on success; sets errno on failure
130      */
131     int sysctl(int[] name, int namelen, Pointer oldp, IntByReference oldlenp, Pointer newp, int newlen);
132 
133     /**
134      * The sysctlbyname() function accepts an ASCII representation of the name and
135      * internally looks up the integer name vector. Apart from that, it behaves the
136      * same as the standard sysctl() function.
137      *
138      * @param name
139      *            ASCII representation of the MIB name
140      * @param oldp
141      *            Information retrieved
142      * @param oldlenp
143      *            Size of information retrieved
144      * @param newp
145      *            Information to be written
146      * @param newlen
147      *            Size of information to be written
148      * @return 0 on success; sets errno on failure
149      */
150     int sysctlbyname(String name, Pointer oldp, IntByReference oldlenp, Pointer newp, int newlen);
151 
152     /**
153      * The sysctlnametomib() function accepts an ASCII representation of the name,
154      * looks up the integer name vector, and returns the numeric representation in
155      * the mib array pointed to by mibp. The number of elements in the mib array is
156      * given by the location specified by sizep before the call, and that location
157      * gives the number of entries copied after a successful call. The resulting mib
158      * and size may be used in subsequent sysctl() calls to get the data associated
159      * with the requested ASCII name. This interface is intended for use by
160      * applications that want to repeatedly request the same variable (the sysctl()
161      * function runs in about a third the time as the same request made via the
162      * sysctlbyname() function).
163      *
164      * The number of elements in the mib array can be determined by calling
165      * sysctlnametomib() with the NULL argument for mibp.
166      *
167      * The sysctlnametomib() function is also useful for fetching mib prefixes. If
168      * size on input is greater than the number of elements written, the array still
169      * contains the additional elements which may be written programmatically.
170      *
171      * @param name
172      *            ASCII representation of the name
173      * @param mibp
174      *            Integer array containing the corresponding name vector.
175      * @param size
176      *            On input, number of elements in the returned array; on output, the
177      *            number of entries copied.
178      * @return 0 on success; sets errno on failure
179      */
180     int sysctlnametomib(String name, Pointer mibp, IntByReference size);
181 
182     /**
183      * The getloadavg() function returns the number of processes in the system run
184      * queue averaged over various periods of time. Up to nelem samples are
185      * retrieved and assigned to successive elements of loadavg[]. The system
186      * imposes a maximum of 3 samples, representing averages over the last 1, 5, and
187      * 15 minutes, respectively.
188      *
189      * @param loadavg
190      *            An array of doubles which will be filled with the results
191      * @param nelem
192      *            Number of samples to return
193      * @return If the load average was unobtainable, -1 is returned; otherwise, the
194      *         number of samples actually retrieved is returned.
195      * @see <A HREF=
196      *      "https://www.freebsd.org/cgi/man.cgi?query=getloadavg&sektion=3">
197      *      getloadavg(3)</A>
198      */
199     int getloadavg(double[] loadavg, int nelem);
200 
201     /**
202      * Returns the process ID of the calling process. The ID is guaranteed to be
203      * unique and is useful for constructing temporary file names.
204      *
205      * @return the process ID of the calling process.
206      */
207     int getpid();
208 
209     /**
210      * Given node and service, which identify an Internet host and a service,
211      * getaddrinfo() returns one or more addrinfo structures, each of which contains
212      * an Internet address that can be specified in a call to bind(2) or connect(2).
213      *
214      * @param node
215      *            a numerical network address or a network hostname, whose network
216      *            addresses are looked up and resolved.
217      * @param service
218      *            sets the port in each returned address structure.
219      * @param hints
220      *            specifies criteria for selecting the socket address structures
221      *            returned in the list pointed to by res.
222      * @param res
223      *            returned address structure
224      * @return 0 on success; sets errno on failure
225      */
226     int getaddrinfo(String node, String service, Addrinfo hints, PointerByReference res);
227 
228     /**
229      * Frees the memory that was allocated for the dynamically allocated linked list
230      * res.
231      *
232      * @param res
233      *            Pointer to linked list returned by getaddrinfo
234      */
235     void freeaddrinfo(Pointer res);
236 
237     /**
238      * Translates getaddrinfo error codes to a human readable string, suitable for
239      * error reporting.
240      *
241      * @param e
242      *            Error code from getaddrinfo
243      * @return A human-readable version of the error code
244      */
245     String gai_strerror(int e);
246 }
