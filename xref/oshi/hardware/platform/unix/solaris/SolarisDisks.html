52      public static boolean updateDiskStats(HWDiskStore diskStore) {
53          Kstat ksp = KstatUtil.kstatLookup(null, 0, diskStore.getName());
54          if (ksp != null && KstatUtil.kstatRead(ksp)) {
55              KstatIO data = new KstatIO(ksp.ks_data);
56              diskStore.setReads(data.reads);
57              diskStore.setWrites(data.writes);
58              diskStore.setReadBytes(data.nread);
59              diskStore.setWriteBytes(data.nwritten);
60              diskStore.setCurrentQueueLength((long) data.wcnt + data.rcnt);
61              // rtime and snaptime are nanoseconds, convert to millis
62              diskStore.setTransferTime(data.rtime / 1000000L);
63              diskStore.setTimeStamp(ksp.ks_snaptime / 1000000L);
64              return true;
65          } else {
66              return false;
67          }
68      }

70      @Override
71      public HWDiskStore[] getDisks() {
72          // Create map indexed by device name for multiple command reference
73          Map<String, HWDiskStore> diskMap = new HashMap<String, HWDiskStore>();
74          // First, run iostat -er to enumerate disks by name. Sample output:
75          // errors
76          // device,s/w,h/w,trn,tot
77          // cmdk0,0,0,0,0
78          // sd0,0,0,0
79          List<String> disks = ExecutingCommand.runNative("iostat -er");
80  
81          // Create map to correlate disk name with block device mount point for
82          // later use in partition info
83          Map<String, String> deviceMap = new HashMap<String, String>();
84          // Also run iostat -ern to get the same list by mount point. Sample
85          // output:
86          // errors
87          // s/w,h/w,trn,tot,device
88          // 0,0,0,0,c1d0
89          // 0,0,0,0,c1t1d0
90          List<String> mountpoints = ExecutingCommand.runNative("iostat -ern");
91          String disk;
92          for (int i = 0; i < disks.size() && i < mountpoints.size(); i++) {
93              // Map disk
94              disk = disks.get(i);
95              String[] diskSplit = disk.split(",");
96              if (diskSplit.length < 5 || "device".equals(diskSplit[0])) {
97                  continue;
98              }
99              HWDiskStorel#HWDiskStore">HWDiskStore store = new HWDiskStore();
100             store.setName(diskSplit[0]);
101             diskMap.put(diskSplit[0], store);
102             // Map mount
103             String mount = mountpoints.get(i);
104             String[] mountSplit = mount.split(",");
105             if (mountSplit.length < 5 || "device".equals(mountSplit[4])) {
106                 continue;
107             }
108             deviceMap.put(diskSplit[0], mountSplit[4]);
109         }
110 
111         // Create map to correlate disk name with blick device mount point for
112         // later use in partition info
113         Map<String, Integer> majorMap = new HashMap<String, Integer>();
114         // Run lshal, if available, to get block device major (we'll use
115         // partition # for minor)
116         List<String> lshal = ExecutingCommand.runNative("lshal");
117         disk = "";
118         for (String line : lshal) {
119             if (line.startsWith("udi ")) {
120                 String udi = ParseUtil.getSingleQuoteStringValue(line);
121                 disk = udi.substring(udi.lastIndexOf('/') + 1);
122                 continue;
123             }
124             line = line.trim();
125             if (line.startsWith("block.major")) {
126                 majorMap.put(disk, ParseUtil.getFirstIntValue(line));
127             }
128         }
129 
130         // Next, run iostat -Er to get model, etc.
131         disks = ExecutingCommand.runNative("iostat -Er");
132         // We'll use Model if available, otherwise Vendor+Product
133         disk = "";
134         String model = "";
135         String vendor = "";
136         String product = "";
137         String serial = "";
138         long size = 0;
139         for (String line : disks) {
140             // The -r switch enables comma delimited for easy parsing!
141             // No guarantees on which line the results appear so we'll nest
142             // a loop iterating on the comma splits
143             String[] split = line.split(",");
144             for (String keyValue : split) {
145                 keyValue = keyValue.trim();
146                 // If entry is tne name of a disk, this is beginning of new
147                 // output for that disk.
148                 if (diskMap.keySet().contains(keyValue)) {
149                     // First, if we have existing output from previous,
150                     // update
151                     if (!disk.isEmpty()) {
152                         updateStore(diskMap.get(disk), model, vendor, product, serial, size, deviceMap.get(disk),
153                                 MapUtil.getOrDefault(majorMap, disk, 0));
154                     }
155                     // Reset values for next iteration
156                     disk = keyValue;
157                     model = "";
158                     vendor = "";
159                     product = "";
160                     serial = "";
161                     size = 0L;
162                     continue;
163                 }
164                 // Otherwise update variables
165                 if (keyValue.startsWith("Model:")) {
166                     model = keyValue.replace("Model:", "").trim();
167                 } else if (keyValue.startsWith("Serial No:")) {
168                     serial = keyValue.replace("Serial No:", "").trim();
169                 } else if (keyValue.startsWith("Vendor:")) {
170                     vendor = keyValue.replace("Vendor:", "").trim();
171                 } else if (keyValue.startsWith("Product:")) {
172                     product = keyValue.replace("Product:", "").trim();
173                 } else if (keyValue.startsWith("Size:")) {
174                     // Size: 1.23GB <1227563008 bytes>
175                     String[] bytes = keyValue.split("<");
176                     if (bytes.length > 1) {
177                         bytes = ParseUtil.whitespaces.split(bytes[1]);
178                         size = ParseUtil.parseLongOrDefault(bytes[0], 0L);
179                     }
180                 }
181             }
182             // At end of output update last entry
183             if (!disk.isEmpty()) {
184                 updateStore(diskMap.get(disk), model, vendor, product, serial, size, deviceMap.get(disk),
185                         MapUtil.getOrDefault(majorMap, disk, 0));
186             }
187         }
188 
189         // Finally use kstat to get reads/writes
190         // simultaneously populate result array
191         HWDiskStoreDiskStore">HWDiskStore[] results = new HWDiskStore[diskMap.keySet().size()];
192         int index = 0;
193         for (Entry<String, HWDiskStore> entry : diskMap.entrySet()) {
194             updateDiskStats(entry.getValue());
195             results[index++] = entry.getValue();
196         }
197 
198         return results;
199     }

201     /**
202      * Updates the HWDiskStore. If model name is nonempty it is used, otherwise
203      * vendor+product are used for model
204      *
205      * @param store
206      *            A HWDiskStore
207      * @param model
208      *            model name, or empty string if none
209      * @param vendor
210      *            vendor name, or empty string if none
211      * @param product
212      *            product nmae, or empty string if none
213      * @param serial
214      *            serial number, or empty string if none
215      * @param size
216      *            size of the drive in bytes
217      * @param mount
218      *            The mount point of this store, used to fetch partition info
219      * @param major
220      *            The major device number for the partition
221      */
222     private void updateStore(HWDiskStore store, String model, String vendor, String product, String serial, long size,
223             String mount, int major) {
224         store.setModel(model.isEmpty() ? (vendor + " " + product).trim() : model);
225         store.setSerial(serial);
226         store.setSize(size);
227 
228         // Temporary list to hold partitions
229         List<HWPartition> partList = new ArrayList<HWPartition>();
230 
231         // Now grab prtvotc output for partitions
232         // This requires sudo permissions; will result in "permission denied
233         // otherwise" in which case we return empty partition list
234         List<String> prtvotc = ExecutingCommand.runNative("prtvtoc /dev/dsk/" + mount);
235         // Sample output - see man prtvtoc
236         if (prtvotc.size() > 1) {
237             int bytesPerSector = 0;
238             String[] split;
239             // We have a result, parse partition table
240             for (String line : prtvotc) {
241                 // If line starts with asterisk we ignore except for the one
242                 // specifying bytes per sector
243                 if (line.startsWith("*")) {
244                     if (line.endsWith("bytes/sector")) {
245                         split = ParseUtil.whitespaces.split(line);
246                         if (split.length > 0) {
247                             bytesPerSector = ParseUtil.parseIntOrDefault(split[1], 0);
248                         }
249                     }
250                     continue;
251                 }
252                 // If bytes/sector is still 0, these are not real partitions so
253                 // ignore.
254                 if (bytesPerSector == 0) {
255                     continue;
256                 }
257                 // Lines without asterisk have 6 or 7 whitespaces-split values
258                 // representing (last field optional):
259                 // Partition Tag Flags Sector Count Sector Mount
260                 split = ParseUtil.whitespaces.split(line.trim());
261                 // Partition 2 is always the whole disk so we ignore it
262                 if (split.length < 6 || "2".equals(split[0])) {
263                     continue;
264                 }
265                 HWPartitionPartition">HWPartition partition = new HWPartition();
266                 // First field is partition number
267                 partition.setIdentification(mount + "s" + split[0]);
268                 partition.setMajor(major);
269                 partition.setMinor(ParseUtil.parseIntOrDefault(split[0], 0));
270                 // Second field is tag. Parse:
271                 switch (ParseUtil.parseIntOrDefault(split[1], 0)) {
272                 case 0x01:
273                 case 0x18:
274                     partition.setName("boot");
275                     break;
276                 case 0x02:
277                     partition.setName("root");
278                     break;
279                 case 0x03:
280                     partition.setName("swap");
281                     break;
282                 case 0x04:
283                     partition.setName("usr");
284                     break;
285                 case 0x05:
286                     partition.setName("backup");
287                     break;
288                 case 0x06:
289                     partition.setName("stand");
290                     break;
291                 case 0x07:
292                     partition.setName("var");
293                     break;
294                 case 0x08:
295                     partition.setName("home");
296                     break;
297                 case 0x09:
298                     partition.setName("altsctr");
299                     break;
300                 case 0x0a:
301                     partition.setName("cache");
302                     break;
303                 case 0x0b:
304                     partition.setName("reserved");
305                     break;
306                 case 0x0c:
307                     partition.setName("system");
308                     break;
309                 case 0x0e:
310                     partition.setName("public region");
311                     break;
312                 case 0x0f:
313                     partition.setName("private region");
314                     break;
315                 default:
316                     partition.setName("unknown");
317                     break;
318                 }
319                 // Third field is flags.
320                 // First character writable, second is mountable
321                 if (split[2].equals("00")) {
322                     partition.setType("wm");
323                 } else if (split[2].equals("10")) {
324                     partition.setType("rm");
325                 } else if (split[2].equals("01")) {
326                     partition.setType("wu");
327                 } else {
328                     partition.setType("ru");
329                 }
330                 // Fifth field is sector count
331                 partition.setSize(bytesPerSector * ParseUtil.parseLongOrDefault(split[4], 0L));
332                 // Seventh field (if present) is mount point
333                 if (split.length > 6) {
334                     partition.setMountPoint(split[6]);
335                 }
336                 partList.add(partition);
337             }
338             store.setPartitions(partList.toArray(new HWPartition[partList.size()]));
339         }
