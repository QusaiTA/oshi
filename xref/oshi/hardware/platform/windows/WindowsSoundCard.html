26  import java.util.ArrayList;
27  import java.util.HashMap;
28  import java.util.List;
29  import java.util.Map;

31  import com.sun.jna.platform.win32.COM.WbemcliUtil; // NOSONAR squid:S1191
32  
33  import oshi.hardware.common.AbstractSoundCard;
34  import oshi.util.platform.windows.WmiQueryHandler;
35  import oshi.util.platform.windows.WmiUtil;

37  /**
38   * Sound Card data obtained by Wmi Queries for Windows.
39   *
40   * @author : BilalAM
41   */
42  public class WindowsSoundCard extends AbstractSoundCard {
43      enum SoundCardName {
44          MANUFACTURER, NAME;
45      }
46  
47      enum SoundCardKernel {
48          DRIVERPROVIDERNAME, DRIVERNAME, DRIVERVERSION, DEVICENAME;
49      }
50  
51      public WindowsSoundCard(String kernelVersion, String name, String codec) {
52          super(kernelVersion, name, codec);
53      }
54  
55      /**
56       * Short method to build our kernel string.
57       *
58       * @param index
59       *            The row to fetch data.
60       * @param cardKernelQueryResult
61       *            The result to take data from.
62       * @return The kernel string.
63       */
64      private static String getAudioCardKernelVersion(int index,
65              WbemcliUtil.WmiResult<SoundCardKernel> cardKernelQueryResult) {
66          String audioCardKernel;
67          audioCardKernel = WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DRIVERPROVIDERNAME, index) + " "
68                  + WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DEVICENAME, index) + " "
69                  + WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DRIVERNAME, index) + " "
70                  + WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DRIVERVERSION, index);
71          return audioCardKernel;
72      }
73  
74      /**
75       * Does the following :
76       * <ul>
77       * <li>Creates and runs our 'built where clause' query</li>
78       * <li>Then iterates over the key values of our map and compares the key with
79       * the DeviceName attribute returned by our where-clause query.If its a match
80       * then we create our SoundCard object</li>
81       * </ul>
82       * <br>
83       * NOTE : The reason why the codec name is same as the card name is because
84       * windows does not provide the name of the codec chip but sometimes the name of
85       * the card returned is infact the name of the codec chip also. Example :
86       * Realtek ALC887 HD Audio Device
87       *
88       * @return List of sound cards
89       */
90      public static List<WindowsSoundCard> getSoundCards() {
91          // Get the map of device manufacturer by name
92          WmiQueryHandler wmiQueryHandler = WmiQueryHandler.createInstance();
93          Map<String, String> deviceManufacturerMap = getDeviceManufacturerMap(wmiQueryHandler);
94          String driverQuery = createClause(deviceManufacturerMap);
95  
96          WbemcliUtil.WmiQuery<SoundCardKernel> cardKernelQuery = new WbemcliUtil.WmiQuery<SoundCardKernel>(driverQuery,
97                  SoundCardKernel.class);
98          WbemcliUtil.WmiResult<SoundCardKernel> cardKernelQueryResult = wmiQueryHandler.queryWMI(cardKernelQuery);
99  
100         List<WindowsSoundCard> soundCards = new ArrayList<WindowsSoundCard>();
101         for (int i = 0; i < cardKernelQueryResult.getResultCount(); i++) {
102             // If the map has a key that is equal to the value returned by
103             // cardKernelQuery
104             if (deviceManufacturerMap
105                     .containsKey(WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DEVICENAME, i))) {
106                 // then build a sound card by extracting values from
107                 // cardKernelQuery
108                 soundCards.add(new WindowsSoundCard(getAudioCardKernelVersion(i, cardKernelQueryResult),
109                         WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DRIVERPROVIDERNAME, i) + " "
110                                 + WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DEVICENAME, i),
111                         WmiUtil.getString(cardKernelQueryResult, SoundCardKernel.DEVICENAME, i)));
112             }
113         }
114         return soundCards;
115     }
116 
117     /**
118      * Gets the map of device manufacturer by name
119      * 
120      * @param wmiQueryHandler
121      * 
122      * @return THe map
123      */
124     private static Map<String, String> getDeviceManufacturerMap(WmiQueryHandler wmiQueryHandler) {
125         Map<String, String> deviceManufacturerMap = new HashMap<String, String>();
126         WbemcliUtil.WmiQuery<SoundCardName> soundCardQuery = new WbemcliUtil.WmiQuery<SoundCardName>(
127                 "Win32_SoundDevice", SoundCardName.class);
128         WbemcliUtil.WmiResult<SoundCardName> soundCardResult = wmiQueryHandler.queryWMI(soundCardQuery);
129         for (int i = 0; i < soundCardResult.getResultCount(); i++) {
130             deviceManufacturerMap.put(WmiUtil.getString(soundCardResult, SoundCardName.NAME, i),
131                     WmiUtil.getString(soundCardResult, SoundCardName.MANUFACTURER, i));
132         }
133         return deviceManufacturerMap;
134     }
135 
136     /**
137      * Creates our Win32_PnPSignedDevice query with the WHERE clause taking the
138      * attributes from our map.
139      *
140      * @param deviceManufacturerMap
141      *            The map
142      * @return The WHERE clause
143      */
144     private static String createClause(Map<String, String> deviceManufacturerMap) {
145         StringBuilder sb = new StringBuilder("Win32_PnPSignedDriver");
146         boolean first = true;
147         for (String key : deviceManufacturerMap.keySet()) {
148             if (first) {
149                 sb.append(" WHERE");
150                 first = false;
151             } else {
152                 sb.append(" OR");
153             }
154             sb.append(" DeviceName LIKE \"%").append(key).append("%\"");
155         }
156         return sb.toString();
157     }
