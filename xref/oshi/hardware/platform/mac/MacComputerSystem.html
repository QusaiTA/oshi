33  /**
34   * Hardware data obtained by system_profiler
35   *
36   * @author SchiTho1 [at] Securiton AG
37   * @author widdis [at] gmail [dot] com
38   */
39  final class MacComputerSystem extends AbstractComputerSystem {
40  
41      private static final long serialVersionUID = 1L;
42  
43      private static final String APPLE = "Apple Inc.";
44  
45      MacComputerSystem() {
46          init();
47      }
48  
49      private void init() {
50  
51          setManufacturer(APPLE);
52  
53          // $ system_profiler SPHardwareDataType
54          // Hardware:
55          //
56          // Hardware Overview:
57          //
58          // Model Name: MacBook Pro
59          // Model Identifier: MacBookPro8,2
60          // Processor Name: Intel Core i7
61          // Processor Speed: 2.3 GHz
62          // Number of Processors: 1
63          // Total Number of Cores: 4
64          // L2 Cache (per Core): 256 KB
65          // L3 Cache: 8 MB
66          // Memory: 16 GB
67          // Boot ROM Version: MBP81.0047.B2C
68          // SMC Version (system): 1.69f4
69          // Serial Number (system): C02FH4XYCB71
70          // Hardware UUID: D92CE829-65AD-58FA-9C32-88968791B7BD
71          // Sudden Motion Sensor:
72          // State: Enabled
73  
74          String modelName = "";
75          final String modelNameMarker = "Model Name:";
76          String modelIdentifier = "";
77          final String modelIdMarker = "Model Identifier:";
78          String serialNumberSystem = "";
79          final String serialNumMarker = "Serial Number (system):";
80          String smcVersion = "";
81          final String smcMarker = "SMC Version (system):";
82          String bootRomVersion = "";
83          final String bootRomMarker = "Boot ROM Version:";
84  
85          final MacFirmwareare.html#MacFirmware">MacFirmware firmware = new MacFirmware();
86          firmware.setManufacturer(APPLE);
87          firmware.setName("EFI");
88  
89          final MacBaseboardrd.html#MacBaseboard">MacBaseboard baseboard = new MacBaseboard();
90          baseboard.setManufacturer(APPLE);
91          baseboard.setModel("SMC");
92  
93          // Populate name and ID
94          for (final String checkLine : ExecutingCommand.runNative("system_profiler SPHardwareDataType")) {
95              if (checkLine.contains(modelNameMarker)) {
96                  modelName = checkLine.split(modelNameMarker)[1].trim();
97              }
98              if (checkLine.contains(modelIdMarker)) {
99                  modelIdentifier = checkLine.split(modelIdMarker)[1].trim();
100             }
101             if (checkLine.contains(bootRomMarker)) {
102                 bootRomVersion = checkLine.split(bootRomMarker)[1].trim();
103             }
104             if (checkLine.contains(smcMarker)) {
105                 smcVersion = checkLine.split(Pattern.quote(smcMarker))[1].trim();
106             }
107             if (checkLine.contains(serialNumMarker)) {
108                 serialNumberSystem = checkLine.split(Pattern.quote(serialNumMarker))[1].trim();
109             }
110         }
111         // Use name (id) if both available; else either one
112         if (!modelName.isEmpty()) {
113             if (!modelIdentifier.isEmpty()) {
114                 setModel(modelName + " (" + modelIdentifier + ")");
115             } else {
116                 setModel(modelName);
117             }
118         } else {
119             if (!modelIdentifier.isEmpty()) {
120                 setModel(modelIdentifier);
121             }
122         }
123         if (serialNumberSystem.isEmpty()) {
124             serialNumberSystem = getSystemSerialNumber();
125         }
126         setSerialNumber(serialNumberSystem);
127         baseboard.setSerialNumber(serialNumberSystem);
128         if (!smcVersion.isEmpty()) {
129             baseboard.setVersion(smcVersion);
130         }
131         if (bootRomVersion != null && !bootRomVersion.isEmpty()) {
132             firmware.setVersion(bootRomVersion);
133         }
134 
135         setFirmware(firmware);
136         setBaseboard(baseboard);
137     }
138 
139     private String getSystemSerialNumber() {
140         String serialNumber = null;
141         int service = IOKitUtil.getMatchingService("IOPlatformExpertDevice");
142         if (service != 0) {
143             // Fetch the serial number
144             serialNumber = IOKitUtil.getIORegistryStringProperty(service, "IOPlatformSerialNumber");
145             IOKit.INSTANCE.IOObjectRelease(service);
146         }
147         if (serialNumber == null) {
148             serialNumber = "unknown";
149         }
150         return serialNumber;
151     }
