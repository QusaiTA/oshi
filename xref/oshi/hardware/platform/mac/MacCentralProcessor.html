54      private static final Logger LOG = LoggerFactory.getLogger(MacCentralProcessor.class);
55  
56      private static final long BOOTTIME;
57      static {
58          Timeval tv = new Timeval();
59          if (!SysctlUtil.sysctl("kern.boottime", tv) || tv.tv_sec.longValue() == 0L) {
60              // Usually this works. If it doesn't, fall back to text parsing.
61              // Boot time will be the first consecutive string of digits.
62              BOOTTIME = ParseUtil.parseLongOrDefault(
63                      ExecutingCommand.getFirstAnswer("sysctl -n kern.boottime").split(",")[0].replaceAll("\\D", ""),
64                      System.currentTimeMillis() / 1000);
65          } else {
66              // tv now points to a 64-bit timeval structure for boot time.
67              // First 4 bytes are seconds, second 4 bytes are microseconds
68              // (we ignore)
69              BOOTTIME = tv.tv_sec.longValue();
70          }
71      }

73      /**
74       * Create a Processor
75       */
76      public MacCentralProcessor() {
77          super();
78          // Initialize class variables
79          initVars();
80          // Initialize tick arrays
81          initTicks();
82  
83          LOG.debug("Initialized Processor");
84      }
85  
86      private void initVars() {
87          setVendor(SysctlUtil.sysctl("machdep.cpu.vendor", ""));
88          setName(SysctlUtil.sysctl("machdep.cpu.brand_string", ""));
89          setCpu64(SysctlUtil.sysctl("hw.cpu64bit_capable", 0) != 0);
90          int i = SysctlUtil.sysctl("machdep.cpu.stepping", -1);
91          setStepping(i < 0 ? "" : Integer.toString(i));
92          i = SysctlUtil.sysctl("machdep.cpu.model", -1);
93          setModel(i < 0 ? "" : Integer.toString(i));
94          i = SysctlUtil.sysctl("machdep.cpu.family", -1);
95          setFamily(i < 0 ? "" : Integer.toString(i));
96          long processorID = 0L;
97          processorID |= SysctlUtil.sysctl("machdep.cpu.signature", 0);
98          processorID |= (SysctlUtil.sysctl("machdep.cpu.feature_bits", 0L) & 0xffffffff) << 32;
99          setProcessorID(String.format("%016X", processorID));
100     }

102     /**
103      * Updates logical and physical processor counts from sysctl calls
104      */
105     @Override
106     protected void calculateProcessorCounts() {
107         this.logicalProcessorCount = SysctlUtil.sysctl("hw.logicalcpu", 1);
108         this.physicalProcessorCount = SysctlUtil.sysctl("hw.physicalcpu", 1);
109         this.physicalPackageCount = SysctlUtil.sysctl("hw.packages", 1);
110     }

112     /**
113      * {@inheritDoc}
114      */
115     @Override
116     public long[] getSystemCpuLoadTicks() {
117         long[] ticks = new long[TickType.values().length];
118         int machPort = SystemB.INSTANCE.mach_host_self();
119         HostCpuLoadInfo cpuLoadInfo = new HostCpuLoadInfo();
120         if (0 != SystemB.INSTANCE.host_statistics(machPort, SystemB.HOST_CPU_LOAD_INFO, cpuLoadInfo,
121                 new IntByReference(cpuLoadInfo.size()))) {
122             LOG.error("Failed to get System CPU ticks. Error code: {} ", Native.getLastError());
123             return ticks;
124         }
125 
126         ticks[TickType.USER.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_USER];
127         ticks[TickType.NICE.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_NICE];
128         ticks[TickType.SYSTEM.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_SYSTEM];
129         ticks[TickType.IDLE.getIndex()] = cpuLoadInfo.cpu_ticks[SystemB.CPU_STATE_IDLE];
130         // Leave IOWait and IRQ values as 0
131         return ticks;
132     }

134     /**
135      * {@inheritDoc}
136      */
137     @Override
138     public double[] getSystemLoadAverage(int nelem) {
139         if (nelem < 1 || nelem > 3) {
140             throw new IllegalArgumentException("Must include from one to three elements.");
141         }
142         double[] average = new double[nelem];
143         int retval = SystemB.INSTANCE.getloadavg(average, nelem);
144         if (retval < nelem) {
145             for (int i = Math.max(retval, 0); i < average.length; i++) {
146                 average[i] = -1d;
147             }
148         }
149         return average;
150     }
151 
152     /**
153      * {@inheritDoc}
154      */
155     @Override
156     public long[][] getProcessorCpuLoadTicks() {
157         long[][] ticks = new long[this.logicalProcessorCount][TickType.values().length];
158 
159         int machPort = SystemB.INSTANCE.mach_host_self();
160 
161         IntByReference procCount = new IntByReference();
162         PointerByReference procCpuLoadInfo = new PointerByReference();
163         IntByReference procInfoCount = new IntByReference();
164         if (0 != SystemB.INSTANCE.host_processor_info(machPort, SystemB.PROCESSOR_CPU_LOAD_INFO, procCount,
165                 procCpuLoadInfo, procInfoCount)) {
166             LOG.error("Failed to update CPU Load. Error code: {}", Native.getLastError());
167             return ticks;
168         }
169 
170         int[] cpuTicks = procCpuLoadInfo.getValue().getIntArray(0, procInfoCount.getValue());
171         for (int cpu = 0; cpu < procCount.getValue(); cpu++) {
172             int offset = cpu * SystemB.CPU_STATE_MAX;
173             ticks[cpu][TickType.USER.getIndex()] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_USER]);
174             ticks[cpu][TickType.NICE.getIndex()] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_NICE]);
175             ticks[cpu][TickType.SYSTEM.getIndex()] = FormatUtil
176                     .getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_SYSTEM]);
177             ticks[cpu][TickType.IDLE.getIndex()] = FormatUtil.getUnsignedInt(cpuTicks[offset + SystemB.CPU_STATE_IDLE]);
178         }
179         return ticks;
180     }

182     /**
183      * {@inheritDoc}
184      */
185     @Override
186     public long getSystemUptime() {
187         return System.currentTimeMillis() / 1000 - BOOTTIME;
188     }

190     /**
191      * {@inheritDoc}
192      */
193     @Override
194     public long getContextSwitches() {
195         int machPort = SystemB.INSTANCE.mach_host_self();
196         VMMeter vmstats = new VMMeter();
197         if (0 != SystemB.INSTANCE.host_statistics(machPort, SystemB.HOST_VM_INFO, vmstats,
198                 new IntByReference(vmstats.size()))) {
199             LOG.error("Failed to update vmstats. Error code: {}", Native.getLastError());
200             return -1;
201         }
202         return ParseUtil.unsignedIntToLong(vmstats.v_swtch);
203     }
