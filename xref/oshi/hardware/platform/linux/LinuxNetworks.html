24  package oshi.hardware.platform.linux;
25  
26  import oshi.hardware.NetworkIF;
27  import oshi.hardware.common.AbstractNetworks;
28  import oshi.util.FileUtil;
29  
30  /**
31   * @author enrico[dot]bianchi[at]gmail[dot]com
32   */
33  public class LinuxNetworks extends AbstractNetworks {
34  
35      private static final long serialVersionUID = 1L;
36  
37      /**
38       * Updates interface network statistics on the given interface. Statistics
39       * include packets and bytes sent and received, and interface speed.
40       *
41       * @param netIF
42       *            The interface on which to update statistics
43       */
44      public static void updateNetworkStats(NetworkIF netIF) {
45          String txBytesPath = String.format("/sys/class/net/%s/statistics/tx_bytes", netIF.getName());
46          String rxBytesPath = String.format("/sys/class/net/%s/statistics/rx_bytes", netIF.getName());
47          String txPacketsPath = String.format("/sys/class/net/%s/statistics/tx_packets", netIF.getName());
48          String rxPacketsPath = String.format("/sys/class/net/%s/statistics/rx_packets", netIF.getName());
49          String txErrorsPath = String.format("/sys/class/net/%s/statistics/tx_errors", netIF.getName());
50          String rxErrorsPath = String.format("/sys/class/net/%s/statistics/rx_errors", netIF.getName());
51          String speed = String.format("/sys/class/net/%s/speed", netIF.getName());
52  
53          netIF.setTimeStamp(System.currentTimeMillis());
54          netIF.setBytesSent(FileUtil.getUnsignedLongFromFile(txBytesPath));
55          netIF.setBytesRecv(FileUtil.getUnsignedLongFromFile(rxBytesPath));
56          netIF.setPacketsSent(FileUtil.getUnsignedLongFromFile(txPacketsPath));
57          netIF.setPacketsRecv(FileUtil.getUnsignedLongFromFile(rxPacketsPath));
58          netIF.setOutErrors(FileUtil.getUnsignedLongFromFile(txErrorsPath));
59          netIF.setInErrors(FileUtil.getUnsignedLongFromFile(rxErrorsPath));
60          long netSpeed = FileUtil.getUnsignedLongFromFile(speed) * 1024 * 1024;
61          netIF.setSpeed(netSpeed < 0 ? 0 : netSpeed);
62      }
63  }
