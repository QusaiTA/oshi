50      public SolarisOperatingSystem() {
51          this.manufacturer = "Oracle";
52          this.family = "SunOS";
53          this.version = new SolarisOSVersionInfoEx();
54          initBitness();
55      }

57      private void initBitness() {
58          if (this.bitness < 64) {
59              this.bitness = ParseUtil.parseIntOrDefault(ExecutingCommand.getFirstAnswer("isainfo -b"), 32);
60          }
61      }

74      @Override
75      public OSProcess[] getProcesses(int limit, ProcessSort sort, boolean slowFields) {
76          List<OSProcess> procs = getProcessListFromPS(
77                  "ps -eo s,pid,ppid,user,uid,group,gid,nlwp,pri,vsz,rss,etime,time,comm,args", -1, slowFields);
78          List<OSProcess> sorted = processSort(procs, limit, sort);
79          return sorted.toArray(new OSProcess[sorted.size()]);
80      }

87          return getProcess(pid, true);

90      private OSProcess getProcess(int pid, boolean slowFields) {
91          List<OSProcess> procs = getProcessListFromPS(
92                  "ps -o s,pid,ppid,user,uid,group,gid,nlwp,pri,vsz,rss,etime,time,comm,args -p ", pid, slowFields);
93          if (procs.isEmpty()) {
94              return null;
95          }
96          return procs.get(0);
97      }

102     @Override
103     public OSProcess[] getChildProcesses(int parentPid, int limit, ProcessSort sort) {
104         List<OSProcess> procs = getProcessListFromPS(
105                 "ps -eo s,pid,ppid,user,uid,group,gid,nlwp,pri,vsz,rss,etime,time,comm,args --ppid", parentPid, true);
106         List<OSProcess> sorted = processSort(procs, limit, sort);
107         return sorted.toArray(new OSProcess[sorted.size()]);
108     }

110     private List<OSProcess> getProcessListFromPS(String psCommand, int pid, boolean slowFields) {
111         Map<Integer, String> cwdMap = LsofUtil.getCwdMap(pid);
112         List<OSProcess> procs = new ArrayList<OSProcess>();
113         List<String> procList = ExecutingCommand.runNative(psCommand + (pid < 0 ? "" : pid));
114         if (procList.isEmpty() || procList.size() < 2) {
115             return procs;
116         }
117         // remove header row
118         procList.remove(0);
119         // Fill list
120         for (String proc : procList) {
121             String[] split = ParseUtil.whitespaces.split(proc.trim(), 15);
122             // Elements should match ps command order
123             if (split.length < 15) {
124                 continue;
125             }
126             long now = System.currentTimeMillis();
127             OSProcesshtml#OSProcess">OSProcess sproc = new OSProcess();
128             switch (split[0].charAt(0)) {
129             case 'O':
130                 sproc.setState(OSProcess.State.RUNNING);
131                 break;
132             case 'S':
133                 sproc.setState(OSProcess.State.SLEEPING);
134                 break;
135             case 'R':
136             case 'W':
137                 sproc.setState(OSProcess.State.WAITING);
138                 break;
139             case 'Z':
140                 sproc.setState(OSProcess.State.ZOMBIE);
141                 break;
142             case 'T':
143                 sproc.setState(OSProcess.State.STOPPED);
144                 break;
145             default:
146                 sproc.setState(OSProcess.State.OTHER);
147                 break;
148             }
149             sproc.setProcessID(ParseUtil.parseIntOrDefault(split[1], 0));
150             sproc.setParentProcessID(ParseUtil.parseIntOrDefault(split[2], 0));
151             sproc.setUser(split[3]);
152             sproc.setUserID(split[4]);
153             sproc.setGroup(split[5]);
154             sproc.setGroupID(split[6]);
155             sproc.setThreadCount(ParseUtil.parseIntOrDefault(split[7], 0));
156             sproc.setPriority(ParseUtil.parseIntOrDefault(split[8], 0));
157             // These are in KB, multiply
158             sproc.setVirtualSize(ParseUtil.parseLongOrDefault(split[9], 0) * 1024);
159             sproc.setResidentSetSize(ParseUtil.parseLongOrDefault(split[10], 0) * 1024);
160             // Avoid divide by zero for processes up less than a second
161             long elapsedTime = ParseUtil.parseDHMSOrDefault(split[11], 0L);
162             sproc.setUpTime(elapsedTime < 1L ? 1L : elapsedTime);
163             sproc.setStartTime(now - sproc.getUpTime());
164             sproc.setUserTime(ParseUtil.parseDHMSOrDefault(split[12], 0L));
165             sproc.setPath(split[13]);
166             sproc.setName(sproc.getPath().substring(sproc.getPath().lastIndexOf('/') + 1));
167             sproc.setCommandLine(split[14]);
168             sproc.setCurrentWorkingDirectory(MapUtil.getOrDefault(cwdMap, sproc.getProcessID(), ""));
169             // bytes read/written not easily available
170 
171             // gets the open files count -- slow
172             if (slowFields) {
173                 List<String> openFilesList = ExecutingCommand.runNative(String.format("lsof -p %d", pid));
174                 sproc.setOpenFiles(openFilesList.size() - 1L);
175             }
176             procs.add(sproc);
177         }
178         return procs;
179     }

197     /**
198      * {@inheritDoc}
199      */
200     @Override
201     public int getThreadCount() {
202         List<String> threadList = ExecutingCommand.runNative("ps -eLo pid");
203         if (!threadList.isEmpty()) {
204             // Subtract 1 for header
205             return threadList.size() - 1;
206         }
207         return getProcessCount();
208     }
