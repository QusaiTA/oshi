41  /**
42   * The Solaris File System contains {@link OSFileStore}s which are a storage
43   * pool, device, partition, volume, concrete file system or other implementation
44   * specific means of file storage. In Solaris, these are found in the
45   * /proc/mount filesystem, excluding temporary and kernel mounts.
46   *
47   * @author widdis[at]gmail[dot]com
48   */
49  public class SolarisFileSystem implements FileSystem {
50  
51      private static final long serialVersionUID = 1L;
52  
53      // Solaris defines a set of virtual file systems
54      private final List<String> pseudofs = Arrays.asList(new String[] { //
55              "proc", // Proc file system
56              "devfs", // Dev temporary file system
57              "ctfs", // Contract file system
58              "objfs", // Object file system
59              "mntfs", // Mount file system
60              "sharefs", // Share file system
61              "lofs", // Library file system
62              "autofs" // Auto mounting fs
63              // "tmpfs", // Temporary file system
64              // NOTE: tmpfs is evaluated apart, because Solaris uses it for
65              // RAMdisks
66      });
67  
68      // System path mounted as tmpfs
69      private final List<String> tmpfsPaths = Arrays.asList(new String[] { "/system", "/tmp", "/dev/fd" });
70  
71      /**
72       * Checks if file path equals or starts with an element in the given list
73       *
74       * @param aList
75       *            A list of path prefixes
76       * @param charSeq
77       *            a path to check
78       * @return true if the charSeq exactly equals, or starts with the directory in
79       *         aList
80       */
81      private boolean listElementStartsWith(List<String> aList, String charSeq) {
82          for (String match : aList) {
83              if (charSeq.equals(match) || charSeq.startsWith(match + "/")) {
84                  return true;
85              }
86          }
87          return false;
88      }
89  
90      /**
91       * Gets File System Information.
92       *
93       * @return An array of {@link OSFileStore} objects representing mounted volumes.
94       *         May return disconnected volumes with
95       *         {@link OSFileStore#getTotalSpace()} = 0.
96       */
97      @Override
98      public OSFileStore[] getFileStores() {
99          List<OSFileStore> fsList = new ArrayList<OSFileStore>();
100 
101         // Get inode usage data
102         Map<String, Long> inodeFreeMap = new HashMap<String, Long>();
103         Map<String, Long> inodeTotalMap = new HashMap<String, Long>();
104         String key = null;
105         String total = null;
106         String free = null;
107         for (String line : ExecutingCommand.runNative("df -g")) {
108             /*- Sample Output:
109             /                  (/dev/md/dsk/d0    ):         8192 block size          1024 frag size
110             41310292 total blocks   18193814 free blocks 17780712 available        2486848 total files
111              2293351 free files     22282240 filesys id
112                  ufs fstype       0x00000004 flag             255 filename length
113             */
114             if (line.startsWith("/")) {
115                 key = ParseUtil.whitespaces.split(line)[0];
116                 total = null;
117             } else if (line.contains("available") && line.contains("total files")) {
118                 total = ParseUtil.getTextBetweenStrings(line, "available", "total files").trim();
119             } else if (line.contains("free files")) {
120                 free = ParseUtil.getTextBetweenStrings(line, "", "free files").trim();
121                 if (key != null && total != null) {
122                     inodeFreeMap.put(key, ParseUtil.parseLongOrDefault(free, 0L));
123                     inodeTotalMap.put(key, ParseUtil.parseLongOrDefault(total, 0L));
124                     key = null;
125                 }
126             }
127         }
128 
129         // Get mount table
130         for (String fs : ExecutingCommand.runNative("cat /etc/mnttab")) {
131             String[] split = ParseUtil.whitespaces.split(fs);
132             if (split.length < 5) {
133                 continue;
134             }
135             // 1st field is volume name
136             // 2nd field is mount point
137             // 3rd field is fs type
138             // other fields ignored
139             String volume = split[0];
140             String path = split[1];
141             String type = split[2];
142 
143             // Exclude pseudo file systems
144             if (this.pseudofs.contains(type) || path.equals("/dev") || listElementStartsWith(this.tmpfsPaths, path)
145                     || volume.startsWith("rpool") && !path.equals("/")) {
146                 continue;
147             }
148 
149             String name = path.substring(path.lastIndexOf('/') + 1);
150             // Special case for /, pull last element of volume instead
151             if (name.isEmpty()) {
152                 name = volume.substring(volume.lastIndexOf('/') + 1);
153             }
154             long totalSpace = new File(path).getTotalSpace();
155             long usableSpace = new File(path).getUsableSpace();
156 
157             String description;
158             if (volume.startsWith("/dev") || path.equals("/")) {
159                 description = "Local Disk";
160             } else if (volume.equals("tmpfs")) {
161                 description = "Ram Disk";
162             } else if (type.startsWith("nfs") || type.equals("cifs")) {
163                 description = "Network Disk";
164             } else {
165                 description = "Mount Point";
166             }
167 
168             // Add to the list
169             OSFileStoreml#OSFileStore">OSFileStore osStore = new OSFileStore();
170             osStore.setName(name);
171             osStore.setVolume(volume);
172             osStore.setMount(path);
173             osStore.setDescription(description);
174             osStore.setType(type);
175             osStore.setUUID(""); // No UUID info on Solaris
176             osStore.setUsableSpace(usableSpace);
177             osStore.setTotalSpace(totalSpace);
178             osStore.setFreeInodes(inodeFreeMap.containsKey(path) ? inodeFreeMap.get(path) : 0L);
179             osStore.setTotalInodes(inodeTotalMap.containsKey(path) ? inodeTotalMap.get(path) : 0L);
180             fsList.add(osStore);
181         }
182         return fsList.toArray(new OSFileStore[fsList.size()]);
183     }

185     @Override
186     public long getOpenFileDescriptors() {
187         Kstat ksp = KstatUtil.kstatLookup(null, -1, "file_cache");
188         // Set values
189         if (ksp != null && KstatUtil.kstatRead(ksp)) {
190             return KstatUtil.kstatDataLookupLong(ksp, "buf_inuse");
191         }
192         return 0L;
193     }
194 
195     @Override
196     public long getMaxFileDescriptors() {
197         Kstat ksp = KstatUtil.kstatLookup(null, -1, "file_cache");
198         // Set values
199         if (ksp != null && KstatUtil.kstatRead(ksp)) {
200             return KstatUtil.kstatDataLookupLong(ksp, "buf_max");
201         }
202         return 0L;
203     }
