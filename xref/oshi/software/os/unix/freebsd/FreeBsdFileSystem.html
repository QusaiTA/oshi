40  /**
41   * The Solaris File System contains {@link OSFileStore}s which are a storage
42   * pool, device, partition, volume, concrete file system or other implementation
43   * specific means of file storage. In Linux, these are found in the /proc/mount
44   * filesystem, excluding temporary and kernel mounts.
45   *
46   * @author widdis[at]gmail[dot]com
47   */
48  public class FreeBsdFileSystem implements FileSystem {
49  
50      private static final long serialVersionUID = 1L;
51  
52      // Linux defines a set of virtual file systems
53      private final List<String> pseudofs = Arrays.asList(new String[] { //
54              "procfs", // Proc file system
55              "devfs", // Dev temporary file system
56              "ctfs", // Contract file system
57              "fdescfs", // fd
58              "objfs", // Object file system
59              "mntfs", // Mount file system
60              "sharefs", // Share file system
61              "lofs", // Library file system
62              "autofs" // Auto mounting fs
63              // "tmpfs", // Temporary file system
64              // NOTE: tmpfs is evaluated apart, because Solaris uses it for
65              // RAMdisks
66      });
67  
68      // System path mounted as tmpfs
69      private final List<String> tmpfsPaths = Arrays.asList(new String[] { "/system", "/tmp", "/dev/fd" });
70  
71      /**
72       * Checks if file path equals or starts with an element in the given list
73       *
74       * @param aList
75       *            A list of path prefixes
76       * @param charSeq
77       *            a path to check
78       * @return true if the charSeq exactly equals, or starts with the directory in
79       *         aList
80       */
81      private boolean listElementStartsWith(List<String> aList, String charSeq) {
82          for (String match : aList) {
83              if (charSeq.equals(match) || charSeq.startsWith(match + "/")) {
84                  return true;
85              }
86          }
87          return false;
88      }
89  
90      /**
91       * Gets File System Information.
92       *
93       * @return An array of {@link OSFileStore} objects representing mounted volumes.
94       *         May return disconnected volumes with
95       *         {@link OSFileStore#getTotalSpace()} = 0.
96       */
97      @Override
98      public OSFileStore[] getFileStores() {
99          // Find any partition UUIDs and map them
100         Map<String, String> uuidMap = new HashMap<String, String>();
101         // Now grab dmssg output
102         String device = "";
103         for (String line : ExecutingCommand.runNative("geom part list")) {
104             if (line.contains("Name: ")) {
105                 device = line.substring(line.lastIndexOf(' ') + 1);
106             }
107             // If we aren't working with a current partition, continue
108             if (device.isEmpty()) {
109                 continue;
110             }
111             line = line.trim();
112             if (line.startsWith("rawuuid:")) {
113                 uuidMap.put(device, line.substring(line.lastIndexOf(' ') + 1));
114                 device = "";
115             }
116         }
117 
118         List<OSFileStore> fsList = new ArrayList<OSFileStore>();
119 
120         // Get inode usage data
121         Map<String, Long> inodeFreeMap = new HashMap<String, Long>();
122         Map<String, Long> inodeTotalMap = new HashMap<String, Long>();
123         for (String line : ExecutingCommand.runNative("df -i")) {
124             /*- Sample Output:
125             Filesystem    1K-blocks   Used   Avail Capacity iused  ifree %iused  Mounted on
126             /dev/twed0s1a   2026030 584112 1279836    31%    2751 279871    1%   /
127             */
128             if (line.startsWith("/")) {
129                 String[] split = ParseUtil.whitespaces.split(line);
130                 if (split.length > 7) {
131                     inodeFreeMap.put(split[0], ParseUtil.parseLongOrDefault(split[6], 0L));
132                     // total is used + free
133                     inodeTotalMap.put(split[0],
134                             inodeFreeMap.get(split[0]) + ParseUtil.parseLongOrDefault(split[5], 0L));
135                 }
136             }
137         }
138 
139         // Get mount table
140         for (String fs : ExecutingCommand.runNative("mount -p")) {
141             String[] split = ParseUtil.whitespaces.split(fs);
142             if (split.length < 5) {
143                 continue;
144             }
145             // 1st field is volume name
146             // 2nd field is mount point
147             // 3rd field is fs type
148             // other fields ignored
149             String volume = split[0];
150             String path = split[1];
151             String type = split[2];
152 
153             // Exclude pseudo file systems
154             if (this.pseudofs.contains(type) || path.equals("/dev") || listElementStartsWith(this.tmpfsPaths, path)
155                     || volume.startsWith("rpool") && !path.equals("/")) {
156                 continue;
157             }
158 
159             String name = path.substring(path.lastIndexOf('/') + 1);
160             // Special case for /, pull last element of volume instead
161             if (name.isEmpty()) {
162                 name = volume.substring(volume.lastIndexOf('/') + 1);
163             }
164             long totalSpace = new File(path).getTotalSpace();
165             long usableSpace = new File(path).getUsableSpace();
166 
167             String description;
168             if (volume.startsWith("/dev") || path.equals("/")) {
169                 description = "Local Disk";
170             } else if (volume.equals("tmpfs")) {
171                 description = "Ram Disk";
172             } else if (type.startsWith("nfs") || type.equals("cifs")) {
173                 description = "Network Disk";
174             } else {
175                 description = "Mount Point";
176             }
177             // Match UUID
178             String uuid = MapUtil.getOrDefault(uuidMap, name, "");
179 
180             // Add to the list
181             OSFileStoreml#OSFileStore">OSFileStore osStore = new OSFileStore();
182             osStore.setName(name);
183             osStore.setVolume(volume);
184             osStore.setMount(path);
185             osStore.setDescription(description);
186             osStore.setType(type);
187             osStore.setUUID(uuid);
188             osStore.setUsableSpace(usableSpace);
189             osStore.setTotalSpace(totalSpace);
190             osStore.setFreeInodes(inodeFreeMap.containsKey(path) ? inodeFreeMap.get(path) : 0L);
191             osStore.setTotalInodes(inodeTotalMap.containsKey(path) ? inodeTotalMap.get(path) : 0L);
192             fsList.add(osStore);
193         }
194         return fsList.toArray(new OSFileStore[fsList.size()]);
195     }

197     @Override
198     public long getOpenFileDescriptors() {
199         return BsdSysctlUtil.sysctl("kern.openfiles", 0);
200     }
201 
202     @Override
203     public long getMaxFileDescriptors() {
204         return BsdSysctlUtil.sysctl("kern.maxfiles", 0);
205     }
