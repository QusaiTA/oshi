77      // Order the field is in /proc/pid/stat
78      enum ProcPidStat {
79          // The parsing implementation in ParseUtil requires these to be declared
80          // in increasing order
81          PPID(4), USER_TIME(14), KERNEL_TIME(15), PRIORITY(18), THREAD_COUNT(20), START_TIME(22), VSZ(23), RSS(24);
82  
83          private int order;
84  
85          public int getOrder() {
86              return this.order;
87          }
88  
89          ProcPidStat(int order) {
90              this.order = order;
91          }
92      }
93  
94      // Get a list of orders to pass to ParseUtil
95      private static final int[] PROC_PID_STAT_ORDERS = new int[ProcPidStat.values().length];
96      static {
97          for (ProcPidStat stat : ProcPidStat.values()) {
98              // The PROC_PID_STAT enum indices are 1-indexed.
99              // Subtract one to get a zero-based index
100             PROC_PID_STAT_ORDERS[stat.ordinal()] = stat.getOrder() - 1;
101         }
102     }

104     // 2.6 Kernel has 44 elements, 3.3 has 47, and 3.5 has 52.
105     // Check /proc/self/stat to find its length
106     private static final int PROC_PID_STAT_LENGTH;
107     static {
108         String stat = FileUtil.getStringFromFile("/proc/self/stat");
109         if (stat.contains(")")) {
110             // add 3 to account for pid, process name in prarenthesis, and state
111             PROC_PID_STAT_LENGTH = ParseUtil.countStringToLongArray(stat, ' ') + 3;
112         } else {
113             // Default assuming recent kernel
114             PROC_PID_STAT_LENGTH = 52;
115         }
116     }

120     // Jiffies per second, used for process time counters.
121     private static final long USER_HZ = ParseUtil.parseLongOrDefault(ExecutingCommand.getFirstAnswer("getconf CLK_TCK"),
122             100L);
123     // Boot time in MS.
124     private static final long BOOT_TIME;
125     static {
126         // Uptime is only in hundredths of seconds but we need thousandths.
127         // We can grab uptime twice and take average to reduce error, getting
128         // current time in between
129         double uptime = ProcUtil.getSystemUptimeSeconds();
130         long now = System.currentTimeMillis();
131         uptime += ProcUtil.getSystemUptimeSeconds();
132         // Uptime is now 2x seconds, so divide by 2, but
133         // we want milliseconds so multiply by 1000
134         // Ultimately multiply by 1000/2 = 500
135         BOOT_TIME = now - (long) (500d * uptime + 0.5);
136         // Cast/truncation is effectively rounding. Boot time could
137         // be +/- 5 ms due to System Uptime rounding to nearest 10ms.
138     }

172     @Override
173     public OSProcess[] getProcesses(int limit, ProcessSort sort, boolean slowFields) {
174         List<OSProcess> procs = new ArrayList<OSProcess>();
175         File[] pids = ProcUtil.getPidFiles();
176         List<Integer> pidsToKeep = new ArrayList<Integer>();
177 
178         // now for each file (with digit name) get process info
179         for (File pidFile : pids) {
180             int pid = ParseUtil.parseIntOrDefault(pidFile.getName(), 0);
181             OSProcess proc = getProcess(pid, slowFields);
182             if (proc != null) {
183                 procs.add(proc);
184                 pidsToKeep.add(pid);
185             }
186         }
187         // Clear out anything not in cache
188         for (Integer pid : new HashSet<Integer>(this.processMap.keySet())) {
189             if (!pidsToKeep.contains(pid)) {
190                 this.processMap.remove(pid);
191             }
192         }
193         // Sort
194         List<OSProcess> sorted = processSort(procs, limit, sort);
195         return sorted.toArray(new OSProcess[sorted.size()]);
196     }

206     private OSProcess getProcess(int pid, boolean slowFields) {
207         String path = "";
208         Memory buf = new Memory(1024);
209         buf.clear();
210         int size = Libc.INSTANCE.readlink(String.format("/proc/%d/exe", pid), buf, 1023);
211         if (size > 0) {
212             path = buf.getString(0).substring(0, size);
213         }
214         Map<String, String> io = FileUtil.getKeyValueMapFromFile(String.format("/proc/%d/io", pid), ":");
215         // See man proc for how to parse /proc/[pid]/stat
216         long now = System.currentTimeMillis();
217         String stat = FileUtil.getStringFromFile(String.format("/proc/%d/stat", pid));
218         // A race condition may leave us with an empty string
219         if (stat.isEmpty()) {
220             return null;
221         }
222         // We can get name and status more easily from /proc/pid/status which we
223         // call later, so just get the numeric bits here
224         long[] statArray = ParseUtil.parseStringToLongArray(stat, PROC_PID_STAT_ORDERS, PROC_PID_STAT_LENGTH, ' ');
225         // Fetch cached process if it exists
226         OSProcess proc = processMap.get(pid);
227         long startTime = BOOT_TIME + statArray[ProcPidStat.START_TIME.ordinal()] * 1000L / USER_HZ;
228         // BOOT_TIME could be up to 5ms off. In rare cases when a process has
229         // started within 5ms of boot it is possible to get negative uptime.
230         if (startTime >= now) {
231             startTime = now - 1;
232         }
233         // New process if start time differs by 200ms or more
234         if (proc == null || Math.abs(startTime - proc.getStartTime()) > 200) {
235             proc = new OSProcess();
236             proc.setProcessID(pid);
237             proc.setStartTime(startTime);
238             // The /proc/pid/cmdline value is null-delimited
239             proc.setCommandLine(FileUtil.getStringFromFile(String.format("/proc/%d/cmdline", pid)));
240             // Add or replace value in the map
241             processMap.put(pid, proc);
242         }
243         proc.setParentProcessID((int) statArray[ProcPidStat.PPID.ordinal()]);
244         proc.setThreadCount((int) statArray[ProcPidStat.THREAD_COUNT.ordinal()]);
245         proc.setPriority((int) statArray[ProcPidStat.PRIORITY.ordinal()]);
246         proc.setVirtualSize(statArray[ProcPidStat.VSZ.ordinal()]);
247         proc.setResidentSetSize(statArray[ProcPidStat.RSS.ordinal()] * this.memoryPageSize);
248         proc.setKernelTime(statArray[ProcPidStat.KERNEL_TIME.ordinal()] * 1000L / USER_HZ);
249         proc.setUserTime(statArray[ProcPidStat.USER_TIME.ordinal()] * 1000L / USER_HZ);
250         proc.setUpTime(now - proc.getStartTime());
251         // See man proc for how to parse /proc/[pid]/io
252         proc.setBytesRead(ParseUtil.parseLongOrDefault(MapUtil.getOrDefault(io, "read_bytes", ""), 0L));
253         proc.setBytesWritten(ParseUtil.parseLongOrDefault(MapUtil.getOrDefault(io, "write_bytes", ""), 0L));
254 
255         // gets the open files count
256         if (slowFields) {
257             List<String> openFilesList = ExecutingCommand.runNative(String.format("ls -f /proc/%d/fd", pid));
258             proc.setOpenFiles(openFilesList.size() - 1L);
259         }
260 
261         Map<String, String> status = FileUtil.getKeyValueMapFromFile(String.format("/proc/%d/status", pid), ":");
262         proc.setName(MapUtil.getOrDefault(status, "Name", ""));
263         proc.setPath(path);
264         switch (MapUtil.getOrDefault(status, "State", "U").charAt(0)) {
265         case 'R':
266             proc.setState(OSProcess.State.RUNNING);
267             break;
268         case 'S':
269             proc.setState(OSProcess.State.SLEEPING);
270             break;
271         case 'D':
272             proc.setState(OSProcess.State.WAITING);
273             break;
274         case 'Z':
275             proc.setState(OSProcess.State.ZOMBIE);
276             break;
277         case 'T':
278             proc.setState(OSProcess.State.STOPPED);
279             break;
280         default:
281             proc.setState(OSProcess.State.OTHER);
282             break;
283         }
284         proc.setUserID(ParseUtil.whitespaces.split(MapUtil.getOrDefault(status, "Uid", ""))[0]);
285         proc.setGroupID(ParseUtil.whitespaces.split(MapUtil.getOrDefault(status, "Gid", ""))[0]);
286         OSUser user = this.userGroupInfo.getUser(proc.getUserID());
287         if (user != null) {
288             proc.setUser(user.getUserName());
289         }
290         proc.setGroup(this.userGroupInfo.getGroupName(proc.getGroupID()));
291 
292         try {
293             String cwdLink = String.format("/proc/%d/cwd", pid);
294             String cwd = new File(cwdLink).getCanonicalPath();
295             if (!cwd.equals(cwdLink)) {
296                 proc.setCurrentWorkingDirectory(cwd);
297             }
298         } catch (IOException e) {
299             LOG.trace("Couldn't find cwd for pid {}: {}", pid, e);
300         }
301         return proc;
302     }

307     @Override
308     public OSProcess[] getChildProcesses(int parentPid, int limit, ProcessSort sort) {
309         List<OSProcess> procs = new ArrayList<OSProcess>();
310         File[] procFiles = ProcUtil.getPidFiles();
311 
312         // now for each file (with digit name) get process info
313         for (File procFile : procFiles) {
314             int pid = ParseUtil.parseIntOrDefault(procFile.getName(), 0);
315             if (parentPid == getParentPidFromProcFile(pid)) {
316                 OSProcess proc = getProcess(pid, true);
317                 if (proc != null) {
318                     procs.add(proc);
319                 }
320             }
321         }
322         List<OSProcess> sorted = processSort(procs, limit, sort);
323         return sorted.toArray(new OSProcess[sorted.size()]);
324     }

326     private static int getParentPidFromProcFile(int pid) {
327         String stat = FileUtil.getStringFromFile(String.format("/proc/%d/stat", pid));
328         long[] statArray = ParseUtil.parseStringToLongArray(stat, PROC_PID_STAT_ORDERS, PROC_PID_STAT_LENGTH, ' ');
329         return (int) statArray[ProcPidStat.PPID.ordinal()];
330     }

353         try {
354             Sysinfo info = new Sysinfo();
355             if (0 != LibC.INSTANCE.sysinfo(info)) {
356                 LOG.error("Failed to get process thread count. Error code: {}", Native.getLastError());
357                 return 0;
358             }
359             return info.procs;
360         } catch (UnsatisfiedLinkError e) {
361             LOG.error("Failed to get procs from sysinfo. {}", e);
362         } catch (NoClassDefFoundError e) {
363             LOG.error("Failed to get procs from sysinfo. {}", e);
364         }
365         return 0;
366     }

376     private void setFamilyFromReleaseFiles() {
377         if (this.family == null) {
378             // There are two competing options for family/version information.
379             // Newer systems are adopting a standard /etc/os-release file:
380             // https://www.freedesktop.org/software/systemd/man/os-release.html
381             //
382             // Some systems are still using the lsb standard which parses a
383             // variety of /etc/*-release files and is most easily accessed via
384             // the commandline lsb_release -a, see here:
385             // http://linux.die.net/man/1/lsb_release
386             // In this case, the /etc/lsb-release file (if it exists) has
387             // optional overrides to the information in the /etc/distrib-release
388             // files, which show: "Distributor release x.x (Codename)"
389             //
390 
391             // Attempt to read /etc/system-release which has more details than
392             // os-release on (CentOS and Fedora)
393             if (readDistribRelease("/etc/system-release")) {
394                 // If successful, we're done. this.family has been set and
395                 // possibly the versionID and codeName
396                 return;
397             }
398 
399             // Attempt to read /etc/os-release file.
400             if (readOsRelease()) {
401                 // If successful, we're done. this.family has been set and
402                 // possibly the versionID and codeName
403                 return;
404             }
405 
406             // Attempt to execute the `lsb_release` command
407             if (execLsbRelease()) {
408                 // If successful, we're done. this.family has been set and
409                 // possibly the versionID and codeName
410                 return;
411             }
412 
413             // The above two options should hopefully work on most
414             // distributions. If not, we keep having fun.
415             // Attempt to read /etc/lsb-release file
416             if (readLsbRelease()) {
417                 // If successful, we're done. this.family has been set and
418                 // possibly the versionID and codeName
419                 return;
420             }
421 
422             // If we're still looking, we search for any /etc/*-release (or
423             // similar) filename, for which the first line should be of the
424             // "Distributor release x.x (Codename)" format or possibly a
425             // "Distributor VERSION x.x (Codename)" format
426             String etcDistribRelease = getReleaseFilename();
427             if (readDistribRelease(etcDistribRelease)) {
428                 // If successful, we're done. this.family has been set and
429                 // possibly the versionID and codeName
430                 return;
431             }
432             // If we've gotten this far with no match, use the distrib-release
433             // filename (defaults will eventually give "Unknown")
434             this.family = filenameToFamily(etcDistribRelease.replace("/etc/", "").replace("release", "")
435                     .replace("version", "").replace("-", "").replace("_", ""));
436         }
437     }

439     /**
440      * Attempts to read /etc/os-release
441      *
442      * @return true if file successfully read and NAME= found
443      */
444     private boolean readOsRelease() {
445         if (new File("/etc/os-release").exists()) {
446             List<String> osRelease = FileUtil.readFile("/etc/os-release");
447             // Search for NAME=
448             for (String line : osRelease) {
449                 if (line.startsWith("VERSION=")) {
450                     LOG.debug("os-release: {}", line);
451                     // remove beginning and ending '"' characters, etc from
452                     // VERSION="14.04.4 LTS, Trusty Tahr" (Ubuntu style)
453                     // or VERSION="17 (Beefy Miracle)" (os-release doc style)
454                     line = line.replace("VERSION=", "").replaceAll("^\"|\"$", "").trim();
455                     String[] split = line.split("[()]");
456                     if (split.length <= 1) {
457                         // If no parentheses, check for Ubuntu's comma format
458                         split = line.split(", ");
459                     }
460                     if (split.length > 0) {
461                         this.versionId = split[0].trim();
462                     }
463                     if (split.length > 1) {
464                         this.codeName = split[1].trim();
465                     }
466                 } else if (line.startsWith("NAME=") && this.family == null) {
467                     LOG.debug("os-release: {}", line);
468                     // remove beginning and ending '"' characters, etc from
469                     // NAME="Ubuntu"
470                     this.family = line.replace("NAME=", "").replaceAll("^\"|\"$", "").trim();
471                 } else if (line.startsWith("VERSION_ID=") && this.versionId == null) {
472                     LOG.debug("os-release: {}", line);
473                     // remove beginning and ending '"' characters, etc from
474                     // VERSION_ID="14.04"
475                     this.versionId = line.replace("VERSION_ID=", "").replaceAll("^\"|\"$", "").trim();
476                 }
477             }
478         }
479         return this.family != null;
480     }

482     /**
483      * Attempts to execute `lsb_release -a`
484      *
485      * @return true if the command successfully executed and Distributor ID: or
486      *         Description: found
487      */
488     private boolean execLsbRelease() {
489         // If description is of the format Distrib release x.x (Codename)
490         // that is primary, otherwise use Distributor ID: which returns the
491         // distribution concatenated, e.g., RedHat instead of Red Hat
492         for (String line : ExecutingCommand.runNative("lsb_release -a")) {
493             if (line.startsWith("Description:")) {
494                 LOG.debug("lsb_release -a: {}", line);
495                 line = line.replace("Description:", "").trim();
496                 if (line.contains(" release ")) {
497                     this.family = parseRelease(line, " release ");
498                 }
499             } else if (line.startsWith("Distributor ID:") && this.family == null) {
500                 LOG.debug("lsb_release -a: {}", line);
501                 this.family = line.replace("Distributor ID:", "").trim();
502             } else if (line.startsWith("Release:") && this.versionId == null) {
503                 LOG.debug("lsb_release -a: {}", line);
504                 this.versionId = line.replace("Release:", "").trim();
505             } else if (line.startsWith("Codename:") && this.codeName == null) {
506                 LOG.debug("lsb_release -a: {}", line);
507                 this.codeName = line.replace("Codename:", "").trim();
508             }
509         }
510         return this.family != null;
511     }

513     /**
514      * Attempts to read /etc/lsb-release
515      *
516      * @return true if file successfully read and DISTRIB_ID or DISTRIB_DESCRIPTION
517      *         found
518      */
519     private boolean readLsbRelease() {
520         if (new File("/etc/lsb-release").exists()) {
521             List<String> osRelease = FileUtil.readFile("/etc/lsb-release");
522             // Search for NAME=
523             for (String line : osRelease) {
524                 if (line.startsWith("DISTRIB_DESCRIPTION=")) {
525                     LOG.debug("lsb-release: {}", line);
526                     line = line.replace("DISTRIB_DESCRIPTION=", "").replaceAll("^\"|\"$", "").trim();
527                     if (line.contains(" release ")) {
528                         this.family = parseRelease(line, " release ");
529                     }
530                 } else if (line.startsWith("DISTRIB_ID=") && this.family == null) {
531                     LOG.debug("lsb-release: {}", line);
532                     this.family = line.replace("DISTRIB_ID=", "").replaceAll("^\"|\"$", "").trim();
533                 } else if (line.startsWith("DISTRIB_RELEASE=") && this.versionId == null) {
534                     LOG.debug("lsb-release: {}", line);
535                     this.versionId = line.replace("DISTRIB_RELEASE=", "").replaceAll("^\"|\"$", "").trim();
536                 } else if (line.startsWith("DISTRIB_CODENAME=") && this.codeName == null) {
537                     LOG.debug("lsb-release: {}", line);
538                     this.codeName = line.replace("DISTRIB_CODENAME=", "").replaceAll("^\"|\"$", "").trim();
539                 }
540             }
541         }
542         return this.family != null;
543     }

545     /**
546      * Attempts to read /etc/distrib-release (for some value of distrib)
547      *
548      * @return true if file successfully read and " release " or " VERSION " found
549      */
550     private boolean readDistribRelease(String filename) {
551         if (new File(filename).exists()) {
552             List<String> osRelease = FileUtil.readFile(filename);
553             // Search for Distrib release x.x (Codename)
554             for (String line : osRelease) {
555                 LOG.debug("{}: {}", filename, line);
556                 if (line.contains(" release ")) {
557                     this.family = parseRelease(line, " release ");
558                     // If this parses properly we're done
559                     break;
560                 } else if (line.contains(" VERSION ")) {
561                     this.family = parseRelease(line, " VERSION ");
562                     // If this parses properly we're done
563                     break;
564                 }
565             }
566         }
567         return this.family != null;
568     }

570     /**
571      * Helper method to parse version description line style
572      *
573      * @param line
574      *            a String of the form "Distributor release x.x (Codename)"
575      * @param splitLine
576      *            A regex to split on, e.g. " release "
577      * @return the parsed family (versionID and codeName may have also been set)
578      */
579     private String parseRelease(String line, String splitLine) {
580         String[] split = line.split(splitLine);
581         String family = split[0].trim();
582         if (split.length > 1) {
583             split = split[1].split("[()]");
584             if (split.length > 0) {
585                 this.versionId = split[0].trim();
586             }
587             if (split.length > 1) {
588                 this.codeName = split[1].trim();
589             }
590         }
591         return family;
592     }

594     /**
595      * Looks for a collection of possible distrib-release filenames
596      *
597      * @return The first valid matching filename
598      */
599     protected static String getReleaseFilename() {
600         // Look for any /etc/*-release, *-version, and variants
601         File etc = new File("/etc");
602         // Find any *_input files in that path
603         File[] matchingFiles = etc.listFiles(new FileFilter() {
604             @Override
605             public boolean accept(File f) {
606                 return (f.getName().endsWith("-release") || f.getName().endsWith("-version")
607                         || f.getName().endsWith("_release") || f.getName().endsWith("_version"))
608                         && !(f.getName().endsWith("os-release") || f.getName().endsWith("lsb-release")
609                                 || f.getName().endsWith("system-release"));
610             }
611         });
612         if (matchingFiles != null && matchingFiles.length > 0) {
613             return matchingFiles[0].getPath();
614         }
615         if (new File("/etc/release").exists()) {
616             return "/etc/release";
617         }
618         // If all else fails, try this
619         return "/etc/issue";
620     }

622     /**
623      * Converts a portion of a filename (e.g. the 'redhat' in /etc/redhat-release)
624      * to a mixed case string representing the family (e.g., Red Hat)
625      *
626      * @param name
627      *            Stripped version of filename after removing /etc and -release
628      * @return Mixed case family
629      */
630     private static String filenameToFamily(String name) {
631         String n = name.toLowerCase();
632         // Handle known special cases
633         if (n.equals("")) {
634             return "Solaris";
635         } else if (n.equals("blackcat")) {
636             return "Black Cat";
637         } else if (n.equals("bluewhite64")) {
638             return "BlueWhite64";
639         } else if (n.equals("e-smith")) {
640             return "SME Server";
641         } else if (n.equals("eos")) {
642             return "FreeEOS";
643         } else if (n.equals("hlfs")) {
644             return "HLFS";
645         } else if (n.equals("lfs")) {
646             return "Linux-From-Scratch";
647         } else if (n.equals("linuxppc")) {
648             return "Linux-PPC";
649         } else if (n.equals("meego")) {
650             return "MeeGo";
651         } else if (n.equals("mandakelinux")) {
652             return "Mandrake";
653         } else if (n.equals("mklinux")) {
654             return "MkLinux";
655         } else if (n.equals("nld")) {
656             return "Novell Linux Desktop";
657         } else if (n.equals("novell") || n.equals("SuSE")) {
658             return "SUSE Linux";
659         } else if (n.equals("pld")) {
660             return "PLD";
661         } else if (n.equals("redhat")) {
662             return "Red Hat Linux";
663         } else if (n.equals("sles")) {
664             return "SUSE Linux ES9";
665         } else if (n.equals("sun")) {
666             return "Sun JDS";
667         } else if (n.equals("synoinfo")) {
668             return "Synology";
669         } else if (n.equals("tinysofa")) {
670             return "Tiny Sofa";
671         } else if (n.equals("turbolinux")) {
672             return "TurboLinux";
673         } else if (n.equals("ultrapenguin")) {
674             return "UltraPenguin";
675         } else if (n.equals("va")) {
676             return "VA-Linux";
677         } else if (n.equals("vmware")) {
678             return "VMWareESX";
679         } else if (n.equals("yellowdog")) {
680             return "Yellow Dog";
681 
682             // /etc/issue will end up here:
683         } else if (n.equals("issue")) {
684             return "Unknown";
685         }
686         // If not a special case just capitalize first letter
687         return name.substring(0, 1).toUpperCase() + name.substring(1);
688     }
