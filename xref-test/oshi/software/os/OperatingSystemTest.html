48      @Test
49      public void testOperatingSystem() {
50          SystemInfo si = new SystemInfo();
51          OperatingSystem os = si.getOperatingSystem();
52          assertNotNull(os.getFamily());
53          assertNotNull(os.getManufacturer());
54          OperatingSystemVersion version = os.getVersion();
55          assertNotNull(version);
56          assertNotNull(version.getVersion());
57          assertNotNull(version.getCodeName());
58          assertNotNull(version.getBuildNumber());
59  
60          assertTrue(os.getProcessCount() >= 1);
61          assertTrue(os.getThreadCount() >= 1);
62          assertTrue(os.getBitness() == 32 || os.getBitness() == 64);
63          assertTrue(os.getProcessId() > 0);
64  
65          assertTrue(os.getProcesses(0, null).length > 0);
66          OSProcess proc = os.getProcess(os.getProcessId());
67          assertTrue(proc.getName().length() > 0);
68          assertTrue(proc.getPath().length() > 0);
69          // assertTrue(proc.getCommandLine().length() > 0);
70          assertNotNull(proc.getCurrentWorkingDirectory());
71          assertNotNull(proc.getUser());
72          assertNotNull(proc.getUserID());
73          assertNotNull(proc.getGroup());
74          assertNotNull(proc.getGroupID());
75          assertNotNull(proc.getState());
76          assertEquals(proc.getProcessID(), os.getProcessId());
77          assertTrue(proc.getParentProcessID() > 0);
78          assertTrue(proc.getThreadCount() > 0);
79          assertTrue(proc.getPriority() >= -20 && proc.getPriority() <= 128);
80          assertTrue(proc.getVirtualSize() >= 0);
81          assertTrue(proc.getResidentSetSize() >= 0);
82          assertTrue(proc.getKernelTime() >= 0);
83          assertTrue(proc.getUserTime() >= 0);
84          assertTrue(proc.getUpTime() >= 0);
85          assertTrue(proc.calculateCpuPercent() >= 0d);
86          assertTrue(proc.getStartTime() >= 0);
87          assertTrue(proc.getBytesRead() >= 0);
88          assertTrue(proc.getBytesWritten() >= 0);
89          assertTrue(proc.getOpenFiles() >= -1);
90      }

92      /**
93       * Tests process query by pid list
94       */
95      @Test
96      public void testProcessQueryByList() {
97          SystemInfo si = new SystemInfo();
98          OperatingSystem os = si.getOperatingSystem();
99          assertNotNull(os.getFamily());
100         assertNotNull(os.getManufacturer());
101         OperatingSystemVersion version = os.getVersion();
102         assertNotNull(version);
103         assertNotNull(version.getVersion());
104         assertNotNull(version.getCodeName());
105         assertNotNull(version.getBuildNumber());
106 
107         assertTrue(os.getProcessCount() >= 1);
108         assertTrue(os.getThreadCount() >= 1);
109         assertTrue(os.getProcessId() > 0);
110 
111         OSProcess[] processes = os.getProcesses(5, null);
112         assertNotNull(processes);
113         // every OS should have at least one process running on it
114         assertTrue(processes.length > 0);
115         // the list of pids we want info on
116         List<Integer> pids = new ArrayList<Integer>();
117         for (OSProcess p : processes) {
118             pids.add(p.getProcessID());
119         }
120         // query for just those processes
121         Collection<OSProcess> processes1 = os.getProcesses(pids);
122         // theres a potential for a race condition here, if a process we queried
123         // for initially wasn't running during the second query. In this case,
124         // try again with the shorter list
125         while (processes1.size() < pids.size()) {
126             pids.clear();
127             for (OSProcess p : processes1) {
128                 pids.add(p.getProcessID());
129             }
130             // query for just those processes
131             processes1 = os.getProcesses(pids);
132         }
133         assertEquals(processes1.size(), pids.size());
134 
135     }

137     /**
138      * Tests child process getter
139      */
140     @Test
141     public void testGetChildProcesses() {
142         // Get list of PIDS
143         SystemInfo si = new SystemInfo();
144         OperatingSystem os = si.getOperatingSystem();
145         OSProcess[] processes = os.getProcesses(0, null);
146         Map<Integer, Integer> childMap = new HashMap<Integer, Integer>();
147         // First iteration to set all 0's
148         for (OSProcess p : processes) {
149             childMap.put(p.getProcessID(), 0);
150             childMap.put(p.getParentProcessID(), 0);
151         }
152         // Second iteration to count children
153         for (OSProcess p : processes) {
154             childMap.put(p.getParentProcessID(), childMap.get(p.getParentProcessID()) + 1);
155         }
156         // Find a PID with 0, 1, and N>1 children
157         int zeroPid = -1;
158         int onePid = -1;
159         int nPid = -1;
160         int nNum = 0;
161         int mPid = -1;
162         int mNum = 0;
163         for (Integer i : childMap.keySet()) {
164             if (zeroPid < 0 && childMap.get(i) == 0) {
165                 zeroPid = i;
166             } else if (onePid < 0 && childMap.get(i) == 1) {
167                 onePid = i;
168             } else if (nPid < 0 && childMap.get(i) > 1) {
169                 // nPid is probably PID=1 with all PIDs with no other parent
170                 nPid = i;
171                 nNum = childMap.get(i);
172             } else if (mPid < 0 && childMap.get(i) > 1) {
173                 mPid = i;
174                 mNum = childMap.get(i);
175             }
176             if (zeroPid >= 0 && onePid >= 0 && nPid >= 0 && mPid >= 0) {
177                 break;
178             }
179         }
180         if (zeroPid >= 0) {
181             assertEquals(0, os.getChildProcesses(zeroPid, 0, null).length);
182         }
183         // Due to race condition, a process may terminate before we count its
184         // children.
185         if (onePid >= 0) {
186             assertTrue(0 <= os.getChildProcesses(onePid, 0, null).length);
187         }
188         if (nPid >= 0) {
189             assertTrue(0 <= os.getChildProcesses(nPid, 0, null).length);
190         }
191         if (mPid >= 0) {
192             assertTrue(0 <= os.getChildProcesses(mPid, 0, null).length);
193         }
194         // At least one of these tests should work.
195         if (onePid >= 0 && nPid >= 0 && mPid >= 0) {
196             assertTrue(os.getChildProcesses(onePid, 0, null).length == 1
197                     || os.getChildProcesses(nPid, 0, null).length == nNum
198                     || os.getChildProcesses(mPid, 0, null).length == mNum);
199         }
200     }

257     @Test
258     public void testGetCommandLine() {
259         int processesWithNonEmptyCmdLine = 0;
260 
261         SystemInfo si = new SystemInfo();
262         OperatingSystem os = si.getOperatingSystem();
263         for (OSProcess process : os.getProcesses(0, null)) {
264             if (!process.getCommandLine().trim().isEmpty()) {
265                 processesWithNonEmptyCmdLine++;
266             }
267         }
268 
269         assertTrue(processesWithNonEmptyCmdLine >= 1);
270     }
