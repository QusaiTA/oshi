180     private static void printCpu(CentralProcessor processor) {
181         System.out.println("Uptime: " + FormatUtil.formatElapsedSecs(processor.getSystemUptime()));
182         System.out.println(
183                 "Context Switches/Interrupts: " + processor.getContextSwitches() + " / " + processor.getInterrupts());
184         long[] prevTicks = processor.getSystemCpuLoadTicks();
185         processor.getProcessorCpuLoadTicks();
186         System.out.println("CPU, IOWait, and IRQ ticks @ 0 sec:" + Arrays.toString(prevTicks));
187         // Wait a second...
188         Util.sleep(1000);
189         long[] ticks = processor.getSystemCpuLoadTicks();
190         System.out.println("CPU, IOWait, and IRQ ticks @ 1 sec:" + Arrays.toString(ticks));
191         long user = ticks[TickType.USER.getIndex()] - prevTicks[TickType.USER.getIndex()];
192         long nice = ticks[TickType.NICE.getIndex()] - prevTicks[TickType.NICE.getIndex()];
193         long sys = ticks[TickType.SYSTEM.getIndex()] - prevTicks[TickType.SYSTEM.getIndex()];
194         long idle = ticks[TickType.IDLE.getIndex()] - prevTicks[TickType.IDLE.getIndex()];
195         long iowait = ticks[TickType.IOWAIT.getIndex()] - prevTicks[TickType.IOWAIT.getIndex()];
196         long irq = ticks[TickType.IRQ.getIndex()] - prevTicks[TickType.IRQ.getIndex()];
197         long softirq = ticks[TickType.SOFTIRQ.getIndex()] - prevTicks[TickType.SOFTIRQ.getIndex()];
198         long steal = ticks[TickType.STEAL.getIndex()] - prevTicks[TickType.STEAL.getIndex()];
199         long totalCpu = user + nice + sys + idle + iowait + irq + softirq + steal;
200 
201         System.out.format(
202                 "User: %.1f%% Nice: %.1f%% System: %.1f%% Idle: %.1f%% IOwait: %.1f%% IRQ: %.1f%% SoftIRQ: %.1f%% Steal: %.1f%%%n",
203                 100d * user / totalCpu, 100d * nice / totalCpu, 100d * sys / totalCpu, 100d * idle / totalCpu,
204                 100d * iowait / totalCpu, 100d * irq / totalCpu, 100d * softirq / totalCpu, 100d * steal / totalCpu);
205         System.out.format("CPU load: %.1f%% (counting ticks)%n", processor.getSystemCpuLoadBetweenTicks() * 100);
206         System.out.format("CPU load: %.1f%% (OS MXBean)%n", processor.getSystemCpuLoad() * 100);
207         double[] loadAverage = processor.getSystemLoadAverage(3);
208         System.out.println("CPU load averages:" + (loadAverage[0] < 0 ? " N/A" : String.format(" %.2f", loadAverage[0]))
209                 + (loadAverage[1] < 0 ? " N/A" : String.format(" %.2f", loadAverage[1]))
210                 + (loadAverage[2] < 0 ? " N/A" : String.format(" %.2f", loadAverage[2])));
211         // per core CPU
212         StringBuilder procCpu = new StringBuilder("CPU load per processor:");
213         double[] load = processor.getProcessorCpuLoadBetweenTicks();
214         for (double avg : load) {
215             procCpu.append(String.format(" %.1f%%", avg * 100));
216         }
217         System.out.println(procCpu.toString());
218     }

220     private static void printProcesses(OperatingSystem os, GlobalMemory memory) {
221         System.out.println("Processes: " + os.getProcessCount() + ", Threads: " + os.getThreadCount());
222         // Sort by highest CPU
223         List<OSProcess> procs = Arrays.asList(os.getProcesses(5, ProcessSort.CPU));
224 
225         System.out.println("   PID  %CPU %MEM       VSZ       RSS Name");
226         for (int i = 0; i < procs.size() && i < 5; i++) {
227             OSProcess p = procs.get(i);
228             System.out.format(" %5d %5.1f %4.1f %9s %9s %s%n", p.getProcessID(),
229                     100d * (p.getKernelTime() + p.getUserTime()) / p.getUpTime(),
230                     100d * p.getResidentSetSize() / memory.getTotal(), FormatUtil.formatBytes(p.getVirtualSize()),
231                     FormatUtil.formatBytes(p.getResidentSetSize()), p.getName());
232         }
233     }

242     private static void printPowerSources(PowerSource[] powerSources) {
243         StringBuilder sb = new StringBuilder("Power: ");
244         if (powerSources.length == 0) {
245             sb.append("Unknown");
246         } else {
247             double timeRemaining = powerSources[0].getTimeRemaining();
248             if (timeRemaining < -1d) {
249                 sb.append("Charging");
250             } else if (timeRemaining < 0d) {
251                 sb.append("Calculating time remaining");
252             } else {
253                 sb.append(String.format("%d:%02d remaining", (int) (timeRemaining / 3600),
254                         (int) (timeRemaining / 60) % 60));
255             }
256         }
257         for (PowerSource pSource : powerSources) {
258             sb.append(String.format("%n %s @ %.1f%%", pSource.getName(), pSource.getRemainingCapacity() * 100d));
259         }
260         System.out.println(sb.toString());
261     }

263     private static void printDisks(HWDiskStore[] diskStores) {
264         System.out.println("Disks:");
265         for (HWDiskStore disk : diskStores) {
266             boolean readwrite = disk.getReads() > 0 || disk.getWrites() > 0;
267             System.out.format(" %s: (model: %s - S/N: %s) size: %s, reads: %s (%s), writes: %s (%s), xfer: %s ms%n",
268                     disk.getName(), disk.getModel(), disk.getSerial(),
269                     disk.getSize() > 0 ? FormatUtil.formatBytesDecimal(disk.getSize()) : "?",
270                     readwrite ? disk.getReads() : "?", readwrite ? FormatUtil.formatBytes(disk.getReadBytes()) : "?",
271                     readwrite ? disk.getWrites() : "?", readwrite ? FormatUtil.formatBytes(disk.getWriteBytes()) : "?",
272                     readwrite ? disk.getTransferTime() : "?");
273             HWPartition[] partitions = disk.getPartitions();
274             if (partitions == null) {
275                 // TODO Remove when all OS's implemented
276                 continue;
277             }
278             for (HWPartition part : partitions) {
279                 System.out.format(" |-- %s: %s (%s) Maj:Min=%d:%d, size: %s%s%n", part.getIdentification(),
280                         part.getName(), part.getType(), part.getMajor(), part.getMinor(),
281                         FormatUtil.formatBytesDecimal(part.getSize()),
282                         part.getMountPoint().isEmpty() ? "" : " @ " + part.getMountPoint());
283             }
284         }
285     }

287     private static void printFileSystem(FileSystem fileSystem) {
288         System.out.println("File System:");
289 
290         System.out.format(" File Descriptors: %d/%d%n", fileSystem.getOpenFileDescriptors(),
291                 fileSystem.getMaxFileDescriptors());
292 
293         OSFileStore[] fsArray = fileSystem.getFileStores();
294         for (OSFileStore fs : fsArray) {
295             long usable = fs.getUsableSpace();
296             long total = fs.getTotalSpace();
297             System.out.format(
298                     " %s (%s) [%s] %s of %s free (%.1f%%), %s of %s files free (%.1f%%) is %s "
299                             + (fs.getLogicalVolume() != null && fs.getLogicalVolume().length() > 0 ? "[%s]" : "%s")
300                             + " and is mounted at %s%n",
301                     fs.getName(), fs.getDescription().isEmpty() ? "file system" : fs.getDescription(), fs.getType(),
302                     FormatUtil.formatBytes(usable), FormatUtil.formatBytes(fs.getTotalSpace()), 100d * usable / total,
303                     fs.getFreeInodes(), fs.getTotalInodes(), 100d * fs.getFreeInodes() / fs.getTotalInodes(),
304                     fs.getVolume(), fs.getLogicalVolume(), fs.getMount());
305         }
306     }

308     private static void printNetworkInterfaces(NetworkIF[] networkIFs) {
309         System.out.println("Network interfaces:");
310         for (NetworkIF net : networkIFs) {
311             System.out.format(" Name: %s (%s)%n", net.getName(), net.getDisplayName());
312             System.out.format("   MAC Address: %s %n", net.getMacaddr());
313             System.out.format("   MTU: %s, Speed: %s %n", net.getMTU(), FormatUtil.formatValue(net.getSpeed(), "bps"));
314             System.out.format("   IPv4: %s %n", Arrays.toString(net.getIPv4addr()));
315             System.out.format("   IPv6: %s %n", Arrays.toString(net.getIPv6addr()));
316             boolean hasData = net.getBytesRecv() > 0 || net.getBytesSent() > 0 || net.getPacketsRecv() > 0
317                     || net.getPacketsSent() > 0;
318             System.out.format("   Traffic: received %s/%s%s; transmitted %s/%s%s %n",
319                     hasData ? net.getPacketsRecv() + " packets" : "?",
320                     hasData ? FormatUtil.formatBytes(net.getBytesRecv()) : "?",
321                     hasData ? " (" + net.getInErrors() + " err)" : "",
322                     hasData ? net.getPacketsSent() + " packets" : "?",
323                     hasData ? FormatUtil.formatBytes(net.getBytesSent()) : "?",
324                     hasData ? " (" + net.getOutErrors() + " err)" : "");
325         }
326     }
