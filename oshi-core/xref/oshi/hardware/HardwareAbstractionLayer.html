24  package oshi.hardware;
25  
26  import java.io.Serializable;
27  
28  /**
29   * A hardware abstraction layer. Provides access to hardware items such as
30   * processors, memory, battery, and disks.
31   *
32   * @author dblock[at]dblock[dot]org
33   */
34  public interface HardwareAbstractionLayer extends Serializable {
35  
36      /**
37       * Instantiates a {@link ComputerSystem} object. This represents the physical
38       * hardware, including components such as BIOS/Firmware and a motherboard, logic
39       * board, etc.
40       *
41       * @return a {@link ComputerSystem} object.
42       */
43      ComputerSystem getComputerSystem();
44  
45      /**
46       * Instantiates a {@link CentralProcessor} object. This represents one or more
47       * Logical CPUs.
48       *
49       * @return A {@link CentralProcessor} object.
50       */
51      CentralProcessor getProcessor();
52  
53      /**
54       * Instantiates a {@link GlobalMemory} object.
55       *
56       * @return A memory object.
57       */
58      GlobalMemory getMemory();
59  
60      /**
61       * Instantiates an array of {@link PowerSource} objects, representing batteries,
62       * etc.
63       *
64       * @return An array of PowerSource objects or an empty array if none are
65       *         present.
66       */
67      PowerSource[] getPowerSources();
68  
69      /**
70       * Instantiates an array of {@link HWDiskStore} objects, representing a physical
71       * hard disk or other similar storage device
72       *
73       * @return An array of HWDiskStore objects or an empty array if none are
74       *         present.
75       */
76      HWDiskStore[] getDiskStores();
77  
78      /**
79       * Instantiates an array of {@link NetworkIF} objects, representing a network
80       * interface
81       *
82       * @return An array of HWNetworkStore objects or an empty array if none are
83       *         present.
84       */
85      NetworkIF[] getNetworkIFs();
86  
87      /**
88       * Instantiates an array of {@link Display} objects, representing monitors or
89       * other video output devices.
90       *
91       * @return An array of Display objects or an empty array if none are present.
92       */
93      Display[] getDisplays();
94  
95      /**
96       * Instantiates a {@link Sensors} object, representing CPU temperature and fan
97       * speed
98       *
99       * @return A Sensors object
100      */
101     Sensors getSensors();
102 
103     /**
104      * Instantiates an array of {@link UsbDevice} objects, representing devices
105      * connected via a usb port (including internal devices).
106      *
107      * If the value of tree is true, the top level devices returned from this method
108      * are the USB Controllers; connected hubs and devices in its device tree share
109      * that controller's bandwidth. If the value of tree is false, USB devices (not
110      * controllers) are listed in a single flat array.
111      *
112      * @param tree
113      *            WHether to display devices in a nested tree format from their
114      *            controllers
115      * @return An array of UsbDevice objects representing (optionally) the USB
116      *         Controllers and devices connected to them, or an empty array if none
117      *         are present
118      */
119     UsbDevice[] getUsbDevices(boolean tree);
120 
121     /**
122      * Instantiates an array of {@link SoundCard} objects, representing the Sound
123      * cards.
124      *
125      * @return An array of SoundCard objects or an empty array if none are present.
126      */
127     SoundCard[] getSoundCards();
128 }
