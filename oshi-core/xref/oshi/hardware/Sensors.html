24  package oshi.hardware;
25  
26  import java.io.Serializable;
27  
28  /**
29   * Sensors include hardwore sensors to monitor temperature, fan speed, and other
30   * information. Drivers may or may not exist to collect this data depending on
31   * Operating System and CPU. In addition, software-hardware communication may
32   * suffer intermittent errors when attempting to access this information, so it
33   * should be considered for information only. Users should test for zero values
34   * and empty arrays which will result if the OS is unable to provide the
35   * information.
36   *
37   * Windows information is retrieved via Windows Management Instrumentation
38   * (WMI). Unfortunately, most hardware providers do not publish values to WMI.
39   * If a value is not available through the Microsoft API, Oshi will attempt to
40   * retrieve values as published by the Open Hardware Monitor
41   * (http://openhardwaremonitor.org/) if it is running.
42   *
43   * @author widdis[at]gmail[dot]com
44   */
45  public interface Sensors extends Serializable {
46      /**
47       * CPU Temperature
48       *
49       * @return CPU Temperature in degrees Celsius if available, 0 otherwise.
50       */
51      double getCpuTemperature();
52  
53      /**
54       * Fan speeds
55       *
56       * @return Speed in rpm for all fans. May return empty array if no fans detected
57       *         or 0 fan speed if unable to measure fan speed.
58       */
59      int[] getFanSpeeds();
60  
61      /**
62       * CPU Voltage
63       *
64       * @return CPU Voltage in Volts if available, 0 otherwise.
65       */
66      double getCpuVoltage();
67  }
