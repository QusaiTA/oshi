24  package oshi.hardware.platform.linux;
25  
26  import java.util.List;
27  
28  import oshi.hardware.common.AbstractGlobalMemory;
29  import oshi.util.ExecutingCommand;
30  import oshi.util.FileUtil;
31  import oshi.util.ParseUtil;
32  
33  /**
34   * Memory obtained by /proc/meminfo and sysinfo.totalram
35   *
36   * @author alessandro[at]perucchi[dot]org
37   * @author widdis[at]gmail[dot]com
38   */
39  public class LinuxGlobalMemory extends AbstractGlobalMemory {
40  
41      private static final long serialVersionUID = 1L;
42  
43      // Values read from /proc/meminfo used for other calculations
44      private long memFree = 0;
45      private long activeFile = 0;
46      private long inactiveFile = 0;
47      private long sReclaimable = 0;
48      private long swapFree = 0;
49  
50      private long lastUpdate = 0;
51  
52      public LinuxGlobalMemory() {
53          this.pageSize = ParseUtil.parseLongOrDefault(ExecutingCommand.getFirstAnswer("getconf PAGE_SIZE"), 4096L);
54      }
55  
56      /**
57       * Updates instance variables from reading /proc/meminfo no more frequently than
58       * every 100ms. While most of the information is available in the sysinfo
59       * structure, the most accurate calculation of MemAvailable is only available
60       * from reading this pseudo-file. The maintainers of the Linux Kernel have
61       * indicated this location will be kept up to date if the calculation changes:
62       * see https://git.kernel.org/cgit/linux/kernel/git/torvalds/linux.git/commit/?
63       * id=34e431b0ae398fc54ea69ff85ec700722c9da773
64       *
65       * Internally, reading /proc/meminfo is faster than sysinfo because it only
66       * spends time populating the memory components of the sysinfo structure.
67       */
68      @Override
69      protected void updateMeminfo() {
70          long now = System.currentTimeMillis();
71          if (now - this.lastUpdate > 100) {
72              List<String> memInfo = FileUtil.readFile("/proc/meminfo");
73              boolean found = false;
74              for (String checkLine : memInfo) {
75                  String[] memorySplit = ParseUtil.whitespaces.split(checkLine);
76                  if (memorySplit.length > 1) {
77                      if (memorySplit[0].equals("MemTotal:")) {
78                          this.memTotal = parseMeminfo(memorySplit);
79                      } else if (memorySplit[0].equals("MemFree:")) {
80                          this.memFree = parseMeminfo(memorySplit);
81                      } else if (memorySplit[0].equals("MemAvailable:")) {
82                          this.memAvailable = parseMeminfo(memorySplit);
83                          found = true;
84                      } else if (memorySplit[0].equals("Active(file):")) {
85                          this.activeFile = parseMeminfo(memorySplit);
86                      } else if (memorySplit[0].equals("Inactive(file):")) {
87                          this.inactiveFile = parseMeminfo(memorySplit);
88                      } else if (memorySplit[0].equals("SReclaimable:")) {
89                          this.sReclaimable = parseMeminfo(memorySplit);
90                      } else if (memorySplit[0].equals("SwapTotal:")) {
91                          this.swapTotal = parseMeminfo(memorySplit);
92                      } else if (memorySplit[0].equals("SwapFree:")) {
93                          this.swapFree = parseMeminfo(memorySplit);
94                      }
95                      // do nothing with other lines
96                  }
97              }
98              this.swapUsed = this.swapTotal - this.swapFree;
99              // If no MemAvailable, calculate from other fields
100             if (!found) {
101                 this.memAvailable = this.memFree + this.activeFile + this.inactiveFile + this.sReclaimable;
102             }
103 
104             memInfo = FileUtil.readFile("/proc/vmstat");
105             for (String checkLine : memInfo) {
106                 String[] memorySplit = ParseUtil.whitespaces.split(checkLine);
107                 if (memorySplit.length > 1) {
108                     if (memorySplit[0].equals("pgpgin")) {
109                         this.swapPagesIn = ParseUtil.parseLongOrDefault(memorySplit[1], 0L);
110                     } else if (memorySplit[0].equals("pgpgout")) {
111                         this.swapPagesOut = ParseUtil.parseLongOrDefault(memorySplit[1], 0L);
112                     }
113                     // do nothing with other lines
114                 }
115             }
116 
117             this.lastUpdate = now;
118         }
119     }
120 
121     /**
122      * {@inheritDoc}
123      */
124     @Override
125     protected void updateSwap() {
126         updateMeminfo();
127     }
128 
129     /**
130      * Parses lines from the display of /proc/meminfo
131      *
132      * @param memorySplit
133      *            Array of Strings representing the 3 columns of /proc/meminfo
134      * @return value, multiplied by 1024 if kB is specified
135      */
136     private long parseMeminfo(String[] memorySplit) {
137         if (memorySplit.length < 2) {
138             return 0l;
139         }
140         long memory = ParseUtil.parseLongOrDefault(memorySplit[1], 0L);
141         if (memorySplit.length > 2 && "kB".equals(memorySplit[2])) {
142             memory *= 1024;
143         }
144         return memory;
145     }
146 }
