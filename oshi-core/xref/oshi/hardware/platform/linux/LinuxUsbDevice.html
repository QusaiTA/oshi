41  public class LinuxUsbDevice extends AbstractUsbDevice {
42  
43      private static final long serialVersionUID = 2L;
44  
45      /*
46       * Maps to store information using device node path as the key
47       */
48      private static Map<String, String> nameMap = new HashMap<String, String>();
49      private static Map<String, String> vendorMap = new HashMap<String, String>();
50      private static Map<String, String> vendorIdMap = new HashMap<String, String>();
51      private static Map<String, String> productIdMap = new HashMap<String, String>();
52      private static Map<String, String> serialMap = new HashMap<String, String>();
53      private static Map<String, List<String>> hubMap = new HashMap<String, List<String>>();
54  
55      public LinuxUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
56              UsbDevice[] connectedDevices) {
57          super(name, vendor, vendorId, productId, serialNumber, connectedDevices);
58      }
59  
60      /**
61       * {@inheritDoc}
62       */
63      public static UsbDevice[] getUsbDevices(boolean tree) {
64          UsbDevice[] devices = getUsbDevices();
65          if (tree) {
66              return devices;
67          }
68          List<UsbDevice> deviceList = new ArrayList<UsbDevice>();
69          // Top level is controllers; they won't be added to the list, but all
70          // their connected devices will be
71          for (UsbDevice device : devices) {
72              deviceList.add(new LinuxUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
73                      device.getProductId(), device.getSerialNumber(), new MacUsbDevice[0]));
74              addDevicesToList(deviceList, device.getConnectedDevices());
75          }
76          return deviceList.toArray(new UsbDevice[deviceList.size()]);
77      }
78  
79      private static void addDevicesToList(List<UsbDevice> deviceList, UsbDevice[] connectedDevices) {
80          for (UsbDevice device : connectedDevices) {
81              deviceList.add(device);
82              addDevicesToList(deviceList, device.getConnectedDevices());
83          }
84      }
85  
86      private static UsbDevice[] getUsbDevices() {
87          // Enumerate all usb devices and build information maps
88          Udev.UdevHandle udev = Udev.INSTANCE.udev_new();
89          // Create a list of the devices in the 'usb' subsystem.
90          UdevEnumerate enumerate = Udev.INSTANCE.udev_enumerate_new(udev);
91          Udev.INSTANCE.udev_enumerate_add_match_subsystem(enumerate, "usb");
92          Udev.INSTANCE.udev_enumerate_scan_devices(enumerate);
93          UdevListEntry devices = Udev.INSTANCE.udev_enumerate_get_list_entry(enumerate);
94  
95          // Build a list of devices with no parent; these will be the roots
96          List<String> usbControllers = new ArrayList<String>();
97          // Empty out maps
98          nameMap.clear();
99          vendorMap.clear();
100         vendorIdMap.clear();
101         productIdMap.clear();
102         serialMap.clear();
103         hubMap.clear();
104 
105         // For each item enumerated, store information in the maps
106         for (UdevListEntry dev_list_entry = devices; dev_list_entry != null; dev_list_entry = Udev.INSTANCE
107                 .udev_list_entry_get_next(dev_list_entry)) {
108 
109             // Get the filename of the /sys entry for the device and create a
110             // udev_device object (dev) representing it
111             String path = Udev.INSTANCE.udev_list_entry_get_name(dev_list_entry);
112             UdevDevice dev = Udev.INSTANCE.udev_device_new_from_syspath(udev, path);
113             // Ignore interfaces
114             if (!"usb_device".equals(Udev.INSTANCE.udev_device_get_devtype(dev))) {
115                 continue;
116             }
117 
118             // Use the path as the key for the maps
119             String value = Udev.INSTANCE.udev_device_get_sysattr_value(dev, "product");
120             if (value != null) {
121                 nameMap.put(path, value);
122             }
123             value = Udev.INSTANCE.udev_device_get_sysattr_value(dev, "manufacturer");
124             if (value != null) {
125                 vendorMap.put(path, value);
126             }
127             value = Udev.INSTANCE.udev_device_get_sysattr_value(dev, "idVendor");
128             if (value != null) {
129                 vendorIdMap.put(path, value);
130             }
131             value = Udev.INSTANCE.udev_device_get_sysattr_value(dev, "idProduct");
132             if (value != null) {
133                 productIdMap.put(path, value);
134             }
135             value = Udev.INSTANCE.udev_device_get_sysattr_value(dev, "serial");
136             if (value != null) {
137                 serialMap.put(path, value);
138             }
139             UdevDevice parent = Udev.INSTANCE.udev_device_get_parent_with_subsystem_devtype(dev, "usb", "usb_device");
140             if (parent == null) {
141                 // This is a controller with no parent, add to list
142                 usbControllers.add(path);
143             } else {
144                 // Add child path (path variable) to parent's path
145                 String parentPath = Udev.INSTANCE.udev_device_get_syspath(parent);
146                 MapUtil.createNewListIfAbsent(hubMap, parentPath).add(path);
147             }
148             Udev.INSTANCE.udev_device_unref(dev);
149         }
150         // Free the enumerator object
151         Udev.INSTANCE.udev_enumerate_unref(enumerate);
152         Udev.INSTANCE.udev_unref(udev);
153 
154         // Build tree and return
155         List<UsbDevice> controllerDevices = new ArrayList<UsbDevice>();
156         for (String controller : usbControllers) {
157             controllerDevices.add(getDeviceAndChildren(controller, "0000", "0000"));
158         }
159         return controllerDevices.toArray(new UsbDevice[controllerDevices.size()]);
160     }
161 
162     /**
163      * Recursively creates LinuxUsbDevices by fetching information from maps to
164      * populate fields
165      *
166      * @param devPath
167      *            The device node path.
168      * @param vid
169      *            The default (parent) vendor ID
170      * @param pid
171      *            The default (parent) product ID
172      * @return A LinuxUsbDevice corresponding to this device
173      */
174     private static LinuxUsbDevice getDeviceAndChildren(String devPath, String vid, String pid) {
175         String vendorId = MapUtil.getOrDefault(vendorIdMap, devPath, vid);
176         String productId = MapUtil.getOrDefault(productIdMap, devPath, pid);
177         List<String> childPaths = MapUtil.getOrDefault(hubMap, devPath, new ArrayList<String>());
178         List<LinuxUsbDevice> usbDevices = new ArrayList<LinuxUsbDevice>();
179         for (String path : childPaths) {
180             usbDevices.add(getDeviceAndChildren(path, vendorId, productId));
181         }
182         Collections.sort(usbDevices);
183         return new LinuxUsbDevice(MapUtil.getOrDefault(nameMap, devPath, vendorId + ":" + productId),
184                 MapUtil.getOrDefault(vendorMap, devPath, ""), vendorId, productId,
185                 MapUtil.getOrDefault(serialMap, devPath, ""), usbDevices.toArray(new UsbDevice[usbDevices.size()]));
186     }
