127     public static boolean updateDiskStats(HWDiskStore diskStore) {
128         String index = null;
129         HWPartition[] partitions = diskStore.getPartitions();
130         if (partitions.length > 0) {
131             // If a partition exists on this drive, the major property
132             // corresponds to the disk index, so use it.
133             index = Integer.toString(partitions[0].getMajor());
134         } else if (diskStore.getName().startsWith(PHYSICALDRIVE_PREFIX)) {
135             // If no partition exists, Windows reliably uses a name to match the
136             // disk index. That said, the skeptical person might wonder why a
137             // disk has read/write statistics without a partition, and wonder
138             // why this branch is even relevant as an option. The author of this
139             // comment does not have an answer for this valid question.
140             index = diskStore.getName().substring(PHYSICALDRIVE_PREFIX.length(), diskStore.getName().length());
141         } else {
142             // The author of this comment cannot fathom a circumstance in which
143             // the code reaches this point, but just in case it does, here's the
144             // correct response. If you get this log warning, the circumstances
145             // would be of great interest to the project's maintainers.
146             LOG.warn("Couldn't match index for {}", diskStore.getName());
147             return false;
148         }
149         DiskStats stats = queryReadWriteStats(index);
150         if (stats.readMap.containsKey(index)) {
151             diskStore.setReads(MapUtil.getOrDefault(stats.readMap, index, 0L));
152             diskStore.setReadBytes(MapUtil.getOrDefault(stats.readByteMap, index, 0L));
153             diskStore.setWrites(MapUtil.getOrDefault(stats.writeMap, index, 0L));
154             diskStore.setWriteBytes(MapUtil.getOrDefault(stats.writeByteMap, index, 0L));
155             diskStore.setCurrentQueueLength(MapUtil.getOrDefault(stats.queueLengthMap, index, 0L));
156             diskStore.setTimeStamp(stats.timeStamp);
157             diskStore.setTransferTime(diskStore.getTimeStamp()
158                     - MapUtil.getOrDefault(stats.idleTimeMap, index, diskStore.getTimeStamp()));
159             return true;
160         } else
161 
162         {
163             return false;
164         }

168     @Override
169     public HWDiskStore[] getDisks() {
170         List<HWDiskStore> result;
171         result = new ArrayList<HWDiskStore>();
172         DiskStats stats = queryReadWriteStats(null);
173         PartitionMaps maps = queryPartitionMaps();
174 
175         WmiResult<DiskDriveProperty> vals = wmiQueryHandler.queryWMI(diskDriveQuery);
176 
177         for (int i = 0; i < vals.getResultCount(); i++) {
178             HWDiskStorel#HWDiskStore">HWDiskStore ds = new HWDiskStore();
179             ds.setName(WmiUtil.getString(vals, DiskDriveProperty.NAME, i));
180             ds.setModel(String.format("%s %s", WmiUtil.getString(vals, DiskDriveProperty.MODEL, i),
181                     WmiUtil.getString(vals, DiskDriveProperty.MANUFACTURER, i)).trim());
182             // Most vendors store serial # as a hex string; convert
183             ds.setSerial(ParseUtil.hexStringToString(WmiUtil.getString(vals, DiskDriveProperty.SERIALNUMBER, i)));
184             String index = Integer.toString(WmiUtil.getUint32(vals, DiskDriveProperty.INDEX, i));
185             ds.setReads(MapUtil.getOrDefault(stats.readMap, index, 0L));
186             ds.setReadBytes(MapUtil.getOrDefault(stats.readByteMap, index, 0L));
187             ds.setWrites(MapUtil.getOrDefault(stats.writeMap, index, 0L));
188             ds.setWriteBytes(MapUtil.getOrDefault(stats.writeByteMap, index, 0L));
189             ds.setCurrentQueueLength(MapUtil.getOrDefault(stats.queueLengthMap, index, 0L));
190             ds.setTimeStamp(stats.timeStamp);
191             ds.setTransferTime(ds.getTimeStamp() - MapUtil.getOrDefault(stats.idleTimeMap, index, ds.getTimeStamp()));
192             ds.setSize(WmiUtil.getUint64(vals, DiskDriveProperty.SIZE, i));
193             // Get partitions
194             List<HWPartition> partitions = new ArrayList<HWPartition>();
195             List<String> partList = maps.driveToPartitionMap.get(ds.getName());
196             if (partList != null && !partList.isEmpty()) {
197                 for (String part : partList) {
198                     if (maps.partitionMap.containsKey(part)) {
199                         partitions.add(maps.partitionMap.get(part));
200                     }
201                 }
202             }
203             ds.setPartitions(partitions.toArray(new HWPartition[partitions.size()]));
204             // Add to list
205             result.add(ds);
206         }
207         return result.toArray(new HWDiskStore[result.size()]);
208 
209     }

211     /**
212      * Populates the maps for the specified index. If the index is null, populates
213      * all the maps
214      *
215      * @param index
216      *            The index to populate/update maps for
217      */
218     private static DiskStats queryReadWriteStats(String index) {
219         // Create object to hold and return results
220         DiskStats stats = (new WindowsDisks()).new DiskStats();
221 
222         Map<PhysicalDiskProperty, List<Long>> valueMap = physicalDiskPerfCounters.queryValuesWildcard();
223         stats.timeStamp = System.currentTimeMillis();
224         List<String> instances = physicalDiskPerfCounters.getInstancesFromLastQuery();
225         List<Long> readList = valueMap.get(PhysicalDiskProperty.DISKREADSPERSEC);
226         List<Long> readByteList = valueMap.get(PhysicalDiskProperty.DISKREADBYTESPERSEC);
227         List<Long> writeList = valueMap.get(PhysicalDiskProperty.DISKWRITESPERSEC);
228         List<Long> writeByteList = valueMap.get(PhysicalDiskProperty.DISKWRITEBYTESPERSEC);
229         List<Long> queueLengthList = valueMap.get(PhysicalDiskProperty.CURRENTDISKQUEUELENGTH);
230         List<Long> idleTimeList = valueMap.get(PhysicalDiskProperty.PERCENTIDLETIME);
231 
232         if (instances.isEmpty() || readList == null || readByteList == null || writeList == null
233                 || writeByteList == null || queueLengthList == null || idleTimeList == null) {
234             return stats;
235         }
236         for (int i = 0; i < instances.size(); i++) {
237             String name = getIndexFromName(instances.get(i));
238             // If index arg passed, only update passed arg
239             if (index != null && !index.equals(name)) {
240                 continue;
241             }
242             stats.readMap.put(name, readList.get(i));
243             stats.readByteMap.put(name, readByteList.get(i));
244             stats.writeMap.put(name, writeList.get(i));
245             stats.writeByteMap.put(name, writeByteList.get(i));
246             stats.queueLengthMap.put(name, queueLengthList.get(i));
247             stats.idleTimeMap.put(name, idleTimeList.get(i) / 10000L);
248         }
249         return stats;
250     }

252     private PartitionMaps queryPartitionMaps() {
253         // Create object to hold and return results
254         PartitionMaps maps = (new WindowsDisks()).new PartitionMaps();
255 
256         // For Regexp matching DeviceIDs
257         Matcher mAnt;
258         Matcher mDep;
259 
260         // Map drives to partitions
261         WmiResult<DriveToPartitionProperty> drivePartitionMap = wmiQueryHandler.queryWMI(driveToPartitionQuery);
262         for (int i = 0; i < drivePartitionMap.getResultCount(); i++) {
263             mAnt = DEVICE_ID.matcher(WmiUtil.getRefString(drivePartitionMap, DriveToPartitionProperty.ANTECEDENT, i));
264             mDep = DEVICE_ID.matcher(WmiUtil.getRefString(drivePartitionMap, DriveToPartitionProperty.DEPENDENT, i));
265             if (mAnt.matches() && mDep.matches()) {
266                 MapUtil.createNewListIfAbsent(maps.driveToPartitionMap, mAnt.group(1).replaceAll("\\\\\\\\", "\\\\"))
267                         .add(mDep.group(1));
268             }
269         }
270 
271         // Map partitions to logical disks
272         WmiResult<DriveToPartitionProperty> diskPartitionMap = wmiQueryHandler.queryWMI(diskToParitionQuery);
273         for (int i = 0; i < diskPartitionMap.getResultCount(); i++) {
274             mAnt = DEVICE_ID.matcher(WmiUtil.getRefString(diskPartitionMap, DriveToPartitionProperty.ANTECEDENT, i));
275             mDep = DEVICE_ID.matcher(WmiUtil.getRefString(diskPartitionMap, DriveToPartitionProperty.DEPENDENT, i));
276             if (mAnt.matches() && mDep.matches()) {
277                 maps.partitionToLogicalDriveMap.put(mAnt.group(1), mDep.group(1) + "\\");
278             }
279         }
280 
281         // Next, get all partitions and create objects
282         WmiResult<DiskPartitionProperty> hwPartitionQueryMap = wmiQueryHandler.queryWMI(partitionQuery);
283         for (int i = 0; i < hwPartitionQueryMap.getResultCount(); i++) {
284             String deviceID = WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.DEVICEID, i);
285             String logicalDrive = MapUtil.getOrDefault(maps.partitionToLogicalDriveMap, deviceID, "");
286             String uuid = "";
287             if (!logicalDrive.isEmpty()) {
288                 // Get matching volume for UUID
289                 char[] volumeChr = new char[BUFSIZE];
290                 Kernel32.INSTANCE.GetVolumeNameForVolumeMountPoint(logicalDrive, volumeChr, BUFSIZE);
291                 uuid = ParseUtil.parseUuidOrDefault(new String(volumeChr).trim(), "");
292             }
293             maps.partitionMap.put(deviceID,
294                     new HWPartition(WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.NAME, i),
295                             WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.TYPE, i),
296                             WmiUtil.getString(hwPartitionQueryMap, DiskPartitionProperty.DESCRIPTION, i), uuid,
297                             WmiUtil.getUint64(hwPartitionQueryMap, DiskPartitionProperty.SIZE, i),
298                             WmiUtil.getUint32(hwPartitionQueryMap, DiskPartitionProperty.DISKINDEX, i),
299                             WmiUtil.getUint32(hwPartitionQueryMap, DiskPartitionProperty.INDEX, i), logicalDrive));
300         }
301         return maps;
302 
303     }

305     /**
306      * Parse a drive name like "0 C:" to just the index "0"
307      *
308      * @param s
309      *            A drive name to parse
310      * @return The first space-delimited value
311      */
312     private static String getIndexFromName(String s) {
313         if (s.isEmpty()) {
314             return s;
315         }
316         return s.split("\\s")[0];
317     }
