95      public static UsbDevice[] getUsbDevices(boolean tree) {
96          UsbDevice[] devices = getUsbDevices();
97          if (tree) {
98              return devices;
99          }
100         List<UsbDevice> deviceList = new ArrayList<UsbDevice>();
101         // Top level is controllers; they won't be added to the list, but all
102         // their connected devices will be
103         for (UsbDevice device : devices) {
104             addDevicesToList(deviceList, device.getConnectedDevices());
105         }
106         return deviceList.toArray(new UsbDevice[deviceList.size()]);
107     }

109     private static void addDevicesToList(List<UsbDevice> deviceList, UsbDevice[] connectedDevices) {
110         for (UsbDevice device : connectedDevices) {
111             deviceList.add(new WindowsUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
112                     device.getProductId(), device.getSerialNumber(), new UsbDevice[0]));
113             addDevicesToList(deviceList, device.getConnectedDevices());
114         }
115     }

117     private static UsbDevice[] getUsbDevices() {
118         // Map to build the recursive tree structure
119         Map<String, List<String>> deviceTreeMap = new HashMap<String, List<String>>();
120         // Track devices seen in the process
121         Set<String> devicesSeen = new HashSet<String>();
122 
123         // Navigate the device tree to track what devices are present
124         List<WindowsUsbDevice> controllerDevices = new ArrayList<WindowsUsbDevice>();
125         WmiQueryHandler wmiQueryHandler = WmiQueryHandler.createInstance();
126         List<String> controllerDeviceIdList = getControllerDeviceIdList(wmiQueryHandler);
127         for (String controllerDeviceId : controllerDeviceIdList) {
128             putChildrenInDeviceTree(controllerDeviceId, 0, deviceTreeMap, devicesSeen);
129         }
130         // remove from cache if no longer present and add to cache if present
131         // but not in cache
132         updateDeviceCache(devicesSeen, wmiQueryHandler);
133         // recursively build results
134         for (String controllerDeviceId : controllerDeviceIdList) {
135             WindowsUsbDevice deviceAndChildren = getDeviceAndChildren(controllerDeviceId, "0000", "0000",
136                     deviceTreeMap);
137             if (deviceAndChildren != null) {
138                 controllerDevices.add(deviceAndChildren);
139             }
140         }
141         return controllerDevices.toArray(new WindowsUsbDevice[controllerDevices.size()]);
142     }

144     private static void updateDeviceCache(Set<String> devicesSeen, WmiQueryHandler wmiQueryHandler) {
145         // The deviceCache stores UsbDevices that we have already returned
146         // Remove devices no longer in tree
147         Set<String> devicesToRemove = new HashSet<String>(usbDeviceCache.keySet());
148         devicesToRemove.removeAll(devicesSeen);
149         for (String deviceID : devicesToRemove) {
150             usbDeviceCache.remove(deviceID);
151             LOG.debug("Removing {} from USB device cache.", deviceID);
152         }
153 
154         // Add devices not in the tree
155         Set<String> devicesToAdd = new HashSet<String>(devicesSeen);
156         devicesToAdd.removeAll(usbDeviceCache.keySet());
157         if (!devicesToAdd.isEmpty()) {
158             StringBuilder sb = new StringBuilder();
159             boolean first = true;
160             for (String deviceID : devicesToAdd) {
161                 if (first) {
162                     sb.append(" WHERE (PnPDeviceID=\"");
163                     first = false;
164                 } else {
165                     sb.append(" OR (PnPDeviceID=\"");
166                 }
167                 sb.append(deviceID).append("\")");
168             }
169             String whereClause = sb.toString();
170             // Query Win32_PnPEntity to populate the maps
171             PNPENTITY_QUERY.setWmiClassName(PNPENTITY_BASE_CLASS + whereClause);
172             WmiResult<PnPEntityProperty> pnpEntity = wmiQueryHandler.queryWMI(PNPENTITY_QUERY);
173             for (int i = 0; i < pnpEntity.getResultCount(); i++) {
174                 String pnpDeviceID = WmiUtil.getString(pnpEntity, PnPEntityProperty.PNPDEVICEID, i);
175                 String name = WmiUtil.getString(pnpEntity, PnPEntityProperty.NAME, i);
176                 String vendor = WmiUtil.getString(pnpEntity, PnPEntityProperty.MANUFACTURER, i);
177                 WindowsUsbDeviceUsbDevice.html#WindowsUsbDevice">WindowsUsbDevice device = new WindowsUsbDevice(name, vendor, null, null, "", new WindowsUsbDevice[0]);
178                 usbDeviceCache.put(pnpDeviceID, device);
179                 LOG.debug("Adding {} to USB device cache.", pnpDeviceID);
180             }
181             // Get serial # for disk drives or other physical media
182             DISKDRIVE_QUERY.setWmiClassName(DISKDRIVE_BASE_CLASS + whereClause);
183             WmiResult<DiskDriveProperty> serialNumber = wmiQueryHandler.queryWMI(DISKDRIVE_QUERY);
184             for (int i = 0; i < serialNumber.getResultCount(); i++) {
185                 String pnpDeviceID = WmiUtil.getString(serialNumber, DiskDriveProperty.PNPDEVICEID, i);
186                 if (usbDeviceCache.containsKey(pnpDeviceID)) {
187                     WindowsUsbDevice device = usbDeviceCache.get(pnpDeviceID);
188                     device.serialNumber = ParseUtil
189                             .hexStringToString(WmiUtil.getString(serialNumber, DiskDriveProperty.SERIALNUMBER, i));
190                 }
191             }
192         }
193     }

195     /**
196      * Navigates the Device Tree to place all children PNPDeviceIDs into the map for
197      * the specified deviceID. Recursively adds children's children, etc.
198      * 
199      * @param deviceId
200      *            The device to add respective children to the map
201      * @param deviceInstance
202      *            The device instance (devnode handle), if known. If set to 0, the
203      *            code will search for a match.
204      * @param deviceTreeMap
205      *            The overall device tree map that starts at the controllers
206      * @param devicesSeen
207      *            Devices we've seen so we can add/remove from cache later
208      */
209     private static void putChildrenInDeviceTree(String deviceId, int deviceInstance,
210             Map<String, List<String>> deviceTreeMap, Set<String> devicesSeen) {
211         devicesSeen.add(deviceId);
212         // If no devInst provided, find it
213         int devInst = deviceInstance;
214         if (devInst == 0) {
215             IntByReference pdnDevInst = new IntByReference();
216             Cfgmgr32.INSTANCE.CM_Locate_DevNode(pdnDevInst, deviceId, 0);
217             devInst = pdnDevInst.getValue();
218         }
219         // Now iterate the children. Call CM_Get_Child to get first child
220         IntByReference child = new IntByReference();
221         if (0 == Cfgmgr32.INSTANCE.CM_Get_Child(child, devInst, 0)) {
222             // Add first child to a list
223             List<String> childList = new ArrayList<String>();
224             String childId = Cfgmgr32Util.CM_Get_Device_ID(child.getValue());
225             childList.add(childId);
226             deviceTreeMap.put(deviceId, childList);
227             putChildrenInDeviceTree(childId, child.getValue(), deviceTreeMap, devicesSeen);
228             // Find any other children
229             IntByReference sibling = new IntByReference();
230             while (0 == Cfgmgr32.INSTANCE.CM_Get_Sibling(sibling, child.getValue(), 0)) {
231                 // Add to the list
232                 String siblingId = Cfgmgr32Util.CM_Get_Device_ID(sibling.getValue());
233                 deviceTreeMap.get(deviceId).add(siblingId);
234                 putChildrenInDeviceTree(siblingId, sibling.getValue(), deviceTreeMap, devicesSeen);
235                 // Make this sibling the new child to find other siblings
236                 child = sibling;
237             }
238         }
239     }

241     /**
242      * Recursively creates WindowsUsbDevices by fetching information from maps to
243      * populate fields
244      *
245      * @param hubDeviceId
246      *            The PNPdeviceID of this device.
247      * @param vid
248      *            The default (parent) vendor ID
249      * @param pid
250      *            The default (parent) product ID
251      * @param deviceTreeMap
252      * @return A WindowsUsbDevice corresponding to this deviceID, or null if unable
253      *         to find
254      */
255     private static WindowsUsbDevice getDeviceAndChildren(String hubDeviceId, String vid, String pid,
256             Map<String, List<String>> deviceTreeMap) {
257         String vendorId = vid;
258         String productId = pid;
259         Matcher m = VENDOR_PRODUCT_ID.matcher(hubDeviceId);
260         if (m.matches()) {
261             vendorId = m.group(1).toLowerCase();
262             productId = m.group(2).toLowerCase();
263         }
264         List<String> pnpDeviceIds = MapUtil.getOrDefault(deviceTreeMap, hubDeviceId, new ArrayList<String>());
265         List<WindowsUsbDevice> usbDevices = new ArrayList<WindowsUsbDevice>();
266         for (String pnpDeviceId : pnpDeviceIds) {
267             WindowsUsbDevice deviceAndChildren = getDeviceAndChildren(pnpDeviceId, vendorId, productId, deviceTreeMap);
268             if (deviceAndChildren != null) {
269                 usbDevices.add(deviceAndChildren);
270             }
271         }
272         Collections.sort(usbDevices);
273         if (usbDeviceCache.containsKey(hubDeviceId)) {
274             WindowsUsbDevice device = usbDeviceCache.get(hubDeviceId);
275             if (device.name.isEmpty()) {
276                 device.name = vendorId + ":" + productId;
277             }
278             device.vendorId = vendorId;
279             device.productId = productId;
280             device.connectedDevices = usbDevices.toArray(new WindowsUsbDevice[usbDevices.size()]);
281             return device;
282         }
283         return null;
284     }

286     /**
287      * Queries the USB Controller list, which doesn't change so we only need to
288      * query it once
289      * 
290      * @param wmiQueryHandler
291      * 
292      * @return A list of Strings of USB Controller PNPDeviceIDs
293      */
294     private static List<String> getControllerDeviceIdList(WmiQueryHandler wmiQueryHandler) {
295         if (controllerDeviceIds == null) {
296             controllerDeviceIds = new ArrayList<String>();
297             // One time lookup of USB Controller PnP Device IDs which don't
298             // change
299             WmiQuery<USBControllerProperty> usbControllerQuery = new WmiQuery<USBControllerProperty>(
300                     "Win32_USBController", USBControllerProperty.class);
301             WmiResult<USBControllerProperty> usbController = wmiQueryHandler.queryWMI(usbControllerQuery);
302             for (int i = 0; i < usbController.getResultCount(); i++) {
303                 controllerDeviceIds.add(WmiUtil.getString(usbController, USBControllerProperty.PNPDEVICEID, i));
304             }
305         }
306         return controllerDeviceIds;
307     }
