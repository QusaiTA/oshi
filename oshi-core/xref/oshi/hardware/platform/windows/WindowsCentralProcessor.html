217     /**
218      * Initializes Class variables
219      */
220     private void initVars() {
221         final String cpuRegistryRoot = "HARDWARE\\DESCRIPTION\\System\\CentralProcessor\\";
222         String[] processorIds = Advapi32Util.registryGetKeys(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryRoot);
223         if (processorIds.length > 0) {
224             String cpuRegistryPath = cpuRegistryRoot + processorIds[0];
225             setVendor(Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
226                     "VendorIdentifier"));
227             setName(Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath,
228                     "ProcessorNameString"));
229             setIdentifier(
230                     Advapi32Util.registryGetStringValue(WinReg.HKEY_LOCAL_MACHINE, cpuRegistryPath, "Identifier"));
231         }
232         SYSTEM_INFO sysinfo = new SYSTEM_INFO();
233         Kernel32.INSTANCE.GetNativeSystemInfo(sysinfo);
234         if (sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 9 // PROCESSOR_ARCHITECTURE_AMD64
235                 || sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 6) { // PROCESSOR_ARCHITECTURE_IA64
236             setCpu64(true);
237         } else if (sysinfo.processorArchitecture.pi.wProcessorArchitecture.intValue() == 0) { // PROCESSOR_ARCHITECTURE_INTEL
238             setCpu64(false);
239         }
240 
241         WmiQuery<ProcessorProperty> processorIdQuery = new WmiQuery<ProcessorProperty>("Win32_Processor",
242                 ProcessorProperty.class);
243         WmiResult<ProcessorProperty> processorId = WmiQueryHandler.createInstance().queryWMI(processorIdQuery);
244         if (processorId.getResultCount() > 0) {
245             setProcessorID(WmiUtil.getString(processorId, ProcessorProperty.PROCESSORID, 0));
246         }
247     }

249     /**
250      * Updates logical and physical processor counts
251      */
252     @Override
253     protected void calculateProcessorCounts() {
254         // Get number of logical processors
255         SYSTEM_INFO sysinfo = new SYSTEM_INFO();
256         Kernel32.INSTANCE.GetSystemInfo(sysinfo);
257         this.logicalProcessorCount = sysinfo.dwNumberOfProcessors.intValue();
258 
259         // Get number of physical processors
260         WinNT.SYSTEM_LOGICAL_PROCESSOR_INFORMATION[] processors = Kernel32Util.getLogicalProcessorInformation();
261         for (SYSTEM_LOGICAL_PROCESSOR_INFORMATION proc : processors) {
262             if (proc.relationship == WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorPackage) {
263                 this.physicalPackageCount++;
264             }
265             if (proc.relationship == WinNT.LOGICAL_PROCESSOR_RELATIONSHIP.RelationProcessorCore) {
266                 this.physicalProcessorCount++;
267             }
268         }
269     }

271     /**
272      * {@inheritDoc}
273      */
274     @Override
275     public long[] getSystemCpuLoadTicks() {
276         long[] ticks = new long[TickType.values().length];
277         WinBase.FILETIME lpIdleTime = new WinBase.FILETIME();
278         WinBase.FILETIME lpKernelTime = new WinBase.FILETIME();
279         WinBase.FILETIME lpUserTime = new WinBase.FILETIME();
280         if (!Kernel32.INSTANCE.GetSystemTimes(lpIdleTime, lpKernelTime, lpUserTime)) {
281             LOG.error("Failed to update system idle/kernel/user times. Error code: {}", Native.getLastError());
282             return ticks;
283         }
284         // IOwait:
285         // Windows does not measure IOWait.
286 
287         // IRQ and ticks:
288         // Percent time raw value is cumulative 100NS-ticks
289         // Divide by 10000 to get milliseconds
290 
291         Map<SystemTickCountProperty, Long> valueMap = this.systemTickPerfCounters.queryValues();
292         ticks[TickType.IRQ.getIndex()] = MapUtil.getOrDefault(valueMap, SystemTickCountProperty.PERCENTINTERRUPTTIME,
293                 0L) / 10000L;
294         ticks[TickType.SOFTIRQ.getIndex()] = MapUtil.getOrDefault(valueMap, SystemTickCountProperty.PERCENTDPCTIME, 0L)
295                 / 10000L;
296 
297         ticks[TickType.IDLE.getIndex()] = lpIdleTime.toDWordLong().longValue() / 10000L;
298         ticks[TickType.SYSTEM.getIndex()] = lpKernelTime.toDWordLong().longValue() / 10000L
299                 - ticks[TickType.IDLE.getIndex()];
300         ticks[TickType.USER.getIndex()] = lpUserTime.toDWordLong().longValue() / 10000L;
301         // Additional decrement to avoid double counting in the total array
302         ticks[TickType.SYSTEM.getIndex()] -= ticks[TickType.IRQ.getIndex()] + ticks[TickType.SOFTIRQ.getIndex()];
303         return ticks;
304     }

306     /**
307      * {@inheritDoc}
308      */
309     @Override
310     public double[] getSystemLoadAverage(int nelem) {
311         if (nelem < 1 || nelem > 3) {
312             throw new IllegalArgumentException("Must include from one to three elements.");
313         }
314         double[] average = new double[nelem];
315         // Windows doesn't have load average
316         for (int i = 0; i < average.length; i++) {
317             average[i] = -1;
318         }
319         return average;
320     }

322     /**
323      * {@inheritDoc}
324      */
325     @Override
326     public long[][] getProcessorCpuLoadTicks() {
327         Map<ProcessorTickCountProperty, List<Long>> valueMap = this.processorTickPerfCounters.queryValuesWildcard();
328         List<String> instances = this.processorTickPerfCounters.getInstancesFromLastQuery();
329         List<Long> systemList = valueMap.get(ProcessorTickCountProperty.PERCENTPRIVILEGEDTIME);
330         List<Long> userList = valueMap.get(ProcessorTickCountProperty.PERCENTUSERTIME);
331         List<Long> irqList = valueMap.get(ProcessorTickCountProperty.PERCENTINTERRUPTTIME);
332         List<Long> softIrqList = valueMap.get(ProcessorTickCountProperty.PERCENTDPCTIME);
333         // % Processor Time is actually Idle time
334         List<Long> idleList = valueMap.get(ProcessorTickCountProperty.PERCENTPROCESSORTIME);
335 
336         long[][] ticks = new long[this.logicalProcessorCount][TickType.values().length];
337         if (instances.isEmpty() || systemList == null || userList == null || irqList == null || softIrqList == null
338                 || idleList == null) {
339             return ticks;
340         }
341         for (int p = 0; p < instances.size(); p++) {
342             int cpu = ParseUtil.parseIntOrDefault(instances.get(p), 0);
343             if (cpu >= this.logicalProcessorCount) {
344                 continue;
345             }
346             ticks[cpu][TickType.SYSTEM.getIndex()] = systemList.get(cpu);
347             ticks[cpu][TickType.USER.getIndex()] = userList.get(cpu);
348             ticks[cpu][TickType.IRQ.getIndex()] = irqList.get(cpu);
349             ticks[cpu][TickType.SOFTIRQ.getIndex()] = softIrqList.get(cpu);
350             ticks[cpu][TickType.IDLE.getIndex()] = idleList.get(cpu);
351 
352             // Additional decrement to avoid double counting in the
353             // total array
354             ticks[cpu][TickType.SYSTEM.getIndex()] -= ticks[cpu][TickType.IRQ.getIndex()]
355                     + ticks[cpu][TickType.SOFTIRQ.getIndex()];
356 
357             // Raw value is cumulative 100NS-ticks
358             // Divide by 10000 to get milliseconds
359             ticks[cpu][TickType.SYSTEM.getIndex()] /= 10000L;
360             ticks[cpu][TickType.USER.getIndex()] /= 10000L;
361             ticks[cpu][TickType.IRQ.getIndex()] /= 10000L;
362             ticks[cpu][TickType.SOFTIRQ.getIndex()] /= 10000L;
363             ticks[cpu][TickType.IDLE.getIndex()] /= 10000L;
364         }
365         // Skipping nice and IOWait, they'll stay 0
366         return ticks;
367     }

369     /**
370      * {@inheritDoc}
371      */
372     @Override
373     public long getSystemUptime() {
374         // Uptime is in seconds so divide milliseconds
375         // GetTickCount64 requires Vista (6.0) or later
376         if (IS_VISTA_OR_GREATER) {
377             return Kernel32.INSTANCE.GetTickCount64() / 1000L;
378         } else {
379             // 32 bit rolls over at ~ 49 days
380             return Kernel32.INSTANCE.GetTickCount() / 1000L;
381         }
382     }
