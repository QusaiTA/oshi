40  /**

45  public class WindowsGlobalMemory extends AbstractGlobalMemory {

47      private static final long serialVersionUID = 1L;
48  
49      private static final Logger LOG = LoggerFactory.getLogger(WindowsGlobalMemory.class);

51      private long lastUpdate = 0L;

53      /*
54       * For pages in/out
55       */
56      public enum PageSwapProperty implements PdhCounterProperty {
57          PAGESINPUTPERSEC(null, "Pages Input/sec"), //
58          PAGESOUTPUTPERSEC(null, "Pages Output/sec");
59  
60          private final String instance;
61          private final String counter;
62  
63          PageSwapProperty(String instance, String counter) {
64              this.instance = instance;
65              this.counter = counter;
66          }
67  
68          /**
69           * {@inheritDoc}
70           */
71          @Override
72          public String getInstance() {
73              return instance;
74          }
75  
76          /**
77           * {@inheritDoc}
78           */
79          @Override
80          public String getCounter() {
81              return counter;
82          }
83      }
84  
85      private transient PerfCounterQuery<PageSwapProperty> memoryPerfCounters = new PerfCounterQuery<PageSwapProperty>(
86              PageSwapProperty.class, "Memory", "Win32_PerfRawData_PerfOS_Memory");
87  
88      /*
89       * For swap file usage
90       */
91      enum PagingPercentProperty implements PdhCounterProperty {
92          PERCENTUSAGE(PerfCounterQuery.TOTAL_INSTANCE, "% Usage");
93  
94          private final String instance;
95          private final String counter;
96  
97          PagingPercentProperty(String instance, String counter) {
98              this.instance = instance;
99              this.counter = counter;
100         }
101 
102         /**
103          * {@inheritDoc}
104          */
105         @Override
106         public String getInstance() {
107             return instance;
108         }
109 
110         /**
111          * {@inheritDoc}
112          */
113         @Override
114         public String getCounter() {
115             return counter;
116         }
117     }
118 
119     private transient PerfCounterQuery<PagingPercentProperty> pagingPerfCounters = new PerfCounterQuery<PagingPercentProperty>(
120             PagingPercentProperty.class, "Paging File", "Win32_PerfRawData_PerfOS_PagingFile");
121 
122     /**
123      * Update the performance information no more frequently than every 100ms
124      */
125     @Override
126     protected void updateMeminfo() {
127         PERFORMANCE_INFORMATION perfInfo = new PERFORMANCE_INFORMATION();
128         long now = System.currentTimeMillis();
129         if (now - this.lastUpdate > 100) {
130             if (!Psapi.INSTANCE.GetPerformanceInfo(perfInfo, perfInfo.size())) {
131                 LOG.error("Failed to get Performance Info. Error code: {}", Kernel32.INSTANCE.GetLastError());
132                 return;
133             }
134             this.pageSize = perfInfo.PageSize.longValue();
135             this.memAvailable = this.pageSize * perfInfo.PhysicalAvailable.longValue();
136             this.memTotal = this.pageSize * perfInfo.PhysicalTotal.longValue();
137             this.swapTotal = this.pageSize * (perfInfo.CommitLimit.longValue() - perfInfo.PhysicalTotal.longValue());
138             if (this.swapTotal > 0) {
139                 updateSwapCounters();
140             }
141             this.lastUpdate = now;
142         }
143     }
144 
145     private void updateSwapCounters() {
146         Map<PageSwapProperty, Long> valueMap = this.memoryPerfCounters.queryValues();
147         this.swapPagesIn = MapUtil.getOrDefault(valueMap, PageSwapProperty.PAGESINPUTPERSEC, 0L);
148         this.swapPagesOut = MapUtil.getOrDefault(valueMap, PageSwapProperty.PAGESOUTPUTPERSEC, 0L);
149     }
150 
151     /**
152      * {@inheritDoc}
153      */
154     @Override
155     protected void updateSwap() {
156         updateMeminfo();
157         Map<PagingPercentProperty, Long> valueMap = this.pagingPerfCounters.queryValues();
158         this.swapUsed = MapUtil.getOrDefault(valueMap, PagingPercentProperty.PERCENTUSAGE, 0L);
159     }
