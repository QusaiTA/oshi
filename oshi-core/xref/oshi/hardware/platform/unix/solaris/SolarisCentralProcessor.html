65      private void initVars() {
66          // Get first result
67          Kstat ksp = KstatUtil.kstatLookup("cpu_info", -1, null);
68          // Set values
69          if (ksp != null && KstatUtil.kstatRead(ksp)) {
70              setVendor(KstatUtil.kstatDataLookupString(ksp, "vendor_id"));
71              setName(KstatUtil.kstatDataLookupString(ksp, "brand"));
72              setStepping(KstatUtil.kstatDataLookupString(ksp, "stepping"));
73              setModel(KstatUtil.kstatDataLookupString(ksp, "model"));
74              setFamily(KstatUtil.kstatDataLookupString(ksp, "family"));
75          }
76          setCpu64("64".equals(ExecutingCommand.getFirstAnswer("isainfo -b").trim()));
77          setProcessorID(getProcessorID(getStepping(), getModel(), getFamily()));
78  
79      }

81      /**
82       * Updates logical and physical processor counts from psrinfo
83       */
84      @Override
85      protected void calculateProcessorCounts() {
86          List<Kstat> kstats = KstatUtil.kstatLookupAll("cpu_info", -1, null);
87          Set<String> chipIDs = new HashSet<String>();
88          Set<String> coreIDs = new HashSet<String>();
89          this.logicalProcessorCount = 0;
90          for (Kstat ksp : kstats) {
91              if (ksp != null && KstatUtil.kstatRead(ksp)) {
92                  this.logicalProcessorCount++;
93                  String chipId = KstatUtil.kstatDataLookupString(ksp, "chip_id");
94                  String coreId = KstatUtil.kstatDataLookupString(ksp, "core_id");
95                  chipIDs.add(chipId);
96                  coreIDs.add(coreId + ":" + chipId);
97              }
98          }
99  
100         this.physicalPackageCount = chipIDs.size();
101         if (this.physicalPackageCount < 1) {
102             LOG.error("Couldn't find physical package count. Assuming 1.");
103             this.physicalPackageCount = 1;
104         }
105         this.physicalProcessorCount = coreIDs.size();
106         if (this.physicalProcessorCount < 1) {
107             LOG.error("Couldn't find physical processor count. Assuming 1.");
108             this.physicalProcessorCount = 1;
109         }
110         if (this.logicalProcessorCount < 1) {
111             LOG.error("Couldn't find logical processor count. Assuming 1.");
112             this.logicalProcessorCount = 1;
113         }
114     }

116     /**
117      * {@inheritDoc}
118      */
119     @Override
120     public synchronized long[] getSystemCpuLoadTicks() {
121         long[] ticks = new long[TickType.values().length];
122         // Average processor ticks
123         long[][] procTicks = getProcessorCpuLoadTicks();
124         for (int i = 0; i < ticks.length; i++) {
125             for (long[] procTick : procTicks) {
126                 ticks[i] += procTick[i];
127             }
128             ticks[i] /= procTicks.length;
129         }
130         return ticks;
131     }

133     /**
134      * {@inheritDoc}
135      */
136     @Override
137     public double[] getSystemLoadAverage(int nelem) {
138         if (nelem < 1 || nelem > 3) {
139             throw new IllegalArgumentException("Must include from one to three elements.");
140         }
141         double[] average = new double[nelem];
142         int retval = Libc.INSTANCE.getloadavg(average, nelem);
143         if (retval < nelem) {
144             for (int i = Math.max(retval, 0); i < average.length; i++) {
145                 average[i] = -1d;
146             }
147         }
148         return average;
149     }

151     /**
152      * {@inheritDoc}
153      */
154     @Override
155     public long[][] getProcessorCpuLoadTicks() {
156         long[][] ticks = new long[this.logicalProcessorCount][TickType.values().length];
157         int cpu = -1;
158         for (Kstat ksp : KstatUtil.kstatLookupAll("cpu", -1, "sys")) {
159             // This is a new CPU
160             if (++cpu >= ticks.length) {
161                 // Shouldn't happen
162                 break;
163             }
164             if (KstatUtil.kstatRead(ksp)) {
165                 ticks[cpu][TickType.IDLE.getIndex()] = KstatUtil.kstatDataLookupLong(ksp, "cpu_ticks_idle");
166                 ticks[cpu][TickType.SYSTEM.getIndex()] = KstatUtil.kstatDataLookupLong(ksp, "cpu_ticks_kernel");
167                 ticks[cpu][TickType.USER.getIndex()] = KstatUtil.kstatDataLookupLong(ksp, "cpu_ticks_user");
168             }
169         }
170         return ticks;
171     }

173     /**
174      * {@inheritDoc}
175      */
176     @Override
177     public long getSystemUptime() {
178         Kstat ksp = KstatUtil.kstatLookup("unix", 0, "system_misc");
179         if (ksp == null) {
180             return 0L;
181         }
182         // Snap Time is in nanoseconds; divide for seconds
183         return ksp.ks_snaptime / 1000000000L;
184     }

186     /**
187      * Fetches the ProcessorID by encoding the stepping, model, family, and feature
188      * flags.
189      *
190      * @param stepping
191      * @param model
192      * @param family
193      * @return The Processor ID string
194      */
195     private String getProcessorID(String stepping, String model, String family) {
196         List<String> isainfo = ExecutingCommand.runNative("isainfo -v");
197         StringBuilder flags = new StringBuilder();
198         for (String line : isainfo) {
199             if (line.startsWith("32-bit")) {
200                 break;
201             } else if (!line.startsWith("64-bit")) {
202                 flags.append(' ').append(line.trim());
203             }
204         }
205         return createProcessorID(stepping, model, family, ParseUtil.whitespaces.split(flags.toString().toLowerCase()));
206     }

208     /**
209      * {@inheritDoc}
210      */
211     @Override
212     public long getContextSwitches() {
213         long swtch = 0;
214         List<String> kstat = ExecutingCommand.runNative("kstat -p cpu_stat:::/pswitch\\\\|inv_swtch/");
215         for (String s : kstat) {
216             swtch += ParseUtil.parseLastLong(s, 0L);
217         }
218         return swtch > 0 ? swtch : -1L;
219     }
220 
221     /**
222      * {@inheritDoc}
223      */
224     @Override
225     public long getInterrupts() {
226         long intr = 0;
227         List<String> kstat = ExecutingCommand.runNative("kstat -p cpu_stat:::/intr/");
228         for (String s : kstat) {
229             intr += ParseUtil.parseLastLong(s, 0L);
230         }
231         return intr > 0 ? intr : -1L;
232     }
