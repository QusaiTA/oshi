38  public class FreeBsdUsbDevice extends AbstractUsbDevice {

40      private static final long serialVersionUID = 2L;

42      /*
43       * Maps to store information using node # as the key
44       */
45      private static Map<String, String> nameMap = new HashMap<String, String>();
46      private static Map<String, String> vendorMap = new HashMap<String, String>();
47      private static Map<String, String> vendorIdMap = new HashMap<String, String>();
48      private static Map<String, String> productIdMap = new HashMap<String, String>();
49      private static Map<String, String> serialMap = new HashMap<String, String>();
50      private static Map<String, String> parentMap = new HashMap<String, String>();
51      private static Map<String, List<String>> hubMap = new HashMap<String, List<String>>();
52  
53      public FreeBsdUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
54              UsbDevice[] connectedDevices) {
55          super(name, vendor, vendorId, productId, serialNumber, connectedDevices);
56      }

58      /**
59       * {@inheritDoc}
60       */
61      public static UsbDevice[] getUsbDevices(boolean tree) {
62  
63          UsbDevice[] devices = getUsbDevices();
64          if (tree) {
65              return devices;
66          }
67          List<UsbDevice> deviceList = new ArrayList<UsbDevice>();
68          // Top level is controllers; they won't be added to the list, but all
69          // their connected devices will be
70          for (UsbDevice device : devices) {
71              deviceList.add(new FreeBsdUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
72                      device.getProductId(), device.getSerialNumber(), new FreeBsdUsbDevice[0]));
73              addDevicesToList(deviceList, device.getConnectedDevices());
74          }
75          return deviceList.toArray(new UsbDevice[deviceList.size()]);
76      }
77  
78      private static void addDevicesToList(List<UsbDevice> deviceList, UsbDevice[] connectedDevices) {
79          for (UsbDevice device : connectedDevices) {
80              deviceList.add(device);
81              addDevicesToList(deviceList, device.getConnectedDevices());
82          }
83      }
84  
85      private static UsbDevice[] getUsbDevices() {
86          // Empty out maps
87          // Empty out maps
88          nameMap.clear();
89          vendorMap.clear();
90          vendorIdMap.clear();
91          productIdMap.clear();
92          serialMap.clear();
93          hubMap.clear();
94          parentMap.clear();
95  
96          // Enumerate all devices and build information maps. This will build the
97          // entire device tree; we will identify the controllers as the parents
98          // of the usbus entries and eventually only populate the returned
99          // results with those
100         List<String> devices = ExecutingCommand.runNative("lshal");
101         if (devices.isEmpty()) {
102             // TODO usbconfig, works as root
103             return new FreeBsdUsbDevice[0];
104         }
105         // For each item enumerated, store information in the maps
106         String key = "";
107         List<String> usBuses = new ArrayList<String>();
108         for (String line : devices) {
109             // udi = ... identifies start of a new tree
110             if (line.startsWith("udi =")) {
111                 // Remove indent for key
112                 key = ParseUtil.getSingleQuoteStringValue(line);
113                 continue;
114             } else if (key.isEmpty()) {
115                 // Ignore everything preceding the first node
116                 continue;
117             }
118             // We are currently processing for node identified by key. Save
119             // approrpriate variables to maps.
120             line = line.trim();
121             if (line.isEmpty()) {
122                 continue;
123             } else if (line.startsWith("freebsd.driver =")
124                     && "usbus".equals(ParseUtil.getSingleQuoteStringValue(line))) {
125                 usBuses.add(key);
126             } else if (line.contains(".parent =")) {
127                 String parent = ParseUtil.getSingleQuoteStringValue(line);
128                 // If this is interface of parent, skip
129                 if (key.replace(parent, "").startsWith("_if")) {
130                     continue;
131                 }
132                 // Store parent for later usbus-skipping
133                 parentMap.put(key, parent);
134                 // Add this key to the parent's hubmap list
135                 MapUtil.createNewListIfAbsent(hubMap, parent).add(key);
136             } else if (line.contains(".vendor =")) {
137                 vendorMap.put(key, ParseUtil.getSingleQuoteStringValue(line));
138             } else if (line.contains(".product =")) {
139                 nameMap.put(key, ParseUtil.getSingleQuoteStringValue(line));
140             } else if (line.contains(".serial =")) {
141                 String serial = ParseUtil.getSingleQuoteStringValue(line);
142                 serialMap.put(key,
143                         serial.startsWith("0x") ? ParseUtil.hexStringToString(serial.replace("0x", "")) : serial);
144             } else if (line.contains(".vendor_id =") || line.contains(".product_id =")) {
145                 vendorIdMap.put(key, String.format("%04x", ParseUtil.getFirstIntValue(line)));
146             }
147         }
148 
149         // Build tree and return
150         List<UsbDevice> controllerDevices = new ArrayList<UsbDevice>();
151         for (String usbus : usBuses) {
152             // Skip the usbuses: make their parents the controllers and replace
153             // parents' children with the buses' children
154             String parent = parentMap.get(usbus);
155             hubMap.put(parent, hubMap.get(usbus));
156             controllerDevices.add(getDeviceAndChildren(parent, "0000", "0000"));
157         }
158         return controllerDevices.toArray(new UsbDevice[controllerDevices.size()]);
159     }
160 
161     /**
162      * Recursively creates SolarisUsbDevices by fetching information from maps to
163      * populate fields
164      *
165      * @param devPath
166      *            The device node path.
167      * @param vid
168      *            The default (parent) vendor ID
169      * @param pid
170      *            The default (parent) product ID
171      * @return A SolarisUsbDevice corresponding to this device
172      */
173     private static FreeBsdUsbDevice getDeviceAndChildren(String devPath, String vid, String pid) {
174         String vendorId = MapUtil.getOrDefault(vendorIdMap, devPath, vid);
175         String productId = MapUtil.getOrDefault(productIdMap, devPath, pid);
176         List<String> childPaths = MapUtil.getOrDefault(hubMap, devPath, new ArrayList<String>());
177         List<FreeBsdUsbDevice> usbDevices = new ArrayList<FreeBsdUsbDevice>();
178         for (String path : childPaths) {
179             usbDevices.add(getDeviceAndChildren(path, vendorId, productId));
180         }
181         Collections.sort(usbDevices);
182         return new FreeBsdUsbDevice(MapUtil.getOrDefault(nameMap, devPath, vendorId + ":" + productId), "", vendorId,
183                 productId, "", usbDevices.toArray(new UsbDevice[usbDevices.size()]));
184     }
