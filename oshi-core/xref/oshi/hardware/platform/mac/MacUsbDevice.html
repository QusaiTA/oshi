52       * Maps to store information using RegistryEntryID as the key
53       */
54      private static Map<Long, String> nameMap = new HashMap<Long, String>();
55      private static Map<Long, String> vendorMap = new HashMap<Long, String>();
56      private static Map<Long, String> vendorIdMap = new HashMap<Long, String>();
57      private static Map<Long, String> productIdMap = new HashMap<Long, String>();
58      private static Map<Long, String> serialMap = new HashMap<Long, String>();
59      private static Map<Long, List<Long>> hubMap = new HashMap<Long, List<Long>>();

61      public MacUsbDevice(String name, String vendor, String vendorId, String productId, String serialNumber,
62              UsbDevice[] connectedDevices) {
63          super(name, vendor, vendorId, productId, serialNumber, connectedDevices);
64      }

66      /**
67       * {@inheritDoc}
68       */
69      public static UsbDevice[] getUsbDevices(boolean tree) {
70          UsbDevice[] devices = getUsbDevices();
71          if (tree) {
72              return devices;
73          }
74          List<UsbDevice> deviceList = new ArrayList<UsbDevice>();
75          // Top level is controllers; they won't be added to the list, but all
76          // their connected devices will be
77          for (UsbDevice device : devices) {
78              addDevicesToList(deviceList, device.getConnectedDevices());
79          }
80          return deviceList.toArray(new UsbDevice[deviceList.size()]);
81      }
82  
83      private static void addDevicesToList(List<UsbDevice> deviceList, UsbDevice[] connectedDevices) {
84          for (UsbDevice device : connectedDevices) {
85              deviceList.add(new MacUsbDevice(device.getName(), device.getVendor(), device.getVendorId(),
86                      device.getProductId(), device.getSerialNumber(), new MacUsbDevice[0]));
87              addDevicesToList(deviceList, device.getConnectedDevices());
88          }
89      }

91      private static UsbDevice[] getUsbDevices() {
92          // Reusable buffer for getting IO name strings
93          Pointer buffer = new Memory(128); // io_name_t is char[128]
94  
95          // Empty out maps
96          nameMap.clear();
97          vendorMap.clear();
98          vendorIdMap.clear();
99          productIdMap.clear();
100         serialMap.clear();
101         hubMap.clear();
102 
103         // Iterate over USB Controllers. All devices are children of one of
104         // these controllers in the "IOService" plane
105         List<Long> usbControllers = new ArrayList<Long>();
106         IntByReference iter = new IntByReference();
107         IOKitUtil.getMatchingServices("IOUSBController", iter);
108         int device = IOKit.INSTANCE.IOIteratorNext(iter.getValue());
109         while (device != 0) {
110             // Unique global identifier for this device
111             LongByReference id = new LongByReference();
112             IOKit.INSTANCE.IORegistryEntryGetRegistryEntryID(device, id);
113             usbControllers.add(id.getValue());
114 
115             // Get device name and store in map
116             IOKit.INSTANCE.IORegistryEntryGetName(device, buffer);
117             nameMap.put(id.getValue(), buffer.getString(0));
118             // The only information we have in registry for this device is the
119             // locationID. Use that to search for matching PCI device to obtain
120             // more information.
121             CFTypeRef ref = IOKit.INSTANCE.IORegistryEntryCreateCFProperty(device, CfUtil.getCFString("locationID"),
122                     CfUtil.ALLOCATOR, 0);
123             if (ref != null && ref.getPointer() != null) {
124                 getControllerIdByLocation(id.getValue(), ref);
125             }
126             CfUtil.release(ref);
127 
128             // Now iterate the children of this device in the "IOService" plane.
129             // If device parent is root, link to the controller
130             IntByReference childIter = new IntByReference();
131             IOKit.INSTANCE.IORegistryEntryGetChildIterator(device, "IOService", childIter);
132             int childDevice = IOKit.INSTANCE.IOIteratorNext(childIter.getValue());
133             while (childDevice != 0) {
134                 // Unique global identifier for this device
135                 LongByReference childId = new LongByReference();
136                 IOKit.INSTANCE.IORegistryEntryGetRegistryEntryID(childDevice, childId);
137 
138                 // Get this device's parent in the "IOUSB" plane
139                 IntByReference parent = new IntByReference();
140                 IOKit.INSTANCE.IORegistryEntryGetParentEntry(childDevice, "IOUSB", parent);
141                 // If the parent is not an IOUSBDevice (will be root), set the
142                 // parentId to the controller
143                 LongByReference parentId = new LongByReference();
144                 if (!IOKit.INSTANCE.IOObjectConformsTo(parent.getValue(), "IOUSBDevice")) {
145                     parentId = id;
146                 } else {
147                     // Unique global identifier for the parent
148                     IOKit.INSTANCE.IORegistryEntryGetRegistryEntryID(parent.getValue(), parentId);
149                 }
150                 // Store parent in map
151                 MapUtil.createNewListIfAbsent(hubMap, parentId.getValue()).add(childId.getValue());
152 
153                 // Get device name and store in map
154                 IOKit.INSTANCE.IORegistryEntryGetName(childDevice, buffer);
155                 nameMap.put(childId.getValue(), buffer.getString(0));
156                 // Get vendor and store in map
157                 String vendor = IOKitUtil.getIORegistryStringProperty(childDevice, "USB Vendor Name");
158                 if (vendor != null) {
159                     vendorMap.put(childId.getValue(), vendor);
160                 }
161                 // Get vendorId and store in map
162                 long vendorId = IOKitUtil.getIORegistryLongProperty(childDevice, "idVendor");
163                 if (vendorId != 0) {
164                     vendorIdMap.put(childId.getValue(), String.format("%04x", 0xffff & vendorId));
165                 }
166                 // Get productId and store in map
167                 long productId = IOKitUtil.getIORegistryLongProperty(childDevice, "idProduct");
168                 if (productId != 0) {
169                     productIdMap.put(childId.getValue(), String.format("%04x", 0xffff & productId));
170                 }
171                 // Get serial and store in map
172                 String serial = IOKitUtil.getIORegistryStringProperty(childDevice, "USB Serial Number");
173                 if (serial != null) {
174                     serialMap.put(childId.getValue(), serial);
175                 }
176                 IOKit.INSTANCE.IOObjectRelease(childDevice);
177                 childDevice = IOKit.INSTANCE.IOIteratorNext(childIter.getValue());
178             }
179             IOKit.INSTANCE.IOObjectRelease(childIter.getValue());
180 
181             IOKit.INSTANCE.IOObjectRelease(device);
182             device = IOKit.INSTANCE.IOIteratorNext(iter.getValue());
183         }
184         IOKit.INSTANCE.IOObjectRelease(iter.getValue());
185 
186         // Build tree and return
187         List<UsbDevice> controllerDevices = new ArrayList<UsbDevice>();
188         for (Long controller : usbControllers) {
189             controllerDevices.add(getDeviceAndChildren(controller, "0000", "0000"));
190         }
191         return controllerDevices.toArray(new UsbDevice[controllerDevices.size()]);
192     }

194     /**
195      * Looks up vendor and product id information for a USB Host Controller by
196      * cross-referencing the location
197      *
198      * @param id
199      *            The global unique ID for the host controller used as a key for
200      *            maps
201      * @param locationId
202      *            The locationID of this controller returned from the registry
203      */
204     private static void getControllerIdByLocation(long id, CFTypeRef locationId) {
205         // Create a matching property dictionary from the locationId
206         CFMutableDictionaryRef propertyDict = CoreFoundation.INSTANCE.CFDictionaryCreateMutable(CfUtil.ALLOCATOR, 0,
207                 null, null);
208         CoreFoundation.INSTANCE.CFDictionarySetValue(propertyDict, CfUtil.getCFString("locationID"), locationId);
209         CFMutableDictionaryRef matchingDict = CoreFoundation.INSTANCE.CFDictionaryCreateMutable(CfUtil.ALLOCATOR, 0,
210                 null, null);
211         CoreFoundation.INSTANCE.CFDictionarySetValue(matchingDict, CfUtil.getCFString("IOPropertyMatch"), propertyDict);
212 
213         // search for all IOservices that match the locationID
214         IntByReference serviceIterator = new IntByReference();
215         IOKitUtil.getMatchingServices(matchingDict, serviceIterator);
216         // getMatchingServices releases matchingDict
217         CfUtil.release(propertyDict);
218 
219         // Iterate matching services looking for devices whose parents have the
220         // vendor and device ids
221         boolean found = false;
222         int matchingService = IOKit.INSTANCE.IOIteratorNext(serviceIterator.getValue());
223         while (matchingService != 0 && !found) {
224             // Get the parent, which contains the keys we need
225             IntByReference parent = new IntByReference();
226             IOKit.INSTANCE.IORegistryEntryGetParentEntry(matchingService, "IOService", parent);
227             // look up the vendor-id by key
228             // vendor-id is a byte array of 4 bytes
229             byte[] vid = IOKitUtil.getIORegistryByteArrayProperty(parent.getValue(), "vendor-id");
230             if (vid != null && vid.length >= 2) {
231                 vendorIdMap.put(id, String.format("%02x%02x", vid[1], vid[0]));
232                 found = true;
233             }
234             // look up the device-id by key
235             // device-id is a byte array of 4 bytes
236             byte[] pid = IOKitUtil.getIORegistryByteArrayProperty(parent.getValue(), "device-id");
237             if (pid != null && pid.length >= 2) {
238                 productIdMap.put(id, String.format("%02x%02x", pid[1], pid[0]));
239                 found = true;
240             }
241             // iterate
242             matchingService = IOKit.INSTANCE.IOIteratorNext(matchingService);
243         }
244     }

246     /**
247      * Recursively creates MacUsbDevices by fetching information from maps to
248      * populate fields
249      *
250      * @param registryEntryId
251      *            The device unique registry id.
252      * @param vid
253      *            The default (parent) vendor ID
254      * @param pid
255      *            The default (parent) product ID
256      * @return A MacUsbDevice corresponding to this device
257      */
258     private static MacUsbDevice getDeviceAndChildren(Long registryEntryId, String vid, String pid) {
259         String vendorId = MapUtil.getOrDefault(vendorIdMap, registryEntryId, vid);
260         String productId = MapUtil.getOrDefault(productIdMap, registryEntryId, pid);
261         List<Long> childIds = MapUtil.getOrDefault(hubMap, registryEntryId, new ArrayList<Long>());
262         List<MacUsbDevice> usbDevices = new ArrayList<MacUsbDevice>();
263         for (Long id : childIds) {
264             usbDevices.add(getDeviceAndChildren(id, vendorId, productId));
265         }
266         Collections.sort(usbDevices);
267         return new MacUsbDevice(MapUtil.getOrDefault(nameMap, registryEntryId, vendorId + ":" + productId),
268                 MapUtil.getOrDefault(vendorMap, registryEntryId, ""), vendorId, productId,
269                 MapUtil.getOrDefault(serialMap, registryEntryId, ""),
270                 usbDevices.toArray(new UsbDevice[usbDevices.size()]));
271     }
