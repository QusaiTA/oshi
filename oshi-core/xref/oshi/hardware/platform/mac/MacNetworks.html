24  package oshi.hardware.platform.mac;
25  
26  import java.util.HashMap;
27  import java.util.Map;
28  
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;
31  
32  import com.sun.jna.Memory; // NOSONAR
33  import com.sun.jna.Pointer;
34  import com.sun.jna.platform.mac.SystemB;
35  import com.sun.jna.platform.mac.SystemB.IFmsgHdr;
36  import com.sun.jna.platform.mac.SystemB.IFmsgHdr2;
37  import com.sun.jna.ptr.IntByReference;
38  
39  import oshi.hardware.NetworkIF;
40  import oshi.hardware.common.AbstractNetworks;
41  
42  /**
43   * @author widdis[at]gmail[dot]com
44   */
45  public class MacNetworks extends AbstractNetworks {
46  
47      private static final long serialVersionUID = 1L;
48  
49      private static final Logger LOG = LoggerFactory.getLogger(MacNetworks.class);
50  
51      private static final int CTL_NET = 4;
52      private static final int PF_ROUTE = 17;
53      private static final int NET_RT_IFLIST2 = 6;
54      private static final int RTM_IFINFO2 = 0x12;
55  
56      /**
57       * Update map no more frequently than 200ms.
58       *
59       * Key is the index of the IF.
60       */
61      private static Map<Integer, IFdata> ifMap = new HashMap<Integer, IFdata>();
62      private static long lastIFmapTime = 0L;
63  
64      /**
65       * Class to store a subset of IF data in the ifMap
66       */
67      private static class IFdata {
68          private long oPackets;
69          private long iPackets;
70          private long oBytes;
71          private long iBytes;
72          private long oErrors;
73          private long iErrors;
74          private long speed;
75  
76          IFdata(long oPackets, long iPackets, long oBytes, long iBytes, long oErrors, long iErrors, long speed) {
77              this.oPackets = oPackets;
78              this.iPackets = iPackets;
79              this.oBytes = oBytes;
80              this.iBytes = iBytes;
81              this.oErrors = oErrors;
82              this.iErrors = iErrors;
83              this.speed = speed;
84          }
85      }
86  
87      /**
88       * Map all network interfaces. Ported from source code of "netstat -ir". See
89       * http://opensource.apple.com/source/network_cmds/network_cmds-457/
90       * netstat.tproj/if.c
91       *
92       * @return timestamp when the data was read
93       */
94      private static synchronized long mapIFs() {
95          if (System.currentTimeMillis() - lastIFmapTime < 200L) {
96              // Polled too recently; do nothing
97              return lastIFmapTime;
98          }
99  
100         // Get buffer of all interface information
101         int[] mib = { CTL_NET, PF_ROUTE, 0, 0, NET_RT_IFLIST2, 0 };
102         IntByReference len = new IntByReference();
103         if (0 != SystemB.INSTANCE.sysctl(mib, 6, null, len, null, 0)) {
104             LOG.error("Didn't get buffer length for IFLIST2");
105             return lastIFmapTime;
106         }
107         Pointer buf = new Memory(len.getValue());
108         if (0 != SystemB.INSTANCE.sysctl(mib, 6, buf, len, null, 0)) {
109             LOG.error("Didn't get buffer for IFLIST2");
110             return lastIFmapTime;
111         }
112         lastIFmapTime = System.currentTimeMillis();
113 
114         // CLear out old hashmap
115         ifMap.clear();
116 
117         // Iterate offset from buf's pointer up to limit of buf
118         int lim = len.getValue();
119         int next = 0;
120         while (next < lim) {
121             // Get pointer to current native part of buf
122             Pointer p = new Pointer(Pointer.nativeValue(buf) + next);
123             // Cast pointer to if_msghdr
124             IFmsgHdr ifm = new IFmsgHdr(p);
125             ifm.read();
126             // Advance next
127             next += ifm.ifm_msglen;
128             // Skip messages which are not the right format
129             if (ifm.ifm_type != RTM_IFINFO2) {
130                 continue;
131             }
132             // Cast pointer to if_msghdr2
133             IFmsgHdr2 if2m = new IFmsgHdr2(p);
134             if2m.read();
135 
136             // Grab data to place in map using index as the key
137             ifMap.put(Integer.valueOf(if2m.ifm_index),
138                     new IFdata(if2m.ifm_data.ifi_opackets, if2m.ifm_data.ifi_ipackets, if2m.ifm_data.ifi_obytes,
139                             if2m.ifm_data.ifi_ibytes, if2m.ifm_data.ifi_oerrors, if2m.ifm_data.ifi_ierrors,
140                             if2m.ifm_data.ifi_baudrate));
141         }
142         return lastIFmapTime;
143     }
144 
145     /**
146      * Updates interface network statistics on the given interface. Statistics
147      * include packets and bytes sent and received, and interface speed.
148      *
149      * @param netIF
150      *            The interface on which to update statistics
151      */
152     public static synchronized void updateNetworkStats(NetworkIF netIF) {
153         // Update data
154         long timeStamp = mapIFs();
155 
156         Integer index = Integer.valueOf(netIF.getNetworkInterface().getIndex());
157         IFdata ifData = ifMap.get(index);
158         if (ifData != null) {
159             // Retrieve values from map and store
160             netIF.setBytesSent(ifData.oBytes);
161             netIF.setBytesRecv(ifData.iBytes);
162             netIF.setPacketsSent(ifData.oPackets);
163             netIF.setPacketsRecv(ifData.iPackets);
164             netIF.setOutErrors(ifData.oErrors);
165             netIF.setInErrors(ifData.iErrors);
166             netIF.setSpeed(ifData.speed);
167             netIF.setTimeStamp(timeStamp);
168         }
169     }
