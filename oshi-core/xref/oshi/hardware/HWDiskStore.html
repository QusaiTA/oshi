101     /**
102      * Make a best effort to update all the statistics about the drive without
103      * needing to recreate the drive list. This method provides for more frequent
104      * periodic updates of drive statistics. It will not detect if a removable drive
105      * has been removed and replaced by a different drive in between method calls.
106      *
107      * @return True if the update was (probably) successful, false if the disk was
108      *         not found
109      */
110     public boolean updateDiskStats() {
111         boolean diskFound = false;
112         switch (SystemInfo.getCurrentPlatformEnum()) {
113         case WINDOWS:
114             diskFound = WindowsDisks.updateDiskStats(this);
115             break;
116         case LINUX:
117             diskFound = LinuxDisks.updateDiskStats(this);
118             break;
119         case MACOSX:
120             diskFound = MacDisks.updateDiskStats(this);
121             break;
122         case SOLARIS:
123             diskFound = SolarisDisks.updateDiskStats(this);
124             break;
125         case FREEBSD:
126             diskFound = FreeBsdDisks.updateDiskStats(this);
127             break;
128         default:
129             LOG.error("Unsupported platform. No update performed.");
130             break;
131         }
132         return diskFound;
133     }
