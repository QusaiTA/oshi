43   * Provides access to kstat information on Solaris

45   * @author widdis[at]gmail[dot]com

47  public class KstatUtil {
48      private static final Logger LOG = LoggerFactory.getLogger(KstatUtil.class);
49  
50      // All active use of this variable must be synchronized
51      private static final KstatCtl kc = LibKstat.INSTANCE.kstat_open();
52  
53      static {
54          Runtime.getRuntime().addShutdownHook(new Thread() {
55              @Override
56              public void run() {
57                  LibKstat.INSTANCE.kstat_close(kc);
58              }
59          });
60      }
61  
62      private KstatUtil() {
63      }

65      /**
66       * Convenience method for kstat_data_lookup() with String return values.
67       * Searches the kstat's data section for the record with the specified name.
68       * This operation is valid only for kstat types which have named data records.
69       * Currently, only the KSTAT_TYPE_NAMED and KSTAT_TYPE_TIMER kstats have named
70       * data records.
71       *
72       * @param ksp
73       *            The kstat to search
74       * @param name
75       *            The key for the name-value pair, or name of the timer as
76       *            applicable
77       * @return The value as a String.
78       */
79      public static synchronized String kstatDataLookupString(Kstat ksp, String name) {
80          if (ksp.ks_type != LibKstat.KSTAT_TYPE_NAMED && ksp.ks_type != LibKstat.KSTAT_TYPE_TIMER) {
81              throw new IllegalArgumentException("Not a kstat_named or kstat_timer kstat.");
82          }
83          Pointer p = LibKstat.INSTANCE.kstat_data_lookup(ksp, name);
84          if (p == null) {
85              LOG.error("Failed lo lookup kstat value for key {}", name);
86              return "";
87          }
88          KstatNamed data = new KstatNamed(p);
89          switch (data.data_type) {
90          case LibKstat.KSTAT_DATA_CHAR:
91              return new String(data.value.charc).trim();
92          case LibKstat.KSTAT_DATA_INT32:
93              return Integer.toString(data.value.i32);
94          case LibKstat.KSTAT_DATA_UINT32:
95              return FormatUtil.toUnsignedString(data.value.ui32);
96          case LibKstat.KSTAT_DATA_INT64:
97              return Long.toString(data.value.i64);
98          case LibKstat.KSTAT_DATA_UINT64:
99              return FormatUtil.toUnsignedString(data.value.ui64);
100         case LibKstat.KSTAT_DATA_STRING:
101             return data.value.str.addr.getString(0);
102         default:
103             LOG.error("Unimplemented kstat data type {}", data.data_type);
104             return "";
105         }
106     }

108     /**
109      * Convenience method for kstat_data_lookup() with numeric return values.
110      * Searches the kstat's data section for the record with the specified name.
111      * This operation is valid only for kstat types which have named data records.
112      * Currently, only the KSTAT_TYPE_NAMED and KSTAT_TYPE_TIMER kstats have named
113      * data records.
114      *
115      * @param ksp
116      *            The kstat to search
117      * @param name
118      *            The key for the name-value pair, or name of the timer as
119      *            applicable
120      * @return The value as a long. If the data type is a character or string type,
121      *         returns 0 and logs an error.
122      */
123     public static synchronized long kstatDataLookupLong(Kstat ksp, String name) {
124         if (ksp.ks_type != LibKstat.KSTAT_TYPE_NAMED && ksp.ks_type != LibKstat.KSTAT_TYPE_TIMER) {
125             throw new IllegalArgumentException("Not a kstat_named or kstat_timer kstat.");
126         }
127         Pointer p = LibKstat.INSTANCE.kstat_data_lookup(ksp, name);
128         if (p == null) {
129             if (LOG.isErrorEnabled()) {
130                 LOG.error("Failed lo lookup kstat value on {}:{}:{} for key {}", new String(ksp.ks_module).trim(),
131                         ksp.ks_instance, new String(ksp.ks_name).trim(), name);
132             }
133             return 0L;
134         }
135         KstatNamed data = new KstatNamed(p);
136         switch (data.data_type) {
137         case LibKstat.KSTAT_DATA_INT32:
138             return data.value.i32;
139         case LibKstat.KSTAT_DATA_UINT32:
140             return FormatUtil.getUnsignedInt(data.value.ui32);
141         case LibKstat.KSTAT_DATA_INT64:
142             return data.value.i64;
143         case LibKstat.KSTAT_DATA_UINT64:
144             return data.value.ui64;
145         default:
146             LOG.error("Unimplemented or non-numeric kstat data type {}", data.data_type);
147             return 0L;
148         }
149     }

151     /**
152      * Convenience method for kstat_read() which gets data from the kernel for the
153      * kstat pointed to by ksp. ksp.ks_data is automatically allocated (or
154      * reallocated) to be large enough to hold all of the data. ksp.ks_ndata is set
155      * to the number of data fields, ksp.ks_data_size is set to the total size of
156      * the data, and ksp.ks_snaptime is set to the high-resolution time at which the
157      * data snapshot was taken.
158      *
159      * @param ksp
160      *            The kstat from which to retrieve data
161      * @return True if successful; false otherwise
162      */
163     public static synchronized boolean kstatRead(Kstat ksp) {
164         int retry = 0;
165         while (0 > LibKstat.INSTANCE.kstat_read(kc, ksp, null)) {
166             if (LibKstat.EAGAIN != Native.getLastError() || 5 <= ++retry) {
167                 if (LOG.isErrorEnabled()) {
168                     LOG.error("Failed to read kstat {}:{}:{}", new String(ksp.ks_module).trim(), ksp.ks_instance,
169                             new String(ksp.ks_name).trim());
170                 }
171                 return false;
172             }
173             Util.sleep(8 << retry);
174         }
175         return true;
176     }

178     /**
179      * Convenience method for kstat_lookup(). Traverses the kstat chain, searching
180      * for a kstat with the same ks_module, ks_instance, and ks_name fields; this
181      * triplet uniquely identifies a kstat. If ks_module is NULL, ks_instance is -1,
182      * or ks_name is NULL, then those fields will be ignored in the search.
183      *
184      * @param module
185      *            The module, or null to ignore
186      * @param instance
187      *            The instance, or -1 to ignore
188      * @param name
189      *            The name, or null to ignore
190      * @return The first match of the requested Kstat structure if found, or null
191      */
192     public static synchronized Kstat kstatLookup(String module, int instance, String name) {
193         int ret = LibKstat.INSTANCE.kstat_chain_update(kc);
194         if (ret < 0) {
195             LOG.error("Failed to update kstat chain");
196             return null;
197         }
198         return LibKstat.INSTANCE.kstat_lookup(kc, module, instance, name);
199     }
200 
201     /**
202      * Convenience method for kstat_lookup(). Traverses the kstat chain, searching
203      * for all kstats with the same ks_module, ks_instance, and ks_name fields; this
204      * triplet uniquely identifies a kstat. If ks_module is NULL, ks_instance is -1,
205      * or ks_name is NULL, then those fields will be ignored in the search.
206      *
207      * @param module
208      *            The module, or null to ignore
209      * @param instance
210      *            The instance, or -1 to ignore
211      * @param name
212      *            The name, or null to ignore
213      * @return All matches of the requested Kstat structure if found, or an empty
214      *         list otherwise
215      */
216     public static synchronized List<Kstat> kstatLookupAll(String module, int instance, String name) {
217         List<Kstat> kstats = new ArrayList<Kstat>();
218         int ret = LibKstat.INSTANCE.kstat_chain_update(kc);
219         if (ret < 0) {
220             LOG.error("Failed to update kstat chain");
221             return kstats;
222         }
223         for (Kstat ksp = LibKstat.INSTANCE.kstat_lookup(kc, module, instance, name); ksp != null; ksp = ksp.next()) {
224             if ((module == null || module.equals(new String(ksp.ks_module).trim()))
225                     && (instance < 0 || instance == ksp.ks_instance)
226                     && (name == null || name.equals(new String(ksp.ks_name).trim()))) {
227                 kstats.add(ksp);
228             }
229         }
230         return kstats;
231     }
