36  public abstract class AbstractOperatingSystem implements OperatingSystem {
37  
38      private static final long serialVersionUID = 1L;
39  
40      protected String manufacturer;
41      protected String family;
42      protected OperatingSystemVersion version;
43      // Initialize based on JVM Bitness. Individual OS implementations will test
44      // if 32-bit JVM running on 64-bit OS
45      protected int bitness = System.getProperty("os.arch").indexOf("64") != -1 ? 64 : 32;

47      /*
48       * Comparators for use in processSort()
49       */
50      private static final Comparator<OSProcess> CPU_DESC_SORT = new Comparator<OSProcess>() {
51          @Override
52          public int compare(OSProcessf="../../../oshi/software/os/OSProcess.html#OSProcess">OSProcess p1, OSProcess p2) {
53              return Double.compare(p2.calculateCpuPercent(), p1.calculateCpuPercent());
54          }
55      };
56      private static final Comparator<OSProcess> RSS_DESC_SORT = new Comparator<OSProcess>() {
57          @Override
58          public int compare(OSProcessf="../../../oshi/software/os/OSProcess.html#OSProcess">OSProcess p1, OSProcess p2) {
59              return longCompare(p2.getResidentSetSize(), p1.getResidentSetSize());
60          }
61      };
62      private static final Comparator<OSProcess> UPTIME_DESC_SORT = new Comparator<OSProcess>() {
63          @Override
64          public int compare(OSProcessf="../../../oshi/software/os/OSProcess.html#OSProcess">OSProcess p1, OSProcess p2) {
65              return longCompare(p2.getUpTime(), p1.getUpTime());
66          }
67      };
68      private static final Comparator<OSProcess> UPTIME_ASC_SORT = new Comparator<OSProcess>() {
69          @Override
70          public int compare(OSProcessf="../../../oshi/software/os/OSProcess.html#OSProcess">OSProcess p1, OSProcess p2) {
71              return longCompare(p1.getUpTime(), p2.getUpTime());
72          }
73      };
74      private static final Comparator<OSProcess> PID_ASC_SORT = new Comparator<OSProcess>() {
75          @Override
76          public int compare(OSProcessf="../../../oshi/software/os/OSProcess.html#OSProcess">OSProcess p1, OSProcess p2) {
77              return intCompare(p1.getProcessID(), p2.getProcessID());
78          }
79      };
80      private static final Comparator<OSProcess> PARENTPID_ASC_SORT = new Comparator<OSProcess>() {
81          @Override
82          public int compare(OSProcessf="../../../oshi/software/os/OSProcess.html#OSProcess">OSProcess p1, OSProcess p2) {
83              return intCompare(p1.getParentProcessID(), p2.getParentProcessID());
84          }
85      };
86      private static final Comparator<OSProcess> NAME_ASC_SORT = new Comparator<OSProcess>() {
87          @Override
88          public int compare(OSProcessf="../../../oshi/software/os/OSProcess.html#OSProcess">OSProcess p1, OSProcess p2) {
89              return p1.getName().toLowerCase().compareTo(p2.getName().toLowerCase());
90          }
91      };
92  
93      private static int longCompare(long x, long y) {
94          if (x < y) {
95              return -1;
96          } else if (x > y) {
97              return 1;
98          }
99          return 0;
100     }
101 
102     private static int intCompare(int x, int y) {
103         if (x < y) {
104             return -1;
105         } else if (x > y) {
106             return 1;
107         }
108         return 0;
109     }

135     /**
136      * Sorts an array of processes using the specified sorting, returning an array
137      * with the top limit results if positive.
138      *
139      * @param processes
140      *            The array to sort
141      * @param limit
142      *            The number of results to return if positive; if zero returns all
143      *            results
144      * @param sort
145      *            The sorting to use, or null
146      * @return An array of size limit (if positive) or of all processes, sorted as
147      *         specified
148      */
149     protected List<OSProcess> processSort(List<OSProcess> processes, int limit, ProcessSort sort) {
150         if (sort != null) {
151             switch (sort) {
152             case CPU:
153                 Collections.sort(processes, CPU_DESC_SORT);
154                 break;
155             case MEMORY:
156                 Collections.sort(processes, RSS_DESC_SORT);
157                 break;
158             case OLDEST:
159                 Collections.sort(processes, UPTIME_DESC_SORT);
160                 break;
161             case NEWEST:
162                 Collections.sort(processes, UPTIME_ASC_SORT);
163                 break;
164             case PID:
165                 Collections.sort(processes, PID_ASC_SORT);
166                 break;
167             case PARENTPID:
168                 Collections.sort(processes, PARENTPID_ASC_SORT);
169                 break;
170             case NAME:
171                 Collections.sort(processes, NAME_ASC_SORT);
172                 break;
173             default:
174                 // Should never get here! If you get this exception you've
175                 // added something to the enum without adding it here. Tsk.
176                 throw new IllegalArgumentException("Unimplemented enum type: " + sort.toString());
177             }
178         }
179         // Return max of limit or process size
180         // Nonpositive limit means return all
181         int maxProcs = processes.size();
182         if (limit > 0 && maxProcs > limit) {
183             maxProcs = limit;
184         } else {
185             return processes;
186         }
187         List<OSProcess> procs = new ArrayList<OSProcess>();
188         for (int i = 0; i < maxProcs; i++) {
189             procs.add(processes.get(i));
190         }
191         return procs;
192     }

194     @Override
195     public String toString() {
196         StringBuilder sb = new StringBuilder();
197         sb.append(getManufacturer()).append(' ').append(getFamily()).append(' ').append(getVersion().toString());
198         return sb.toString();
199     }

201     /**
202      * {@inheritDoc}
203      */
204     @Override
205     public OSProcess[] getProcesses(int limit, ProcessSort sort) {
206         return getProcesses(limit, sort, false);
207     }
208 
209     /**
210      * {@inheritDoc}
211      */
212     @Override
213     public List<OSProcess> getProcesses(Collection<Integer> pids) {
214         List<OSProcess> returnValue = new ArrayList<OSProcess>(pids.size());
215         for (Integer pid : pids) {
216             OSProcess process = getProcess(pid);
217             if (process != null) {
218                 returnValue.add(process);
219             }
220         }
221         return returnValue;
222     }

227     @Override
228     public int getBitness() {
229         return this.bitness;
230     }
