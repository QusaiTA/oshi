71      /*
72       * For handle counts

74      enum HandleCountProperty implements PdhCounterWildcardProperty {
75          // First element defines WMI instance name field and PDH instance filter
76          NAME(PerfCounterQuery.TOTAL_INSTANCE),
77          // Remaining elements define counters
78          HANDLECOUNT("Handle Count");

80          private final String counter;

82          HandleCountProperty(String counter) {
83              this.counter = counter;
84          }

86          /**
87           * {@inheritDoc}
88           */
89          @Override
90          public String getCounter() {
91              return counter;
92          }

98      private static final long MAX_WINDOWS_HANDLES;
99      static {
100         // Determine whether 32-bit or 64-bit handle limit, although both are
101         // essentially infinite for practical purposes. See
102         // https://blogs.technet.microsoft.com/markrussinovich/2009/09/29/pushing-the-limits-of-windows-handles/
103         if (System.getenv("ProgramFiles(x86)") == null) {
104             MAX_WINDOWS_HANDLES = 16777216L - 32768L;
105         } else {
106             MAX_WINDOWS_HANDLES = 16777216L - 65536L;
107         }
108     }

115     /**
116      * Gets File System Information.
117      *
118      * @return An array of {@link OSFileStore} objects representing mounted volumes.
119      *         May return disconnected volumes with
120      *         {@link OSFileStore#getTotalSpace()} = 0.
121      */
122     @Override
123     public OSFileStore[] getFileStores() {
124         // Create list to hold results
125         ArrayList<OSFileStore> result;
126 
127         // Begin with all the local volumes
128         result = getLocalVolumes();
129 
130         // Build a map of existing mount point to OSFileStore
131         Map<String, OSFileStore> volumeMap = new HashMap<String, OSFileStore>();
132         for (OSFileStore volume : result) {
133             volumeMap.put(volume.getMount(), volume);
134         }
135 
136         // Iterate through volumes in WMI and update description (if it exists)
137         // or add new if it doesn't (expected for network drives)
138         for (OSFileStore wmiVolume : getWmiVolumes()) {
139             if (volumeMap.containsKey(wmiVolume.getMount())) {
140                 // If the volume is already in our list, update the name field
141                 // using WMI's more verbose name
142                 volumeMap.get(wmiVolume.getMount()).setName(wmiVolume.getName());
143             } else {
144                 // Otherwise add the new volume in its entirety
145                 result.add(wmiVolume);
146             }
147         }
148         return result.toArray(new OSFileStore[result.size()]);
149     }

151     /**
152      * Private method for getting all mounted local drives.
153      *
154      * @return A list of {@link OSFileStore} objects representing all local mounted
155      *         volumes
156      */
157     private ArrayList<OSFileStore> getLocalVolumes() {
158         ArrayList<OSFileStore> fs;
159         String volume;
160         String strFsType;
161         String strName;
162         String strMount;
163         WinNT.HANDLE hVol;
164         WinNT.LARGE_INTEGER userFreeBytes;
165         WinNT.LARGE_INTEGER totalBytes;
166         WinNT.LARGE_INTEGER systemFreeBytes;
167         boolean retVal;
168         char[] aVolume;
169         char[] fstype;
170         char[] name;
171         char[] mount;
172 
173         fs = new ArrayList<OSFileStore>();
174         aVolume = new char[BUFSIZE];
175 
176         hVol = Kernel32.INSTANCE.FindFirstVolume(aVolume, BUFSIZE);
177         if (WinBase.INVALID_HANDLE_VALUE.equals(hVol)) {
178             return fs;
179         }
180 
181         while (true) {
182             fstype = new char[16];
183             name = new char[BUFSIZE];
184             mount = new char[BUFSIZE];
185 
186             userFreeBytes = new WinNT.LARGE_INTEGER(0L);
187             totalBytes = new WinNT.LARGE_INTEGER(0L);
188             systemFreeBytes = new WinNT.LARGE_INTEGER(0L);
189 
190             volume = new String(aVolume).trim();
191             Kernel32.INSTANCE.GetVolumeInformation(volume, name, BUFSIZE, null, null, null, fstype, 16);
192             Kernel32.INSTANCE.GetVolumePathNamesForVolumeName(volume, mount, BUFSIZE, null);
193             Kernel32.INSTANCE.GetDiskFreeSpaceEx(volume, userFreeBytes, totalBytes, systemFreeBytes);
194 
195             strMount = new String(mount).trim();
196             strName = new String(name).trim();
197             strFsType = new String(fstype).trim();
198             // Parse uuid from volume name
199             String uuid = ParseUtil.parseUuidOrDefault(volume, "");
200 
201             if (!strMount.isEmpty()) {
202                 // Volume is mounted
203                 OSFileStoreOSFileStore">OSFileStore osStore = new OSFileStore();
204                 osStore.setName(String.format("%s (%s)", strName, strMount));
205                 osStore.setVolume(volume);
206                 osStore.setMount(strMount);
207                 osStore.setDescription(getDriveType(strMount));
208                 osStore.setType(strFsType);
209                 osStore.setUUID(uuid);
210                 osStore.setUsableSpace(systemFreeBytes.getValue());
211                 osStore.setTotalSpace(totalBytes.getValue());
212                 fs.add(osStore);
213             }
214             retVal = Kernel32.INSTANCE.FindNextVolume(hVol, aVolume, BUFSIZE);
215             if (!retVal) {
216                 Kernel32.INSTANCE.FindVolumeClose(hVol);
217                 break;
218             }
219         }
220 
221         return fs;
222     }

224     /**
225      * Private method for getting logical drives listed in WMI.
226      *
227      * @return A list of {@link OSFileStore} objects representing all network
228      *         mounted volumes
229      */
230     private List<OSFileStore> getWmiVolumes() {
231         long free;
232         long total;
233         List<OSFileStore> fs = new ArrayList<OSFileStore>();
234 
235         WmiResult<LogicalDiskProperty> drives = wmiQueryHandler.queryWMI(this.logicalDiskQuery);
236 
237         for (int i = 0; i < drives.getResultCount(); i++) {
238             free = WmiUtil.getUint64(drives, LogicalDiskProperty.FREESPACE, i);
239             total = WmiUtil.getUint64(drives, LogicalDiskProperty.SIZE, i);
240             String description = WmiUtil.getString(drives, LogicalDiskProperty.DESCRIPTION, i);
241             String name = WmiUtil.getString(drives, LogicalDiskProperty.NAME, i);
242             int type = WmiUtil.getUint32(drives, LogicalDiskProperty.DRIVETYPE, i);
243             String volume;
244             if (type != 4) {
245                 char[] chrVolume = new char[BUFSIZE];
246                 Kernel32.INSTANCE.GetVolumeNameForVolumeMountPoint(name + "\\", chrVolume, BUFSIZE);
247                 volume = new String(chrVolume).trim();
248             } else {
249                 volume = WmiUtil.getString(drives, LogicalDiskProperty.PROVIDERNAME, i);
250                 String[] split = volume.split("\\\\");
251                 if (split.length > 1 && split[split.length - 1].length() > 0) {
252                     description = split[split.length - 1];
253                 }
254             }
255 
256             OSFileStoreOSFileStore">OSFileStore osStore = new OSFileStore();
257             osStore.setName(String.format("%s (%s)", description, name));
258             osStore.setVolume(volume);
259             osStore.setMount(name + "\\");
260             osStore.setDescription(getDriveType(name));
261             osStore.setType(WmiUtil.getString(drives, LogicalDiskProperty.FILESYSTEM, i));
262             osStore.setUUID("");
263             osStore.setUsableSpace(free);
264             osStore.setTotalSpace(total);
265             fs.add(osStore);
266         }
267 
268         return fs;
269     }

271     /**
272      * Private method for getting mounted drive type.
273      *
274      * @param drive
275      *            Mounted drive
276      * @return A drive type description
277      */
278     private String getDriveType(String drive) {
279         switch (Kernel32.INSTANCE.GetDriveType(drive)) {
280         case 2:
281             return "Removable drive";
282         case 3:
283             return "Fixed drive";
284         case 4:
285             return "Network drive";
286         case 5:
287             return "CD-ROM";
288         case 6:
289             return "RAM drive";
290         default:
291             return "Unknown drive type";
292         }
293     }

295     @Override
296     public long getOpenFileDescriptors() {
297         Map<HandleCountProperty, List<Long>> valueListMap = this.handlePerfCounters.queryValuesWildcard();
298         List<Long> valueList = valueListMap.get(HandleCountProperty.HANDLECOUNT);
299         long descriptors = 0L;
300         if (valueList != null) {
301             for (int i = 0; i < valueList.size(); i++) {
302                 descriptors += valueList.get(i);
303             }
304         }
305         return descriptors;
306     }
307 
308     @Override
309     public long getMaxFileDescriptors() {
310         return MAX_WINDOWS_HANDLES;
