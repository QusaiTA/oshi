144     private void initRegistry() {
145         // Get the title indices
146         int priorityBaseIndex;
147         int elapsedTimeIndex;
148         int idProcessIndex;
149         int creatingProcessIdIndex;
150         int ioReadIndex;
151         int ioWriteIndex;
152         int workingSetPrivateIndex;
153         try {
154             this.processIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("Process");
155             priorityBaseIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("Priority Base");
156             elapsedTimeIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("Elapsed Time");
157             idProcessIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("ID Process");
158             creatingProcessIdIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("Creating Process ID");
159             ioReadIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("IO Read Bytes/sec");
160             ioWriteIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("IO Write Bytes/sec");
161             workingSetPrivateIndex = PdhUtil.PdhLookupPerfIndexByEnglishName("Working Set - Private");
162         } catch (Win32Exception e) {
163             LOG.error("Unable to locate English counter names in registry Perflib 009. Assuming English counters.");
164             DWORDByReference index = new DWORDByReference();
165             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Process", index);
166             this.processIndex = index.getValue().intValue();
167             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Priority Base", index);
168             priorityBaseIndex = index.getValue().intValue();
169             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Elapsed Time", index);
170             elapsedTimeIndex = index.getValue().intValue();
171             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "ID Process", index);
172             idProcessIndex = index.getValue().intValue();
173             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Creating Process ID", index);
174             creatingProcessIdIndex = index.getValue().intValue();
175             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "IO Read Bytes/sec", index);
176             ioReadIndex = index.getValue().intValue();
177             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "IO Write Bytes/sec", index);
178             ioWriteIndex = index.getValue().intValue();
179             Pdh.INSTANCE.PdhLookupPerfIndexByName(null, "Working Set - Private", index);
180             workingSetPrivateIndex = index.getValue().intValue();
181         }
182         this.processIndexStr = Integer.toString(this.processIndex);
183 
184         // now load the Process registry to match up the offsets
185         // Sequentially increase the buffer until everything fits.
186         // Save this buffer size for later use
187         IntByReference lpcbData = new IntByReference(this.perfDataBufferSize);
188         Pointer pPerfData = new Memory(this.perfDataBufferSize);
189         int ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
190                 pPerfData, lpcbData);
191         if (ret != WinError.ERROR_SUCCESS && ret != WinError.ERROR_MORE_DATA) {
192             LOG.error("Error {} reading HKEY_PERFORMANCE_DATA from the registry.", ret);
193             return;
194         }
195         while (ret == WinError.ERROR_MORE_DATA) {
196             this.perfDataBufferSize += 4096;
197             lpcbData.setValue(this.perfDataBufferSize);
198             pPerfData = new Memory(this.perfDataBufferSize);
199             ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
200                     pPerfData, lpcbData);
201         }
202 
203         PERF_DATA_BLOCK perfData = new PERF_DATA_BLOCK(pPerfData.share(0));
204 
205         // See format at
206         // https://msdn.microsoft.com/en-us/library/windows/desktop/aa373105(v=vs.85).aspx
207         // [ ] Object Type
208         // [ ][ ][ ] Multiple counter definitions
209         // Then multiple:
210         // [ ] Instance Definition
211         // [ ] Instance name
212         // [ ] Counter Block
213         // [ ][ ][ ] Counter data for each definition above
214 
215         long perfObjectOffset = perfData.HeaderLength;
216 
217         // Iterate object types. For Process should only be one here
218         for (int obj = 0; obj < perfData.NumObjectTypes; obj++) {
219             PERF_OBJECT_TYPE perfObject = new PERF_OBJECT_TYPE(pPerfData.share(perfObjectOffset));
220             // Identify where counter definitions start
221             long perfCounterOffset = perfObjectOffset + perfObject.HeaderLength;
222             // If this isn't the Process object, ignore
223             if (perfObject.ObjectNameTitleIndex == this.processIndex) {
224                 for (int counter = 0; counter < perfObject.NumCounters; counter++) {
225                     PERF_COUNTER_DEFINITION perfCounter = new PERF_COUNTER_DEFINITION(
226                             pPerfData.share(perfCounterOffset));
227                     if (perfCounter.CounterNameTitleIndex == priorityBaseIndex) {
228                         this.priorityBaseOffset = perfCounter.CounterOffset;
229                     } else if (perfCounter.CounterNameTitleIndex == elapsedTimeIndex) {
230                         this.elapsedTimeOffset = perfCounter.CounterOffset;
231                     } else if (perfCounter.CounterNameTitleIndex == creatingProcessIdIndex) {
232                         this.creatingProcessIdOffset = perfCounter.CounterOffset;
233                     } else if (perfCounter.CounterNameTitleIndex == idProcessIndex) {
234                         this.idProcessOffset = perfCounter.CounterOffset;
235                     } else if (perfCounter.CounterNameTitleIndex == ioReadIndex) {
236                         this.ioReadOffset = perfCounter.CounterOffset;
237                     } else if (perfCounter.CounterNameTitleIndex == ioWriteIndex) {
238                         this.ioWriteOffset = perfCounter.CounterOffset;
239                     } else if (perfCounter.CounterNameTitleIndex == workingSetPrivateIndex) {
240                         this.workingSetPrivateOffset = perfCounter.CounterOffset;
241                     }
242                     // Increment for next Counter
243                     perfCounterOffset += perfCounter.ByteLength;
244                 }
245                 // We're done, break the loop
246                 break;
247             }
248             // Increment for next object (should never need this)
249             perfObjectOffset += perfObject.TotalByteLength;
250         }
251     }

253     private void initBitness() {
254         // If bitness is 64 we are 64 bit.
255         // If 32 test if we are on 64-bit OS
256         if (this.bitness < 64) {
257             // Try the easy way
258             if (System.getenv("ProgramFiles(x86)") != null) {
259                 this.bitness = 64;
260             } else {
261                 WmiQuery<BitnessProperty> bitnessQuery = new WmiQuery<BitnessProperty>("Win32_Processor",
262                         BitnessProperty.class);
263                 WmiResult<BitnessProperty> bitnessMap = wmiQueryHandler.queryWMI(bitnessQuery);
264                 if (bitnessMap.getResultCount() > 0) {
265                     this.bitness = WmiUtil.getUint16(bitnessMap, BitnessProperty.ADDRESSWIDTH, 0);
266                 }
267             }
268         }

292     @Override
293     public OSProcess[] getChildProcesses(int parentPid, int limit, ProcessSort sort) {
294         Set<Integer> childPids = new HashSet<Integer>();
295         // Get processes from ToolHelp API for parent PID
296         Tlhelp32.PROCESSENTRY32.ByReference processEntry = new Tlhelp32.PROCESSENTRY32.ByReference();
297         WinNT.HANDLE snapshot = Kernel32.INSTANCE.CreateToolhelp32Snapshot(Tlhelp32.TH32CS_SNAPPROCESS, new DWORD(0));
298         try {
299             while (Kernel32.INSTANCE.Process32Next(snapshot, processEntry)) {
300                 if (processEntry.th32ParentProcessID.intValue() == parentPid) {
301                     childPids.add(processEntry.th32ProcessID.intValue());
302                 }
303             }
304         } finally {
305             Kernel32.INSTANCE.CloseHandle(snapshot);
306         }
307         List<OSProcess> procList = getProcesses(childPids);
308         List<OSProcess> sorted = processSort(procList, limit, sort);
309         return sorted.toArray(new OSProcess[sorted.size()]);
310     }

335     /**
336      * Private method to do the heavy lifting for all the getProcess functions.
337      *
338      * @param pids
339      *            A collection of pids to query. If null, the entire process list
340      *            will be queried.
341      * @param slowFields
342      *            Whether to include fields that incur processor latency
343      * @return A corresponding list of processes
344      */
345     private List<OSProcess> processMapToList(Collection<Integer> pids, boolean slowFields) {
346         // Get data from the registry to update cache
347         updateProcessMapFromRegistry(pids);
348 
349         // define here to avoid object repeated creation overhead later
350         List<String> groupList = new ArrayList<String>();
351         List<String> groupIDList = new ArrayList<String>();
352         int myPid = getProcessId();
353 
354         // Structure we'll fill from native memory pointer for Vista+
355         Pointer pProcessInfo = null;
356         WTS_PROCESS_INFO_EX[] processInfo = null;
357         IntByReference pCount = new IntByReference(0);
358 
359         // WMI result we'll use for pre-Vista
360         WmiResult<ProcessXPProperty> processWmiResult = null;
361 
362         // Get processes from WTS (post-XP)
363         if (IS_WINDOWS7_OR_GREATER) {
364             final PointerByReference ppProcessInfo = new PointerByReference();
365             if (!Wtsapi32.INSTANCE.WTSEnumerateProcessesEx(Wtsapi32.WTS_CURRENT_SERVER_HANDLE,
366                     new IntByReference(Wtsapi32.WTS_PROCESS_INFO_LEVEL_1), Wtsapi32.WTS_ANY_SESSION, ppProcessInfo,
367                     pCount)) {
368                 LOG.error("Failed to enumerate Processes. Error code: {}", Kernel32.INSTANCE.GetLastError());
369                 return new ArrayList<OSProcess>(0);
370             }
371             // extract the pointed-to pointer and create array
372             pProcessInfo = ppProcessInfo.getValue();
373             final WTS_PROCESS_INFO_EX processInfoRef = new WTS_PROCESS_INFO_EX(pProcessInfo);
374             processInfo = (WTS_PROCESS_INFO_EX[]) processInfoRef.toArray(pCount.getValue());
375         } else {
376             // Pre-Vista we can't use WTSEnumerateProcessesEx so we'll grab the
377             // same info from WMI and fake the array
378             StringBuilder sb = new StringBuilder(PROCESS_BASE_CLASS);
379             if (pids != null) {
380                 boolean first = true;
381                 for (Integer pid : pids) {
382                     if (first) {
383                         sb.append(" WHERE ProcessID=");
384                         first = false;
385                     } else {
386                         sb.append(" OR ProcessID=");
387                     }
388                     sb.append(pid);
389                 }
390             }
391             PROCESS_QUERY_XP.setWmiClassName(sb.toString());
392             processWmiResult = wmiQueryHandler.queryWMI(PROCESS_QUERY_XP);
393         }
394 
395         // Store a subset of processes in a list to later return.
396         List<OSProcess> processList = new ArrayList<OSProcess>();
397 
398         int procCount = IS_WINDOWS7_OR_GREATER ? processInfo.length : processWmiResult.getResultCount();
399         for (int i = 0; i < procCount; i++) {
400 
401             // Skip if only updating a subset of pids, or if not in cache.
402             // (Cache should have just been updated from registry so this will
403             // only occur in a race condition for a just-started process.)
404             // However, when the cache is empty, there was a problem with
405             // filling
406             // the cache using performance information. When this happens, we
407             // ignore
408             // the cache completely.
409 
410             int pid = IS_WINDOWS7_OR_GREATER ? processInfo[i].ProcessId
411                     : WmiUtil.getUint32(processWmiResult, ProcessXPProperty.PROCESSID, i);
412             OSProcess proc = null;
413             if (this.processMap.isEmpty()) {
414                 if (pids != null && !pids.contains(pid)) {
415                     continue;
416                 }
417                 proc = new OSProcess();
418                 proc.setProcessID(pid);
419                 proc.setName(IS_WINDOWS7_OR_GREATER ? processInfo[i].pProcessName
420                         : WmiUtil.getString(processWmiResult, ProcessXPProperty.NAME, i));
421             } else {
422                 proc = this.processMap.get(pid);
423                 if (proc == null || pids != null && !pids.contains(pid)) {
424                     continue;
425                 }
426             }
427             // For my own process, set CWD
428             if (pid == myPid) {
429                 String cwd = new File(".").getAbsolutePath();
430                 // trim off trailing "."
431                 proc.setCurrentWorkingDirectory(cwd.isEmpty() ? "" : cwd.substring(0, cwd.length() - 1));
432             }
433 
434             if (IS_WINDOWS7_OR_GREATER) {
435                 WTS_PROCESS_INFO_EX procInfo = processInfo[i];
436                 proc.setKernelTime(procInfo.KernelTime.getValue() / 10000L);
437                 proc.setUserTime(procInfo.UserTime.getValue() / 10000L);
438                 proc.setThreadCount(procInfo.NumberOfThreads);
439                 proc.setVirtualSize(procInfo.PagefileUsage & 0xffffffffL);
440                 proc.setOpenFiles(procInfo.HandleCount);
441             } else {
442                 proc.setKernelTime(WmiUtil.getUint64(processWmiResult, ProcessXPProperty.KERNELMODETIME, i) / 10000L);
443                 proc.setUserTime(WmiUtil.getUint64(processWmiResult, ProcessXPProperty.USERMODETIME, i) / 10000L);
444                 proc.setThreadCount(WmiUtil.getUint32(processWmiResult, ProcessXPProperty.THREADCOUNT, i));
445                 // WMI Pagefile usage is in KB
446                 proc.setVirtualSize(
447                         1024 * (WmiUtil.getUint32(processWmiResult, ProcessXPProperty.PAGEFILEUSAGE, i) & 0xffffffffL));
448                 proc.setOpenFiles(WmiUtil.getUint32(processWmiResult, ProcessXPProperty.HANDLECOUNT, i));
449             }
450 
451             // Get a handle to the process for various extended info. Only gets
452             // current user unless running as administrator
453             final HANDLE pHandle = Kernel32.INSTANCE.OpenProcess(WinNT.PROCESS_QUERY_INFORMATION, false,
454                     proc.getProcessID());
455             if (pHandle != null) {
456                 // Full path
457                 final HANDLEByReference phToken = new HANDLEByReference();
458                 try {// EXECUTABLEPATH
459                     proc.setPath(IS_WINDOWS7_OR_GREATER ? Kernel32Util.QueryFullProcessImageName(pHandle, 0)
460                             : WmiUtil.getString(processWmiResult, ProcessXPProperty.EXECUTABLEPATH, i));
461                     if (Advapi32.INSTANCE.OpenProcessToken(pHandle, WinNT.TOKEN_DUPLICATE | WinNT.TOKEN_QUERY,
462                             phToken)) {
463                         Account account = Advapi32Util.getTokenAccount(phToken.getValue());
464                         proc.setUser(account.name);
465                         proc.setUserID(account.sidString);
466                         // Fetching group information incurs ~10ms per process.
467                         if (slowFields) {
468                             Account[] accounts = Advapi32Util.getTokenGroups(phToken.getValue());
469                             // get groups
470                             groupList.clear();
471                             groupIDList.clear();
472                             for (Account a : accounts) {
473                                 groupList.add(a.name);
474                                 groupIDList.add(a.sidString);
475                             }
476                             proc.setGroup(StringUtil.join(",", groupList));
477                             proc.setGroupID(StringUtil.join(",", groupIDList));
478                         }
479                     } else {
480                         int error = Kernel32.INSTANCE.GetLastError();
481                         // Access denied errors are common. Fail silently.
482                         if (error != WinError.ERROR_ACCESS_DENIED) {
483                             LOG.error("Failed to get process token for process {}: {}", proc.getProcessID(),
484                                     Kernel32.INSTANCE.GetLastError());
485                         }
486                     }
487                 } catch (Win32Exception e) {
488                     handleWin32ExceptionOnGetProcessInfo(proc, e);
489                 } finally {
490                     final HANDLE token = phToken.getValue();
491                     if (token != null) {
492                         Kernel32.INSTANCE.CloseHandle(token);
493                     }
494                 }
495             }
496             Kernel32.INSTANCE.CloseHandle(pHandle);
497 
498             // There is no easy way to get ExecutuionState for a process.
499             // The WMI value is null. It's possible to get thread Execution
500             // State and possibly roll up.
501             proc.setState(OSProcess.State.RUNNING);
502 
503             processList.add(proc);
504         }
505         // Clean up memory allocated in C (only Vista+ but null pointer
506         // effectively tests)
507         if (pProcessInfo != null && !Wtsapi32.INSTANCE.WTSFreeMemoryEx(Wtsapi32.WTS_PROCESS_INFO_LEVEL_1, pProcessInfo,
508                 pCount.getValue()))
509 
510         {
511             LOG.error("Failed to Free Memory for Processes. Error code: {}", Kernel32.INSTANCE.GetLastError());
512             return new ArrayList<OSProcess>(0);
513         }
514 
515         // Command Line only accessible via WMI.
516         // Utilize cache to only update new processes
517         Set<Integer> emptyCommandLines = new HashSet<Integer>();
518         for (OSProcess cachedProcess : processList) {
519             // If the process in the cache has an empty command line.
520             if (cachedProcess.getCommandLine().isEmpty()) {
521                 emptyCommandLines.add(cachedProcess.getProcessID());
522             }
523         }
524         if (!emptyCommandLines.isEmpty()) {
525             StringBuilder sb = new StringBuilder(PROCESS_BASE_CLASS);
526             boolean first = true;
527             for (Integer pid : emptyCommandLines) {
528                 if (first) {
529                     sb.append(" WHERE ProcessID=");
530                     first = false;
531                 } else {
532                     sb.append(" OR ProcessID=");
533                 }
534                 sb.append(pid);
535             }
536             PROCESS_QUERY.setWmiClassName(sb.toString());
537             WmiResult<ProcessProperty> commandLineProcs = wmiQueryHandler.queryWMI(PROCESS_QUERY);
538 
539             for (int p = 0; p < commandLineProcs.getResultCount(); p++) {
540                 int pid = WmiUtil.getUint32(commandLineProcs, ProcessProperty.PROCESSID, p);
541                 // This should always be true because emptyCommandLines was
542                 // built from a subset of the cache, but just in case, protect
543                 // against dereferencing null
544                 if (this.processMap.containsKey(pid)) {
545                     OSProcess proc = this.processMap.get(pid);
546                     proc.setCommandLine(WmiUtil.getString(commandLineProcs, ProcessProperty.COMMANDLINE, p));
547                 }
548             }
549         }
550 
551         return processList;
552     }

554     protected void handleWin32ExceptionOnGetProcessInfo(OSProcess proc, Win32Exception ex) {
555         LOG.warn("Failed to set path or get user/group on PID {}. It may have terminated. {}", proc.getProcessID(),
556                 ex.getMessage());
557     }

559     private void updateProcessMapFromRegistry(Collection<Integer> pids) {
560         List<Integer> pidsToKeep = new ArrayList<Integer>();
561 
562         // Grab the PERF_DATA_BLOCK from the registry.
563         // Sequentially increase the buffer until everything fits.
564         IntByReference lpcbData = new IntByReference(this.perfDataBufferSize);
565         Pointer pPerfData = new Memory(this.perfDataBufferSize);
566         int ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
567                 pPerfData, lpcbData);
568         if (ret != WinError.ERROR_SUCCESS && ret != WinError.ERROR_MORE_DATA) {
569             LOG.error("Error {} reading HKEY_PERFORMANCE_DATA from the registry.", ret);
570             return;
571         }
572         while (ret == WinError.ERROR_MORE_DATA) {
573             this.perfDataBufferSize += 4096;
574             lpcbData.setValue(this.perfDataBufferSize);
575             pPerfData = new Memory(this.perfDataBufferSize);
576             ret = Advapi32.INSTANCE.RegQueryValueEx(WinReg.HKEY_PERFORMANCE_DATA, this.processIndexStr, 0, null,
577                     pPerfData, lpcbData);
578         }
579 
580         PERF_DATA_BLOCK perfData = new PERF_DATA_BLOCK(pPerfData.share(0));
581         long perfTime100nSec = perfData.PerfTime100nSec.getValue(); // 1601
582         long now = System.currentTimeMillis(); // 1970 epoch
583 
584         // See format at
585         // https://msdn.microsoft.com/en-us/library/windows/desktop/aa373105(v=vs.85).aspx
586         // [ ] Object Type
587         // [ ][ ][ ] Multiple counter definitions
588         // Then multiple:
589         // [ ] Instance Definition
590         // [ ] Instance name
591         // [ ] Counter Block
592         // [ ][ ][ ] Counter data for each definition above
593 
594         long perfObjectOffset = perfData.HeaderLength;
595 
596         // Iterate object types. For Process should only be one here
597         for (int obj = 0; obj < perfData.NumObjectTypes; obj++) {
598             PERF_OBJECT_TYPE perfObject = new PERF_OBJECT_TYPE(pPerfData.share(perfObjectOffset));
599             // If this isn't the Process object, ignore
600             if (perfObject.ObjectNameTitleIndex == this.processIndex) {
601                 // Skip over counter definitions
602                 // There will be many of these, this points to the first one
603                 long perfInstanceOffset = perfObjectOffset + perfObject.DefinitionLength;
604 
605                 // We need this for every process, initialize outside loop to
606                 // save overhead
607                 PERF_COUNTER_BLOCK perfCounterBlock = null;
608                 // Iterate instances.
609                 // The last instance is _Total so subtract 1 from max
610                 for (int inst = 0; inst < perfObject.NumInstances - 1; inst++) {
611                     PERF_INSTANCE_DEFINITION perfInstance = new PERF_INSTANCE_DEFINITION(
612                             pPerfData.share(perfInstanceOffset));
613                     long perfCounterBlockOffset = perfInstanceOffset + perfInstance.ByteLength;
614 
615                     int pid = pPerfData.getInt(perfCounterBlockOffset + this.idProcessOffset);
616                     if (pids == null || pids.contains(pid)) {
617                         pidsToKeep.add(pid);
618 
619                         // If process exists fetch from cache
620                         OSProcess proc = null;
621                         if (this.processMap.containsKey(pid)) {
622                             proc = this.processMap.get(pid);
623                         }
624                         // If not in cache or if start time differs too much,
625                         // create new process to add/replace cache value
626                         long upTime = (perfTime100nSec
627                                 - pPerfData.getLong(perfCounterBlockOffset + this.elapsedTimeOffset)) / 10000L;
628                         long startTime = now - upTime;
629                         if (proc == null || Math.abs(startTime - proc.getStartTime()) > 200) {
630                             proc = new OSProcess();
631                             proc.setProcessID(pid);
632                             proc.setStartTime(startTime);
633                             proc.setName(pPerfData.getWideString(perfInstanceOffset + perfInstance.NameOffset));
634                             // Adds or replaces previous
635                             this.processMap.put(pid, proc);
636                         }
637 
638                         // Update stats
639                         proc.setUpTime(upTime < 1L ? 1L : upTime);
640                         proc.setBytesRead(pPerfData.getLong(perfCounterBlockOffset + this.ioReadOffset));
641                         proc.setBytesWritten(pPerfData.getLong(perfCounterBlockOffset + this.ioWriteOffset));
642                         proc.setResidentSetSize(
643                                 pPerfData.getLong(perfCounterBlockOffset + this.workingSetPrivateOffset));
644                         proc.setParentProcessID(
645                                 pPerfData.getInt(perfCounterBlockOffset + this.creatingProcessIdOffset));
646                         proc.setPriority(pPerfData.getInt(perfCounterBlockOffset + this.priorityBaseOffset));
647                     }
648 
649                     // Increment to next instance
650                     perfCounterBlock = new PERF_COUNTER_BLOCK(pPerfData.share(perfCounterBlockOffset));
651                     perfInstanceOffset = perfCounterBlockOffset + perfCounterBlock.ByteLength;
652                 }
653                 // We've found the process object and are done, no need to look
654                 // at any other objects (shouldn't be any). Break the loop
655                 break;
656             }
657             // Increment for next object (should never need this)
658             perfObjectOffset += perfObject.TotalByteLength;
659         }
660         // If this was a full update, delete any pid we didn't find from the
661         // cache.
662         if (pids == null) {
663             for (Integer pid : new HashSet<Integer>(this.processMap.keySet())) {
664                 if (!pidsToKeep.contains(pid)) {
665                     this.processMap.remove(pid);
666                 }
667             }
668         }
669     }

713     /**
714      * Enables debug privileges for this process, required for OpenProcess() to get
715      * processes other than the current user
716      */
717     private static void enableDebugPrivilege() {
718         HANDLEByReference hToken = new HANDLEByReference();
719         boolean success = Advapi32.INSTANCE.OpenProcessToken(Kernel32.INSTANCE.GetCurrentProcess(),
720                 WinNT.TOKEN_QUERY | WinNT.TOKEN_ADJUST_PRIVILEGES, hToken);
721         if (!success) {
722             LOG.error("OpenProcessToken failed. Error: {}", Native.getLastError());
723             return;
724         }
725         WinNT.LUID luid = new WinNT.LUID();
726         success = Advapi32.INSTANCE.LookupPrivilegeValue(null, WinNT.SE_DEBUG_NAME, luid);
727         if (!success) {
728             LOG.error("LookupprivilegeValue failed. Error: {}", Native.getLastError());
729             Kernel32.INSTANCE.CloseHandle(hToken.getValue());
730             return;
731         }
732         WinNT.TOKEN_PRIVILEGES tkp = new WinNT.TOKEN_PRIVILEGES(1);
733         tkp.Privileges[0] = new WinNT.LUID_AND_ATTRIBUTES(luid, new DWORD(WinNT.SE_PRIVILEGE_ENABLED));
734         success = Advapi32.INSTANCE.AdjustTokenPrivileges(hToken.getValue(), false, tkp, 0, null, null);
735         if (!success) {
736             LOG.error("AdjustTokenPrivileges failed. Error: {}", Native.getLastError());
737         }
738         Kernel32.INSTANCE.CloseHandle(hToken.getValue());
739     }
