44  public class WindowsOSVersionInfoEx extends AbstractOSVersionInfoEx {
45  
46      private static final long serialVersionUID = 1L;
47  
48      private static final Logger LOG = LoggerFactory.getLogger(WindowsOSVersionInfoEx.class);
49  
50      enum OSVersionProperty {
51          VERSION, PRODUCTTYPE, BUILDNUMBER, CSDVERSION, SUITEMASK;
52      }
53  
54      public WindowsOSVersionInfoEx() {
55          init();
56      }
57  
58      private void init() {
59          // Populate a key-value map from WMI
60          WmiQuery<OSVersionProperty> osVersionQuery = new WmiQuery<OSVersionProperty>("Win32_OperatingSystem",
61                  OSVersionProperty.class);
62          WmiResult<OSVersionProperty> versionInfo = WmiQueryHandler.createInstance().queryWMI(osVersionQuery);
63          if (versionInfo.getResultCount() < 1) {
64              handleNoVersionInfo();
65          } else {
66              // Guaranteed that versionInfo is not null and lists non-empty
67              // before calling the parse*() methods
68              int suiteMask = WmiUtil.getUint32(versionInfo, OSVersionProperty.SUITEMASK, 0);
69              setBuildNumber(WmiUtil.getString(versionInfo, OSVersionProperty.BUILDNUMBER, 0));
70              setVersion(parseVersion(versionInfo, suiteMask));
71              setCodeName(parseCodeName(suiteMask));
72              LOG.debug("Initialized OSVersionInfoEx");
73          }
74      }
75  
76      protected void handleNoVersionInfo() {
77          LOG.warn("No version data available.");
78          setVersion(System.getProperty("os.version"));
79          setCodeName("");
80          setBuildNumber("");
81      }
82  
83      /**
84       * Gets the operating system version
85       *
86       * @param suiteMask
87       *
88       * @return Version
89       */
90      private String parseVersion(WmiResult<OSVersionProperty> versionInfo, int suiteMask) {
91  
92          // Initialize a default, sane value
93          String version = System.getProperty("os.version");
94  
95          // Version is major.minor.build. Parse the version string for
96          // major/minor and get the build number separately
97          String[] verSplit = WmiUtil.getString(versionInfo, OSVersionProperty.VERSION, 0).split("\\D");
98          int major = verSplit.length > 0 ? ParseUtil.parseIntOrDefault(verSplit[0], 0) : 0;
99          int minor = verSplit.length > 1 ? ParseUtil.parseIntOrDefault(verSplit[1], 0) : 0;
100 
101         // see
102         // http://msdn.microsoft.com/en-us/library/windows/desktop/ms724833%28v=vs.85%29.aspx
103         boolean ntWorkstation = WmiUtil.getUint32(versionInfo, OSVersionProperty.PRODUCTTYPE,
104                 0) == WinNT.VER_NT_WORKSTATION;
105         switch (major) {
106         case 10:
107             if (minor == 0) {
108                 if (ntWorkstation) {
109                     version = "10";
110                 } else {
111                     // Build numbers greater than 17762 is Server 2019 for OS
112                     // Version 10.0
113                     version = (ParseUtil.parseLongOrDefault(getBuildNumber(), 0L) > 17762) ? "Server 2019"
114                             : "Server 2016";
115                 }
116             }
117             break;
118         case 6:
119             if (minor == 3) {
120                 version = ntWorkstation ? "8.1" : "Server 2012 R2";
121             } else if (minor == 2) {
122                 version = ntWorkstation ? "8" : "Server 2012";
123             } else if (minor == 1) {
124                 version = ntWorkstation ? "7" : "Server 2008 R2";
125             } else if (minor == 0) {
126                 version = ntWorkstation ? "Vista" : "Server 2008";
127             }
128             break;
129         case 5:
130             if (minor == 2) {
131                 if ((suiteMask & 0x00008000) != 0) {// VER_SUITE_WH_SERVER
132                     version = "Home Server";
133                 } else if (ntWorkstation) {
134                     version = "XP"; // 64 bits
135                 } else {
136                     version = User32.INSTANCE.GetSystemMetrics(WinUser.SM_SERVERR2) != 0 ? "Server 2003"
137                             : "Server 2003 R2";
138                 }
139             } else if (minor == 1) {
140                 version = "XP"; // 32 bits
141             } else if (minor == 0) {
142                 version = "2000";
143             }
144             break;
145         default:
146             break;
147         }
148 
149         String sp = WmiUtil.getString(versionInfo, OSVersionProperty.CSDVERSION, 0);
150         if (!sp.isEmpty() && !"unknown".equals(sp)) {
151             version = version + " " + sp.replace("Service Pack ", "SP");
152         }
153 
154         return version;
155     }
156 
157     /**
158      * Gets suites available on the system and return as a codename
159      *
160      * @param suiteMask
161      *
162      * @return Suites
163      */
164     private String parseCodeName(int suiteMask) {
165         List<String> suites = new ArrayList<String>();
166         if ((suiteMask & 0x00000002) != 0) {
167             suites.add("Enterprise");
168         }
169         if ((suiteMask & 0x00000004) != 0) {
170             suites.add("BackOffice");
171         }
172         if ((suiteMask & 0x00000008) != 0) {
173             suites.add("Communication Server");
174         }
175         if ((suiteMask & 0x00000080) != 0) {
176             suites.add("Datacenter");
177         }
178         if ((suiteMask & 0x00000200) != 0) {
179             suites.add("Home");
180         }
181         if ((suiteMask & 0x00000400) != 0) {
182             suites.add("Web Server");
183         }
184         if ((suiteMask & 0x00002000) != 0) {
185             suites.add("Storage Server");
186         }
187         if ((suiteMask & 0x00004000) != 0) {
188             suites.add("Compute Cluster");
189         }
190         // 0x8000, Home Server, is included in main version name
191         return StringUtil.join(",", suites);
192     }
