48  /**
49   * The Linux File System contains {@link OSFileStore}s which are a storage pool,
50   * device, partition, volume, concrete file system or other implementation
51   * specific means of file storage. In Linux, these are found in the /proc/mount
52   * filesystem, excluding temporary and kernel mounts.
53   *
54   * @author widdis[at]gmail[dot]com
55   */
56  public class LinuxFileSystem implements FileSystem {
57  
58      private static final long serialVersionUID = 1L;
59  
60      private static final Logger LOG = LoggerFactory.getLogger(LinuxFileSystem.class);

62      // Linux defines a set of virtual file systems
63      private final List<String> pseudofs = Arrays.asList(new String[] { //
64              "rootfs", // Minimal fs to support kernel boot
65              "sysfs", // SysFS file system
66              "proc", // Proc file system
67              "devtmpfs", // Dev temporary file system
68              "devpts", // Dev pseudo terminal devices file system
69              "securityfs", // Kernel security file system
70              "cgroup", // Cgroup file system
71              "pstore", // Pstore file system
72              "hugetlbfs", // Huge pages support file system
73              "configfs", // Config file system
74              "selinuxfs", // SELinux file system
75              "systemd-1", // Systemd file system
76              "binfmt_misc", // Binary format support file system
77              "mqueue", // Message queue file system
78              "debugfs", // Debug file system
79              "nfsd", // NFS file system
80              "sunrpc", // Sun RPC file system
81              "rpc_pipefs", // Sun RPC file system
82              "fusectl", // FUSE control file system
83              // NOTE: FUSE's fuseblk is not evaluated because used as file system
84              // representation of a FUSE block storage
85              // "fuseblk" // FUSE block file system
86              // "tmpfs", // Temporary file system
87              // NOTE: tmpfs is evaluated apart, because Linux uses it for
88              // RAMdisks
89              "overlay", // Overlay file system
90                         // https://wiki.archlinux.org/index.php/Overlay_filesystem
91      });
92  
93      // System path mounted as tmpfs
94      private final List<String> tmpfsPaths = Arrays.asList(new String[] { "/run", "/sys", "/proc" });
95  
96      /**
97       * Checks if file path equals or starts with an element in the given list
98       *
99       * @param aList
100      *            A list of path prefixes
101      * @param charSeq
102      *            a path to check
103      * @return true if the charSeq exactly equals, or starts with the directory in
104      *         aList
105      */
106     private boolean listElementStartsWith(List<String> aList, String charSeq) {
107         for (String match : aList) {
108             if (charSeq.equals(match) || charSeq.startsWith(match + "/")) {
109                 return true;
110             }
111         }
112         return false;
113     }

115     /**
116      * Gets File System Information.
117      *
118      * @return An array of {@link OSFileStore} objects representing mounted volumes.
119      *         May return disconnected volumes with
120      *         {@link OSFileStore#getTotalSpace()} = 0.
121      */
122     @Override
123     public OSFileStore[] getFileStores() {
124         // Map uuids with device path as key
125         Map<String, String> uuidMap = new HashMap<String, String>();
126         File uuidDir = new File("/dev/disk/by-uuid");
127         if (uuidDir.listFiles() != null) {
128             for (File uuid : uuidDir.listFiles()) {
129                 try {
130                     // Store UUID as value with path (e.g., /dev/sda1) as key
131                     uuidMap.put(uuid.getCanonicalPath(), uuid.getName().toLowerCase());
132                 } catch (IOException e) {
133                     LOG.error("Couldn't get canonical path for {}. {}", uuid.getName(), e);
134                 }
135             }
136         }
137 
138         // List file systems
139         List<OSFileStore> fsList = new ArrayList<OSFileStore>();
140 
141         // Parse /proc/self/mounts to get fs types
142         List<String> mounts = FileUtil.readFile("/proc/self/mounts");
143         for (String mount : mounts) {
144             String[] split = mount.split(" ");
145             // As reported in fstab(5) manpage, struct is:
146             // 1st field is volume name
147             // 2nd field is path with spaces escaped as \040
148             // 3rd field is fs type
149             // 4th field is mount options (ignored)
150             // 5th field is used by dump(8) (ignored)
151             // 6th field is fsck order (ignored)
152             if (split.length < 6) {
153                 continue;
154             }
155 
156             // Exclude pseudo file systems
157             String path = split[1].replaceAll("\\\\040", " ");
158             String type = split[2];
159             if (this.pseudofs.contains(type) // exclude non-fs types
160                     || path.equals("/dev") // exclude plain dev directory
161                     || listElementStartsWith(this.tmpfsPaths, path) || path.endsWith("/shm") // exclude shared memory
162             ) {
163                 continue;
164             }
165 
166             String name = split[0].replaceAll("\\\\040", " ");
167             if (path.equals("/")) {
168                 name = "/";
169             }
170             String volume = split[0].replaceAll("\\\\040", " ");
171             String uuid = MapUtil.getOrDefault(uuidMap, split[0], "");
172 
173             String description;
174             if (volume.startsWith("/dev")) {
175                 description = "Local Disk";
176             } else if (volume.equals("tmpfs")) {
177                 description = "Ram Disk";
178             } else if (type.startsWith("nfs") || type.equals("cifs")) {
179                 description = "Network Disk";
180             } else {
181                 description = "Mount Point";
182             }
183 
184             // Add in logical volume found at /dev/mapper, useful when linking
185             // file system with drive.
186             String logicalVolume = "";
187             // Attempt to read the symbolic link.
188             // If it's not a symbolic link we'll fail with EINVAL and ignore
189             Memory buf = new Memory(1024);
190             buf.clear();
191             int size = Libc.INSTANCE.readlink(volume, buf, 1023);
192             if (size > 0) {
193                 logicalVolume = buf.getString(0).substring(0, size);
194             }
195 
196             long totalInodes = 0L;
197             long freeInodes = 0L;
198             long totalSpace = 0L;
199             long usableSpace = 0L;
200 
201             try {
202                 LibC.Statvfs vfsStat = new LibC.Statvfs();
203                 if (0 == LibC.INSTANCE.statvfs(path, vfsStat)) {
204                     totalInodes = vfsStat.f_files.longValue();
205                     freeInodes = vfsStat.f_ffree.longValue();
206                     totalSpace = vfsStat.f_blocks.longValue() * vfsStat.f_bsize.longValue();
207                     usableSpace = vfsStat.f_bfree.longValue() * vfsStat.f_bsize.longValue();
208                 } else {
209                     File tmpFile = new File(path);
210                     totalSpace = tmpFile.getTotalSpace();
211                     usableSpace = tmpFile.getUsableSpace();
212                     LOG.warn("Failed to get information to use statvfs. path: {}, Error code: {}", path,
213                             Native.getLastError());
214                 }
215             } catch (UnsatisfiedLinkError e) {
216                 LOG.error("Failed to get file counts from statvfs.", e);
217             } catch (NoClassDefFoundError e) {
218                 LOG.error("Failed to get file counts from statvfs.", e);
219             }
220 
221             OSFileStoreOSFileStore">OSFileStore osStore = new OSFileStore();
222             osStore.setName(name);
223             osStore.setVolume(volume);
224             osStore.setMount(path);
225             osStore.setDescription(description);
226             osStore.setType(type);
227             osStore.setUUID(uuid);
228             osStore.setUsableSpace(usableSpace);
229             osStore.setTotalSpace(totalSpace);
230             osStore.setFreeInodes(freeInodes);
231             osStore.setTotalInodes(totalInodes);
232             osStore.setLogicalVolume(logicalVolume);
233 
234             fsList.add(osStore);
235         }
236 
237         return fsList.toArray(new OSFileStore[fsList.size()]);
238     }

240     @Override
241     public long getOpenFileDescriptors() {
242         return getFileDescriptors(0);
243     }
244 
245     @Override
246     public long getMaxFileDescriptors() {
247         return getFileDescriptors(2);
248     }

250     /**
251      * Returns a value from the Linux system file /proc/sys/fs/file-nr.
252      *
253      * @param index
254      *            The index of the value to retrieve. 0 returns the total allocated
255      *            file descriptors. 1 returns the number of used file descriptors
256      *            for kernel 2.4, or the number of unused file descriptors for
257      *            kernel 2.6. 2 returns the maximum number of file descriptors that
258      *            can be allocated.
259      * @return Corresponding file descriptor value from the Linux system file.
260      */
261     private long getFileDescriptors(int index) {
262         String filename = "/proc/sys/fs/file-nr";
263         if (index < 0 || index > 2) {
264             throw new IllegalArgumentException("Index must be between 0 and 2.");
265         }
266         List<String> osDescriptors = FileUtil.readFile(filename);
267         if (!osDescriptors.isEmpty()) {
268             String[] splittedLine = osDescriptors.get(0).split("\\D+");
269             return ParseUtil.parseLongOrDefault(splittedLine[index], 0L);
270         }
271         return 0L;
272     }
